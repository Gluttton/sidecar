79 bool
80 CPIAlgorithm::processInput(const Messages::PRIMessage::Ref& msg)
81 {
82  static Logger::ProcLog log("processInput", getLog());
83  LOGINFO << std::endl;
84 
85  bool rc = true;
86  LOGDEBUG << "process msg: " << msg->getRIUInfo().sequenceCounter << std::endl;
87 
88  uint32_t prf_code = msg->getRIUInfo().prfEncoding;
89  size_t cpiSpan = cpiSpan_->getValue();
90 
91  // Handle special case: the arrival of the very first message
92  //
93  if (! current_prf_code_) current_prf_code_ = prf_code;
94  size_t num_msgs = buffer_.size();
95 
96  // Look for a change in the prf encoding. This indicates the arrival of the first PRI message of a new CPI.
97  //
98  if (prf_code != current_prf_code_) {
99 
100  beginProcessingCPI_ = Time::TimeStamp::Now();
101  bool missing = false;
102  if (dropIncompleteCPI_->getValue()) {
103  uint32_t lastSeq = (*buffer_.begin())->getSequenceCounter();
104  for (MessageQueue::iterator itr = buffer_.begin() + 1; itr != buffer_.end() && !missing; ++itr) {
105  missing = missing || ((*itr)->getSequenceCounter() - lastSeq != 1);
106  lastSeq = (*itr)->getSequenceCounter();
107  }
108  }
109 
110  if (! missing) rc = processCPI();
111 
112  maxMsgSize_ = 0;
113  buffer_.clear();
114  current_prf_code_ = prf_code;
115 
116  // Delta represents the amount of time it took to process the CPI, since our stats are reported in PRI
117  // messages / unit time, need to distribute time costs across number of PRIs forming the CPI.
118  //
119  Time::TimeStamp delta = Time::TimeStamp::Now() - beginProcessingCPI_;
120  delta *= double(1.0 / cpiSpan);
121 
122  // Add a stat sample for each PRI message in the CPI because data rates are recorded at the PRI message
123  // rate.
124  //
125  for (size_t i = 0; i < cpiSpan; i++) getController().addProcessingStatSample(delta);
126  }
127 
128  // Buffer PRI messages belonging to the current CPI until the first message of the next CPI arrives.
129  //
130  buffer_.push_back(msg);
131  if (msg->size() > maxMsgSize_) maxMsgSize_ = msg->size();
132 
133  return rc;
134 }
