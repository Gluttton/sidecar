46 // This routine is responsible for taking a set of PRIs and computing an encoded RGB value based on the Doppler-range
47 // values composing this CPI.
48 //
49 bool
50 RGBConverter::processCPI()
51 {
52  static Logger::ProcLog log("processCPI", getLog());
53  LOGINFO << std::endl;
54  LOGDEBUG << "Process CPI with " << buffer_.size() << " msgs" << std::endl;
55  size_t cpiSpan = cpiSpan_->getValue();
56 
57  if (buffer_.size() != cpiSpan) {
58  LOGWARNING << "Kicking an incomplete CPI! Expected " << cpiSpan << ", but received " << buffer_.size()
59  << std::endl;
60  return true;
61  }
62 
63  std::deque< Messages::Video::Ref > msg_buffer_;
64  for(size_t i = 0; i < buffer_.size(); i++) {
65  msg_buffer_.push_back(boost::dynamic_pointer_cast<Messages::Video>(buffer_[i]));
66  }
67 
68  size_t msg_size = msg_buffer_.front()->size();
69 
70  // Build the output message
71  //
72  Messages::Video::Ref out(Messages::Video::Make(getName(), msg_buffer_[0]));
73  Messages::Video::Container& outputData(out->getData());
74 
75  // Compute the RGB values for each range across the CPI
76  //
77  double r, g, b;
78 
79  // Get the normalization range parameters
80  //
81  int min = min_->getValue();
82  int max = max_->getValue();
83  int span = max - min;
84 
85  // Compute where to center the windows for the RGB calculations
86  //
87  int K = int(::ceil(cpiSpan / 3));
88  int R = int(::floor(K / 2));
89  int len = 2*R + 1;
90 
91  uint16_t encoded;
92  uint16_t r_short;
93  uint16_t g_short;
94  uint16_t b_short;
95 
96  // Pivot points for RGB values are 0, 1/3, 2/3 of cpiSpan
97  //
98  int pR = K;
99  int pG = 0;
100  int pB = int(::ceil(2.0*cpiSpan / 3.0));
101 
102  bool rc = true;
103 
104  // Compute the summations
105  //
106  for (size_t i = 0; i < msg_size; i++) {
107 
108  r = g = b = 0.0;
109 
110  // Handle special case where pivot point is zero and thus wraps around the buffer
111  //
112  g = msg_buffer_[pG][i];
113  for(int j = 0; j < R; j++) {
114  g += msg_buffer_[cpiSpan - 1 - j][i];
115  g += msg_buffer_[1 + j][i];
116  }
117 
118  // Compute summations for other R and B windows
119  //
120  for(int j = 0; j < len; j++) {
121  r += msg_buffer_[pR - R + j][i];
122  b += msg_buffer_[pB - R + j][i];
123  }
124 
125  // Compute the average Doppler values
126  //
127  r /= len;
128  g /= len;
129  b /= len;
130 
131  // Convert to dB
132  //
133  r = 20.0 * ::log10(r);
134  g = 20.0 * ::log10(g);
135  b = 20.0 * ::log10(b);
136 
137  // Normalize the values, capping vals that fall below/above the min/max user-specified values
138  //
139  r -= min;
140  g -= min;
141  b -= min;
142 
143  r /= span;
144  g /= span;
145  b /= span;
146 
147  r = (r < 0.0) ? 0.0 : r;
148  g = (g < 0.0) ? 0.0 : g;
149  b = (b < 0.0) ? 0.0 : b;
150 
151  r = (r > 1.0) ? 1.0 : r;
152  g = (g > 1.0) ? 1.0 : g;
153  b = (b > 1.0) ? 1.0 : b;
154 
155  // Encode the normalized values into RGB (655) color scheme where R gets the 6 most significant bits of a a
156  // 16-bit number, then G gets the next 5 most significant bits, and then B gets the last 5 bits
157  //
158  r_short = uint16_t(::round(r * 63));
159  g_short = uint16_t(::round(g * 31));
160  b_short = uint16_t(::round(b * 31));
161  encoded = uint16_t(::rint((r_short << 10) | (g_short << 5) | b_short));
162 
163  // Push data into the output buffer
164  //
165  outputData.push_back(Messages::Video::DatumType(encoded));
166  }
167 
168  msg_buffer_.clear();
169 
170  // Send the output message out
171  //
172  rc = send(out);
173 
174  return rc;
175 }
