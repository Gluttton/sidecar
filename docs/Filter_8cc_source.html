222 void
223 Filter::resizePriData()
224 {
225  static Logger::ProcLog log("resizePriData", getLog());
226 
227  if (!m_maxRangeBinValid) return;
228 
229  //the m_priData should always be of size rows x cols (as follows)
230 #ifndef USE_FAST_CONVOLUTION
231  PRI_COUNT rows = (PRI_COUNT)m_wrappedKernels[0]->size(0);
232 #else
233  PRI_COUNT rows = (PRI_COUNT)m_kernel->size(0);
234 #endif
235 
236  RANGEBIN cols = m_maxRangeBin;
237 
238  if (!m_priData) {
239  LOGDEBUG << "creating data (" << rows << "x" << cols << ")"
240  << std::endl;
241  m_priData = DataMatrixPtr(new DataMatrix(rows, cols, 0));
242  //LOGDEBUG << (*m_priData) << std::endl;
243 
244  }
245  else if ((rows != (PRI_COUNT)m_priData->size(0))) {
246  //if the number of required rows has changed, then the kernel has
247  // changed just discard all data
248  LOGDEBUG << "resizing the number of rows (" << rows << "x" << cols
249  << ")" << std::endl;
250  m_priData = DataMatrixPtr(new DataMatrix(rows, cols, 0));
251 
252  }
253  else if ((cols != (RANGEBIN)m_priData->size(1))) {
254  //if only the columns are mismatched, copy the data (the number of cols
255  // required can change as the pri length changes -- e.g., as the
256  // result of PRF stagger)
257  LOGDEBUG << "resizing the number of cols (" << rows << "x" << cols
258  << ")" << std::endl;
259  DataMatrixPtr newData (new DataMatrix(rows, cols, 0));
260  RANGEBIN copyCols = std::min(cols, (RANGEBIN)m_priData->size(1));
261  (*newData)(D2(D1(rows), D1(copyCols))) =
262  (*m_priData)(D2(D1(rows), D1(copyCols)));
263  m_priData = newData;
264  }
265 
266 #ifndef USE_FAST_CONVOLUTION
267  //resize the scratch matrix as appropriate
268  vsip::length_type outputSize = m_priData->size(1) -
269  m_wrappedKernels[0]->size(1) + 1;
270  m_scratch = DataMatrixPtr (new DataMatrix (1, outputSize));
271 #else
272  //initialize the fft and ifft structures. also, pre-calcualte the
273  // fft of the kernel.
274  LOGDEBUG << "initializing the fft...";
275  m_fft = FFTPtr (new FFT(D2(D1(rows), D1(cols)), 1.0f));
276 
277  LOGDEBUG << "done" << std::endl << "initializing the inverse fft...";
278  m_ifft = IFFTPtr(new IFFT(D2(D1(rows), D1(cols)),
279  1.0f/(FFTDATA)(rows*cols)));
280 
281  DataMatrix zeroPaddedKernel (rows, cols, COMPLEXFFTDATA(0.0f, 0.0f));
282  zeroPaddedKernel(D2(D1(m_kernel->size(0)), D1(m_kernel->size(1)))) =
283  *m_kernel;
284 
285  m_fftKernel = DataMatrixPtr(new DataMatrix(rows,cols));
286  m_scratch = DataMatrixPtr(new DataMatrix(rows,cols));
287 
288  LOGDEBUG << "done" << std::endl << "calculating the fft of the kernel...";
289  (*m_fft)(zeroPaddedKernel, *m_fftKernel);
290 
291  LOGDEBUG << "done" << std::endl;
292 
293  //LOGDEBUG << "kernel" << std::endl << zeroPaddedKernel << std::endl
294  // << "fft(kernel)" << std::endl << *m_fftKernel << std::endl;
295 #endif

328 void
329 Filter::doInternalSetup(DataMatrixPtr kernel)
330 {
331  static Logger::ProcLog log("createWrappedKernels", getLog());
332 
333 #ifndef USE_FAST_CONVOLUTION
334  //delete any existing kernels
336  m_wrappedKernels.clear();
337 
338  //add the original kernel to the list
339  m_wrappedKernels.push_back(kernel);
340 
341  for (vsip::index_type i = 1; i < kernel->size(0); i++) {
342  //rotate the kernel by one row
343  DataMatrixPtr rotated1Row(new DataMatrix(kernel->size(0),
344  kernel->size(1)));
345  //assume rotated1Row and kernel have dimensions [0:M] x [0:N]
346  //rotated1Row(0:M-1,:) = kernel(1:M,:)
347  (*rotated1Row)(D2(D1(0,1,kernel->size(0)-1), D1(kernel->size(1)))) =
348  (*kernel) (D2(D1(1,1,kernel->size(0)-1), D1(kernel->size(1))));
349  //rotated1Row(M,:) = kernel(0,:)
350  (*rotated1Row)(D2(D1(kernel->size(0)-1,1,1), D1(kernel->size(1)))) =
351  (*kernel) (D2(D1(0,1,1), D1(kernel->size(1))));
352 
353  m_wrappedKernels.push_back(rotated1Row);
354  kernel = rotated1Row;
355  }
356 
357  for (int j = 0; j < (int)m_wrappedKernels.size(); j++) {
358  LOGDEBUG << "kernel #" << j << std::endl
359  << *(m_wrappedKernels[j]) << std::endl;
360  }
362 #else
363  //just save the kernel, nothing to do
365  m_kernel = kernel;
366 
368 #endif
369 }
