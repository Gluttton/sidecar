103 RawVideo::convert(const std::string& producer) const
104 {
105  static Logger::ProcLog log("convert", Log());
106  LOGINFO << std::endl;
107 
108  // ACE_CDR_BYTE_ORDER == 0 for PowerPC (big-endian) ACE_CDR_BYTE_ORDER == 1 for Intel (little-endian)
109  //
110  if (! raw_) {
111  LOGERROR << "NULL raw_ attribute" << std::endl;
112  return Video::Ref();
113  }
114 
115  // Fetch the raw msgDesc value. It is always in network-byte order.
116  //
117  VMEDataMessage* rawPtr =
118  reinterpret_cast<VMEDataMessage*>(raw_->rd_ptr());
119  uint32_t msgDesc = ntohl(rawPtr->header.msgDesc);
120  bool isLittleEndian = (msgDesc & VMEHeader::kEndianessMask);
121 
122  // Create an input CDR stream to fetch binary data from. Handles the byte-swapping for us when necessary.
123  //
124  ACE_InputCDR cdr(raw_, isLittleEndian ? 1 : 0);
125 
126  // Extract the VME message header, but since the msgDesc is always in network byte-order (big-endian), we
127  // revert to the raw value, regardless of endianess.
128  //
129  VMEDataMessage vme;
130  vme.load(cdr);
131  vme.header.msgDesc = msgDesc;
132  LOGDEBUG << vme << std::endl;
133 
134  // Adjust the count if we have both I and Q samples.
135  //
136  size_t count = vme.numSamples;
137  bool isComplex = vme.header.getFormat() == VMEHeader::kPackedIQ;
138  if (isComplex)
139  count *= 2;
140 
141  // Make sure we have enough bytes in the input CDR stream. If not, then process what we can, but raise a
142  // stink about it. Most likely, we have a endianess or format issue with the VME code.
143  //
144  size_t available = cdr.length() / 2;
145  if (count > available) {
146  LOGERROR << vme << std::hex << msgDesc << std::dec << " numSamples ("
147  << count << ") != available (" << available << ")"
148  << std::endl;
149  count = available;
150  LOGDEBUG << "reduced count: " << count << std::endl;
151  }
152 
153  LOGDEBUG << "count: " << count << std::endl;
154 
155  if (! isComplex) {
156  Video::Ref ref(Video::Make(producer, vme, count));
157 
158  // The VME system will emit data in two formats: inverted * 4 integers, and normal integers. The
159  // inverted * 4 values require a bit-wise inversion followed by a right shift of 2 bits (dividing by 4).
160  // The other format simply requires us to take the values from the ACE CDR, which handles any
161  // byte-swapping that may be necessary. NOTE: the VME also emits kPackedReal data that is NOT inverted *
162  // 4; thus the additional check for the data being in little-endian format.
163  //
164  if ((vme.header.getFormat() == VMEHeader::kPackedReal &&
165  isLittleEndian) ||
166  vme.header.getFormat() == VMEHeader::kUnpackedReal) {
167 
168  LOGDEBUG << "handling kPackedReal/kUnpackedReal" << std::endl;
169  Video::DatumType value;
170  Video::Container& data(ref->getData());
171  while (count--) {
172  cdr >> value;
173  data.push_back((~value) >> 2);
174  }
175  }
176  else {
177  ref->resize(count);
178  cdr.read_short_array(&ref[0], count);
179  }
180  return ref;
181  }
182  else {
183  Video::Ref ref(Video::Make(producer, vme, count));
184  ref->resize(count);
185  cdr.read_short_array(reinterpret_cast<int16_t*>(&ref[0]), count);
186  return ref;
187  }
