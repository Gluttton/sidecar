75  ACE_INET_Addr addr;
76  addr.set(port_, service->getHost().toStdString().c_str(), 1, PF_INET);
77  if (! reader_.join(addr)) {
78  LOGERROR << "failed to open multicast socket - " << Utils::showErrno()
79  << std::endl;
80  return false;
81  }
82 
83  // Create a proxy object that we can connect slots to.
84  //
85  Q_ASSERT(proxy_ == 0);
86  proxy_ = new QUdpSocket(this);
87  LOGDEBUG << "reader socket: " << reader_.getDevice().get_handle()
88  << std::endl;
89  if (! proxy_->setSocketDescriptor(reader_.getDevice().get_handle())) {
90  LOGERROR << "failed setSocketDescriptor - "
91  << proxy_->error() << ' ' << proxy_->errorString()
92  << std::endl;
93  return false;
94  }
95 
96  connect(proxy_, SIGNAL(readyRead()), SLOT(socketReadyRead()));
97  connect(proxy_, SIGNAL(error(QAbstractSocket::SocketError)),
98  SLOT(socketError(QAbstractSocket::SocketError)));
99 
100  heartBeatWriter_ = new QUdpSocket(this);
101  if (! heartBeatWriter_->bind()) {
102  LOGERROR << "failed heartBeatWriter_->bind()" << std::endl;
103  }
104 
105  heartBeatHost_.clear();
106  QHostInfo hostInfo = QHostInfo::fromName(service->getNativeHost());
107  if (hostInfo.error() != QHostInfo::NoError) {
108  LOGERROR << "failed QHostInfo::fromName() with "
109  << service->getNativeHost() << std::endl;
110  heartBeatHost_ = "127.0.0.1";
111  }
112  else {
113  foreach (QHostAddress address, hostInfo.addresses()) {
114  if (address.protocol() == QAbstractSocket::IPv4Protocol) {
115  heartBeatHost_ = address;
116  break;
117  }
118  }
119  }
120

122  heartBeatPort_ = portText.toUShort();
123  LOGDEBUG << "heartBeatHost: " << heartBeatHost_.toString()
124  << " port: " << heartBeatPort_ << std::endl;

155 {
156  static const qint64 kMaxSize = 64 * 1024 - 1;
157  static Logger::ProcLog log("socketReadyRead", Log());
158  LOGINFO << std::endl;
159 
160  ACE_Message_Block* raw = 0;
161  do {
162 
163  qint64 size = proxy_->pendingDatagramSize();
164  if (size == -1)
165  continue;
166 
167  if (size > kMaxSize) {
168  LOGERROR << this << " invalid size from pendingDatagramSize - "
169  << size << std::endl;
170  continue;
171  }
172 
173  if (! raw || raw->size() < size_t(size)) {
174  if (raw) raw->release();
175  raw = MessageManager::MakeMessageBlock(size);
176  }
177 
178  QHostAddress host;
179  quint16 port;
180  size = proxy_->readDatagram(raw->wr_ptr(), size, &host, &port);
181  if (size == -1) {
182  LOGERROR << "failed to fetch data" << std::endl;
183  continue;
184  }
185 
186  if (port_ != port) {
187  LOGERROR << "wrong port value: " << port_ << " != " << port
188  << std::endl;
189  continue;
190  }
191 
192  if (! connected_) {
193  connected_ = true;
194  emit connected();
195  }
196 
197  raw->wr_ptr(size);
198  addRawData(raw);
199  raw = 0;
200 
201  } while (proxy_->hasPendingDatagrams());
202 
203  if (raw)
204  raw->release();
