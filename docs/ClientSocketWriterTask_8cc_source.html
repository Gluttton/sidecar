113 int
114 ClientSocketWriterTask::OutputHandler::svc()
115 {
116  Logger::ProcLog log("OutputHandler::svc", Log());
117  LOGINFO << std::endl;
118 
119  ACE_Message_Block* data = 0;
120 
121  for (;;) {
122 
123  if (data == 0) {
124 
125  // If we fail here, it could be that the message queue is in a PULSED state, which indicates that
126  // the connection to the server has gone down. This will continue until our open() method is invoked
127  // after a new connection has been established.
128  //
129  if (getq(data) == -1) {
130  if (msg_queue()->state() == ACE_Message_Queue_Base::PULSED) {
131  if (! connector_->reconnect()) break;
132  }
133  else {
134  break;
135  }
136  continue;
137  }
138  }
139  else {
140 
141  MessageManager manager(data);
142  if (manager.hasNative()) {
143 
144  // If our writer fails to write out data, assume it is a connection problem and try and
145  // reconnect. NOTE: the MessageManager always takes ownership of the given data block, so if we
146  // fail to write but we resume a connection then we obtain a duplicate of the data block before
147  // we loop back. Otherwise, our held data pointer will be destroyed when the MessageManager
148  // object goes out of scope.
149  //
150  if (! writer_.write(manager)) {
151  if (! connector_->reconnect()) {
152  break;
153  }
154 
155  data = data->duplicate();
156  continue;
157  }
158  }
159 
160  data = 0;
161  }
162  }
163 
164  LOGWARNING << "exiting" << std::endl;
165  return 0;
166 }
