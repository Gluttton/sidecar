381 bool
382 MainWindow::openLogFile(const QFileInfo& fileInfo)
383 {
384  QFile headerFile(fileInfo.filePath());
385  if (! headerFile.open(QIODevice::ReadOnly | QIODevice::Text)) {
386  statusBar()->showMessage(QString("Failed to open file %1").arg(fileInfo.fileName()));
387  return false;
388  }
389 
390  openConfigFile(fileInfo);
391 
392  QTextStream in(&headerFile);
393  float rangeMin;
394  QString token;
395  if ((in >> token >> token >> rangeMin).status()) {
396  statusBar()->showMessage("Failed to read minimum range value", 5000);
397  return false;
398  }
399 
400  rangeMin_->setText(QString::number(rangeMin));
401 
402  float rangeMax;
403  if ((in >> token >> token >> rangeMax).status()) {
404  statusBar()->showMessage("Failed to read max range value", 5000);
405  return false;
406  }
407 
408  rangeMax_->setText(QString::number(rangeMax));
409 
410  int samplesPerPulse;
411  if ((in >> token >> samplesPerPulse).status()) {
412  statusBar()->showMessage("Failed to read samples/pulse value", 5000);
413  return false;
414  }
415 
416  if (samplesPerPulse == 0) {
417  statusBar()->showMessage("Samples/pulse value is zero", 5000);
418  return false;
419  }
420 
421  pulseSize_->setText(QString::number(samplesPerPulse));
422 
423  Time::TimeStamp deltaSum;
424  Time::TimeStamp lastTime;
425 
426  double when;
427  double azimuthStart;
428  double azimuthEnd;
429  double beamWidth = 0.0;
430 
431  using VideoVector = std::vector<Messages::Video::Ref>;
432  VideoVector messages;
433 
434  Messages::VMEDataMessage vme;
435  vme.header.msgDesc = ((Messages::VMEHeader::kPackedReal << 16) |
436  Messages::VMEHeader::kIRIGValidMask |
437  Messages::VMEHeader::kAzimuthValidMask |
438  Messages::VMEHeader::kPRIValidMask);
439  vme.header.pri = 0;
440  vme.header.timeStamp = 0;
441 
442  while (! (in >> when >> azimuthStart >> azimuthEnd >> token).status()) {
443  degreesToRadians(azimuthStart);
444  degreesToRadians(azimuthEnd);
445 
446  vme.header.irigTime = when;
447  ++vme.header.pri;
448 
449  if (! beamWidth) {
450  beamWidth = azimuthEnd - azimuthStart;
451  if (beamWidth < 0.0) beamWidth += M_PI * 2.0;
452  }
453 
454  vme.header.azimuth =
455  uint32_t(::rint(azimuthStart / (M_PI * 2.0) *
456  Messages::RadarConfig::GetShaftEncodingMax()));
457 
458  Messages::Video::Ref msg(
459  Messages::Video::Make("PRIEmitter", vme, samplesPerPulse));
460  msg->setCreatedTimeStamp(Time::TimeStamp(when));
461 
462  if (messages.empty()) {
463  lastTime = msg->getCreatedTimeStamp();
464  }
465  else {
466  Time::TimeStamp delta(msg->getCreatedTimeStamp());
467  delta -= lastTime;
468  deltaSum += delta;
469  lastTime = msg->getCreatedTimeStamp();
470  }
471 
472  messages.push_back(msg);
473  }
474 
475  QFileInfo dataFileInfo(fileInfo.dir(), fileInfo.baseName() + ".log");
476 
477  QFile dataFile(dataFileInfo.filePath());
478  if (! dataFile.open(QIODevice::ReadOnly)) {
479  statusBar()->showMessage(QString("Failed to open data file %1")
480  .arg(dataFileInfo.fileName()));
481  return false;
482  }
483 
484  QByteArray data = dataFile.readAll();
485  QFileInfo priFileInfo(dataFileInfo.dir(), dataFileInfo.baseName() + ".pri");
486 
487  ::unlink(priFileInfo.filePath().toStdString().c_str());
488  ACE_FILE_Addr outputAddr(priFileInfo.filePath().toStdString().c_str());
489  ACE_FILE_Connector fileConnector;
490  IO::FileWriter::Ref writer(new IO::FileWriter);
491  if (fileConnector.connect(writer->getDevice(), outputAddr) == -1) {
492  statusBar()->showMessage(QString("Failed to open PRI output file %1").arg(priFileInfo.fileName()));
493  return false;
494  }
495 
496  ByteOrder byteOrder;
497  // byteOrder.setDecodingByteOrder(ByteOrder::kLittleEndian);
498  short* ptr = reinterpret_cast<short*>(data.data());
499 
500  short valueMin = 32767;
501  short valueMax = -valueMin;
502 
503  VideoVector::iterator nextMessage = messages.begin();
504  VideoVector::iterator end = messages.end();
505  while (nextMessage != end) {
506  Messages::Video::Ref msg = *nextMessage++;
507  for (int count = samplesPerPulse; count; --count) {
508  byteOrder.decode(ptr, sizeof(short));
509  if (*ptr < valueMin) valueMin = *ptr;
510  if (*ptr > valueMax) valueMax = *ptr;
511  msg->push_back(*ptr++);
512  }
513 
514  IO::MessageManager mgr(msg);
515  if (! writer->write(mgr.getMessage())) {
516  statusBar()->showMessage(QString("Failed to encode message"), 5000);
517  }
518  }
519 
520  writer->getDevice().close();
521 
522  return openPRIFile(priFileInfo);
523 }

525 bool
526 MainWindow::openPRIFile(const QFileInfo& fileInfo)
527 {
528  if (file_) {
529  if (reader_) {
530  reader_->reset();
531  }
532  file_->close();
533  delete file_;
534  file_ = 0;
535  }
536 
537  file_ = new QFile(fileInfo.filePath());
538  if (! file_->open(QFile::ReadOnly)) {
539  statusBar()->showMessage(QString("Failed to open file %1").arg(fileInfo.fileName()));
540  delete file_;
541  file_ = 0;
542  return false;
543  }
544 
545  reader_ = IO::FileReader::Make();
546  reader_->getDevice().set_handle(file_->handle());
547 
548  openConfigFile(fileInfo);
549 
550  Time::TimeStamp deltaSum;
551  Time::TimeStamp lastTime;
552 
553  double fileSize = file_->size();
554  size_t messageCount = 0;
555  size_t numGates = 0;
556  short valueMin = 32767;
557  short valueMax = -valueMin;
558  while (reader_->fetchInput()) {
559  if (reader_->isMessageAvailable()) {
560 
561  IO::Decoder decoder(reader_->getMessage());
562  Messages::Video::Ref msg(decoder.decode<Messages::Video>());
563  if (messageCount == 0) {
564  lastTime = msg->getCreatedTimeStamp();
565  numGates = msg->size();
566  std::clog << lastTime << std::endl;
567  }
568  else {
569  Time::TimeStamp delta(msg->getCreatedTimeStamp());
570  delta -= lastTime;
571  deltaSum += delta;
572  lastTime = msg->getCreatedTimeStamp();
573  if (msg->size() > numGates) {
574  numGates = msg->size();
575  }
576  }
577 
578  ++messageCount;
579  short min = *(std::min_element(msg->begin(), msg->end()));
580  if (min < valueMin) valueMin = min;
581  short max = *(std::max_element(msg->begin(), msg->end()));
582  if (max > valueMax) valueMax = max;
583  }
584 
585  double percentComplete = reader_->getDevice().tell() / fileSize;
586  progressDialog_->setValue(int(percentComplete * 100.0));
587  qApp->processEvents();
588  if (progressDialog_->wasCanceled()) {
589  break;
590  }
591  }
592 
593  rewind();
594 
595  if (messageCount) {
596  if (messageCount > 1) {
597  double delta = deltaSum.asDouble();
598  delta /= (messageCount - 1);
599  if (delta != 0.0) {
600  int freq = int(1.0 / delta);
601  fileFrequency_->setText(QString::number(freq));
602  QSettings settings;
603  frequency_->setValue(settings.value(file_->fileName() + "Frequency", freq).toInt());
604  }
605  }
606 
607  rangeMin_->setText(QString::number(Messages::RadarConfig::GetRangeMin_deprecated()));
608  rangeMax_->setText(QString::number(Messages::RadarConfig::GetRangeMax()));
609 
610  pulseSize_->setText(QString::number(numGates));
611  beamWidthValue_ = Messages::RadarConfig::GetBeamWidth();
612  beamWidth_->setText(QString::number(beamWidthValue_));
613  }
614  else {
615  rangeMin_->setText("---");
616  rangeMax_->setText("---");
617  pulseSize_->setText("---");
618  beamWidth_->setText("---");
619  fileFrequency_->setText("---");
620  }
621 
622  valueMin_->setText(QString::number(valueMin));
623  valueMax_->setText(QString::number(valueMax));
624  pulseCount_->setText(QString::number(messageCount));
625 
626  return true;
627 }

629 void
630 MainWindow::emitMessage()
631 {
632  static Logger::ProcLog log("emitMessages", Log());
633 
634  while (1) {
635  if (! reader_->fetchInput()) {
636  rewind();
637  if (! loopAtEnd_->isChecked()) {
638  on_startStop__clicked();
639  return;
640  }
641 
642  LOGINFO << "looping" << std::endl;
643  QString when("Looping");
644  statusBar()->showMessage("Looping", 5000);
645  continue;
646  }
647 
648  if (reader_->isMessageAvailable()) break;
649  }
650 
651  IO::Decoder decoder(reader_->getMessage());
652  Messages::Video::Ref msg(decoder.decode<Messages::Video>());
653  msg->setCreatedTimeStamp(Time::TimeStamp::Now());
654 
655  if (simAz_->isChecked()) {
656  msg->getRIUInfo().shaftEncoding = uint32_t(::rint(simulatedShaft_));
657  simulatedShaft_ += shaftDelta_;
658  if (simulatedShaft_ > Messages::RadarConfig::GetShaftEncodingMax())
659  simulatedShaft_ -= Messages::RadarConfig::GetShaftEncodingMax();
660  }
661 
662  IO::MessageManager mgr(msg);
663  if (! writer_->writeMessage(mgr)) {
664  makeWriter();
665  }
