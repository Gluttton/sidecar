281  void getSlice(int offset, int span, VsipComplexVector& slice, bool doPop)
282  {
283  if (isEmpty()) return;
284 
285  // NOTE: important to hold reference to message in case we pop it from the deque.
286  //
287  typename T::Ref msg(getFront());
288  typename T::Container& container(msg->getData());
289  if (doPop) popFront();
290 
291  // Make sure that offset + span is no greater than the size of the message.
292  //
293  int limit = span;
294  if ((offset + span) * 2 > int(container.size())) {
295  limit = container.size() / 2 - offset;
296  if (limit < 0) limit = 0;
297  }
298 
299  // Now copy over the data as complex values.
300  //
301  typename T::DatumType* ptr = &container[offset * 2];
302  for (int index = 0; index < limit; ++index, ptr += 2) {
303  slice.put(index, ComplexType(*ptr, *(ptr + 1)));
304  }
305 
306  // Make sure that the slices we return are of the requested size, even if our message is too short.
307  //
308  while (limit < span) {
309  slice.put(limit++, ComplexType());
310  }
311  }
