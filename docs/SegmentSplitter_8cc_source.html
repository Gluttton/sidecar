51 bool
52 SegmentSplitter::processSegment(SegmentMessage::Ref pri)
53 {
54  // Read in the message
55  const std::list<Segment> &in=pri->data()->data();
56 
57  // identify the peaks in this extraction a peak is the highest value in its 9-cell region
58  //
59  std::list<Segment>::const_iterator seg;
60  std::list<Segment>::const_iterator stop=in.end();
61  Peak peak;
62  std::list<Peak> peaks;
63  for(seg=in.begin(); seg!=stop; seg++)
64  {
65  size_t azimuth=seg->azimuth;
66 
67  Video::DatumType oldValue=buffer.get(azimuth, seg->start - 1);
68  Video::DatumType value=buffer.get(azimuth, seg->start);
69  Video::DatumType newValue=buffer.get(azimuth, seg->start + 1);
70  for(size_t i=seg->start; i <= seg->stop; i++)
71  {
72  if(value>=newValue && value>=oldValue)
73  {
74  // found a peak on this azimuth, check to either side
75  Video::DatumType tmp;
76  peak.sum=0;
77  if((tmp=buffer.get(azimuth-1, i-1), peak.sum+=tmp, tmp<=value) &&
78  (tmp=buffer.get(azimuth-1, i), peak.sum+=tmp, tmp<=value) &&
79  (tmp=buffer.get(azimuth-1, i+1), peak.sum+=tmp, tmp<=value) &&
80  (tmp=buffer.get(azimuth+1, i-1), peak.sum+=tmp, tmp<=value) &&
81  (tmp=buffer.get(azimuth+1, i), peak.sum+=tmp, tmp<=value) &&
82  (tmp=buffer.get(azimuth+1, i+1), peak.sum+=tmp, tmp<=value))
83  {
84  // its a local maximum...
85  peak.range=i;
86  peak.azimuth=azimuth;
87  peak.value=value;
88  peak.sum+=oldValue + value + newValue;
89  peaks.push_back(peak);
90  }
91  }
92 
93  // shift
94  oldValue=value;
95  value=newValue;
96  newValue=buffer.get(azimuth, i+1);
97  }
98  }
99 
100  // Determine distance thresholds
101  //
102  const int wRange=deltaRange->getValue(); // ellipse axes
103  const int wAz=deltaAz->getValue();
104  const float close=wRange*wAz*minDist; // distance between origins
105  const float close2=close*close;
106  const int scan = RadarConfig::GetShaftEncodingMax() + 1;
107  const int scan_2 = scan / 2;
108 
109  // Sort through the peaks, picking the highest peak "per object"
110  //
111  std::list<Peak> extractions;
112  while(peaks.size())
113  {
114  // Find the highest peak
115  std::list<Peak>::iterator peak=peaks.begin();
116  std::list<Peak>::iterator max=peak; // start assuming the first peak is highest
117  std::list<Peak>::const_iterator stop=peaks.end();
118  for(peak++ ; peak!=stop; peak++) // then check the other peaks
119  {
120  if((peak->value > max->value) ||
121  ((peak->value == max->value) && (peak->sum > max->sum)))
122  {
123  max=peak;
124  }
125  }
126 
127  // declare the extraction
128  Peak summit=*max;
129  extractions.push_back(summit);
130 
131  // remove any peaks shadowed by max
132  peaks.erase(max);
133  for(peak=peaks.begin(); peak!=stop; /* increment in the loop */)
134  {
135  int dRange=wAz*(summit.range - peak->range); // cross-multiply instead of dividing
136  int dAz=summit.azimuth - peak->azimuth;
137  if(dAz < -scan_2)
138  dAz+=scan;
139  if(dAz > scan_2)
140  dAz-=scan;
141  dAz*=wRange;
142 
143  if(dRange*dRange + dAz*dAz < close2)
144  {
145  peak=peaks.erase(peak);
146  }
147  else
148  {
149  peak++;
150  }
151  }
152  }
153 
154  // Create a new SegmentMessage for each extraction
155  //
156  std::list<Peak>::const_iterator ext;
157  std::list<Peak>::const_iterator stopExt=extractions.end();
158  Segment s;
159  for(ext=extractions.begin(); ext!=stopExt; ext++)
160  {
161  // create a SegmentMessage
162  SegmentMessage::Ref output(
163  new SegmentMessage(kAlgorithmName, pri, pri->getRangeMin(),
164  pri->getRangeFactor()));
165  SegmentList &out=*output->data();
166 
167  // Set the peak
168  out.peakPower=ext->value;
169  out.peakRange=ext->range;
170  out.peakAzimuth=ext->azimuth;
171 
172  // Specify the bounds for this extraction
173  size_t az0=ext->azimuth;
174  size_t rangeMin=ext->range - wRange;
175  size_t rangeMax=ext->range + wRange;
176 
177  // scan over the original segments, and select segments which are close to this peak
178  // this time, use the 1-norm
179  for(seg=in.begin(); seg!=stop; seg++)
180  {
181  s.azimuth=seg->azimuth;
182 
183  int dAz=az0-s.azimuth;
184  if(dAz < -scan_2)
185  dAz+=scan;
186  if(dAz > scan_2)
187  dAz-=scan;
188 
189  if(abs(dAz) <= wAz)
190  {
191  s.start=seg->start;
192  if(s.start < rangeMin)
193  s.start=rangeMin;
194  s.stop=seg->stop;
195  if(s.stop > rangeMax)
196  s.stop=rangeMax;
197 
198  out.merge(s);
199  }
200  }
201 
202  // send out the SegmentMessage
203  if(!send(output))
204  return false;
205  }
206 
207  return true;
208 }
