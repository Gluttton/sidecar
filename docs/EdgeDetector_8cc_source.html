50 bool
51 EdgeDetector::processInput(const Messages::Video::Ref& msg)
52 {
53  static Logger::ProcLog log("processInput", getLog());
54 
55  Messages::BinaryVideo::Ref out(Messages::BinaryVideo::Make("EdgeDetector", msg));
56 
57  // We need three points to work with: prev - previous point curr - current point next - next point Thus, we
58  // won't have a calculated value for the first or last sample in the input message; these values will be
59  // false in the output.
60  //
61  Messages::Video::const_iterator next = msg->begin();
62  Messages::Video::const_iterator prev = next++;
63  Messages::Video::const_iterator curr = next++;
64  Messages::Video::const_iterator end = msg->end();
65 
66  out->push_back(false);
67  if (edge_->getValue() == kLeading) {
68  while (next < end) {
69  bool v = *prev < *curr && *curr <= *next;
70  if (v)
71  ++detectedCount_;
72  ++examinedCount_;
73  ++prev;
74  ++curr;
75  ++next;
76  out->push_back(v);
77  }
78  }
79  else {
80  while (next < end) {
81  bool v = *prev >= *curr && *curr > *next;
82  if (v)
83  ++detectedCount_;
84  ++examinedCount_;
85  ++prev;
86  ++curr;
87  ++next;
88  out->push_back(v);
89  }
90  }
91 
92  out->push_back(false);
93 
94  bool rc = send(out);
95  LOGDEBUG << "rc: " << rc << std::endl;
96  return rc;
97 }
