38  // We use the STL transform method to do looping for us. We tell it the first data point to process, a
39  // marker indicating the end of the data, and a function object to invoke to do the inversion. We use two
40  // STL methods to generate a function object: std::mem_fun - wrapper around a member function/method
41  // std::bind1st - a value binder that assigns a value to the 1st value of a method. For member functions,
42  // the first argument is always an object of the type that has the method
43  //
44  std::transform(msg->begin(), msg->end(), msg->begin(),
45  std::bind1st(std::mem_fun(&Inverter::invert), this));
46 
47  // Transformation is done -- send out on the default output device.
48  //
49  bool rc = send(msg);
50  LOGDEBUG << "rc: " << rc << std::endl;
51 
52  return rc;
53 }
