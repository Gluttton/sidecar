91 bool
92 ClutterMap::process(const Messages::Video::Ref& msg)
93 {
94  static Logger::ProcLog log("processVideo", getLog());
95  LOGINFO << std::endl;
96 
97  // Special case to detect very first PRI seen so we can then detect north (0 azimuth) crossings.
98  //
99  size_t shaftEncoding = msg->getShaftEncoding();
100  LOGDEBUG << "shaftEncoding: " << shaftEncoding << std::endl;
101 
102  if (scanCounter_ == -1) {
103  scanCounter_ = 0;
104  lastShaftEncoding_ = shaftEncoding;
105  return send(msg);
106  }
107 
108  // Detect north crossing.
109  //
110  static const size_t kShaftEncodingHysteresis = 100;
111 
112  if (shaftEncoding + kShaftEncodingHysteresis < lastShaftEncoding_) {
113  LOGINFO << "north mark crossing - scanCounter: " << scanCounter_
114  << " limit: " << learningScanCount_->getValue()
115  << std::endl;
116 
117  if (! mapBuffer_) {
118  mapBuffer_.reset(new MapBuffer(getName(), radialPartitionCount_->getValue(), alpha_->getValue()));
119  }
120  else if (! mapBuffer_->isFrozen()) {
121  if (++scanCounter_ == learningScanCount_->getValue()) {
122  LOGWARNING << "freezing clutter map" << std::endl;
123  mapBuffer_->freeze();
124  std::string path = saveFilePath_->getValue();
125  if (path.size()) {
126  if (! saveMap(path)) {
127  LOGERROR << "failed to save frozen map" << std::endl;
128  }
129  }
130  }
131  }
132  }
133 
134  lastShaftEncoding_ = shaftEncoding;
135  if (! enabled_->getValue() || ! mapBuffer_) {
136  return send(msg);
137  }
138 
139  Messages::Video::Ref out = mapBuffer_->add(msg);
140  bool rc = send(out);
141 
142  LOGDEBUG << "rc: " << rc << std::endl;
143 
144  return rc;
145 }
