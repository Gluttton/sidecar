88  map_.insert(Map::value_type(0x109A, "GTH"));

90  map_.insert(Map::value_type(0x132D, "GTJ"));

94  map_.insert(Map::value_type(0x1038, "MRD"));
95  map_.insert(Map::value_type(0x1007, "MRE"));

97  map_.insert(Map::value_type(0x1009, "MRG"));

175 double
176 TSPI::MakeEFGCoordinateFromRawBytes(const uint8_t* ptr)
177 {
178  // Raw-data arrives in big-endian format. If on a little-endian machine we need to do some byte-reordering
179  //
180  int32_t tmp;
181  if (QSysInfo::ByteOrder == QSysInfo::LittleEndian) {
182  // swap byte-order on little endian machine
183  //
184  tmp = (ptr[3] << 24) | (ptr[2] << 16) | (ptr[1] << 8) |
185  ptr[0];
186  }
187  else { // on a big-endian machine, no nothing to bytes
188  tmp = *((uint32_t*) ptr);
189  }
190 
191  // TSPI values are scaled by 256 to preserve some fractional component.
192  //
193  return tmp / 256.0;
194 }

196 TSPI::TSPI(const std::string& producer, ACE_Message_Block* raw)
197  : Header(producer, GetMetaTypeInfo()), when_(0.0), flags_(0), tag_(""),
198  llh_(), rae_(), xyz_()
199 {
200  Logger::ProcLog log("TSPI", Log());
201  LOGINFO << std::endl;
202 
203  const uint8_t* bytes = reinterpret_cast<const uint8_t*>(raw->rd_ptr());
204 
205  uint16_t systemId = (bytes[1] << 8) | bytes[2];
206  when_ = (bytes[3] << 8) | bytes[4];
207  flags_ = bytes[17];
208  tag_ = GetSystemIdTag(systemId);
209 
210  // Reconstruct EFG geodetic coordinates.
211  //
212  efg_[GEO_E] = MakeEFGCoordinateFromRawBytes(bytes + 5);
213  efg_[GEO_F] = MakeEFGCoordinateFromRawBytes(bytes + 9);
214  efg_[GEO_G] = MakeEFGCoordinateFromRawBytes(bytes + 13);
215 
216  if (log.showsDebug1()) dump();
217 }
