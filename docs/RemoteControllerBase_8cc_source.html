133 {
134  Logger::ProcLog log("stop", Log());
135  LOGINFO << std::endl;
136 
137  if (! running_) {
138  LOGERROR << "not running" << std::endl;
139  return false;
140  }
141 
142  // Stop the publication of the XML-RPC service connection information, and delete the publisher
143  //
144  connectionPublisher_->stop();
145  connectionPublisher_.reset();
146 
147  // Tell the service thread to stop processing, then wait for the thread to stop.
148  //
149  running_ = 0;
150 
151  // Tell the XML-RPC server to shutdown. However, this won't take affect until something triggers the server
152  // to do some work. So, create a bogus connection to the server.
153  //
154  rpcServer_->shutdown();
155  ACE_SOCK_Stream bogus;
156  ACE_SOCK_Connector(bogus, address_);
157 
158  // Now wait for the service thread to exit. We *must* do this in order to be sure that no XML-RPC requests
159  // are in progress when we delete the XML-RPC server below.
160  //
161  LOGINFO << "waiting for rpc thread to exit: " << running_ << std::endl;
162  if (wait() == -1) {
163  LOGERROR << "failed wait() on service thread" << std::endl;
164  return false;
165  }
166 
167  // Delete the XML-RPC server.
168  //
169  rpcServer_.reset();
170 
171  return true;
172 }
