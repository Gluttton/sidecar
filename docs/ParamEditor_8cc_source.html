657 bool
658 ParamEditor::buildParameters()
659 {
660  if (! controllerItem_->getParameters(definition_)) {
661  QMessageBox::information(
662  qApp->activeWindow(), "Edit Failed",
663  QString("<p>Unable to request the configuration for the service '"
664  "%1'. The service may no longer exists or it may not "
665  "be responding to XML/RPC requests.</p>")
666  .arg(controllerItem_->getName()),
667  QMessageBox::Ok);
668  return false;
669  }
670 
671  if (definition_.size() == 0) {
672  QMessageBox::information(qApp->activeWindow(),
673  "No Parameters",
674  "<p>The service you attempted to modify has "
675  "no editable parameters.</p>",
676  QMessageBox::Ok);
677  return false;
678  }
679 
680  QVBoxLayout* vbox = new QVBoxLayout(widgetContainer_);
681  vbox->setMargin(9);
682  vbox->setSpacing(6);
683 
684  // Create label/widget pairs for each of the parameters. Remember the edit widget and original value of the
685  // parameter.
686  //
687  for (int index = 0; index < definition_.size(); ++index) {
688  XmlRpc::XmlRpcValue& vs(definition_[index]);
689  std::string type(vs["type"]);
690  bool isAdvanced(vs["advanced"]);
691  ParamBase* item = 0;
692 
693  if (type == "int") {
694  item = new IntParam(vbox, vs);
695  }
696  else if (type == "double") {
697  item = new DoubleParam(vbox, vs);
698  }
699  else if (type == "string") {
700  item = new StringParam(vbox, vs);
701  }
702  else if (type == "readPath") {
703  item = new PathParam(vbox, vs, PathParam::kReadPath);
704  }
705  else if (type == "writePath") {
706  item = new PathParam(vbox, vs, PathParam::kWritePath);
707  }
708  else if (type == "enum") {
709  item = new EnumParam(vbox, vs);
710  }
711  else if (type == "bool") {
712  item = new BoolParam(vbox, vs);
713  }
714  else if (type == "notification") {
715  item = new NotificationParam(vbox, vs);
716  connect(item, SIGNAL(sendNotification()), SLOT(apply()));
717  }
718  else {
719  QMessageBox::information(
720  qApp->activeWindow(), "Edit Failed",
721  QString("Unexpected data type from XML/RPC response - %1")
722  .arg(type.c_str()), QMessageBox::Ok);
723  return false;
724  }
725 
726  connect(item, SIGNAL(valueChanged()), SLOT(updateButtons()));
727  parameters_.append(item);
728 
729  if (isAdvanced) {
730  item->setVisible(advanced_->isChecked());
731  connect(advanced_, SIGNAL(clicked(bool)), item,
732  SLOT(setVisible(bool)));
733  }
734  }
735 
736  adjustSize();
737  updateButtons();
738 
739  return true;
740 }

786 void
787 ParamEditor::endEdit(int status)
788 {
789  // If the user rejected the changes (via the Cancel button) then just ignore the entire edit.
790  //
791  if (status == Rejected)
792  return;
793 
794  // Build the XML-RCP request with new parameter values.
795  //
796  XmlRpc::XmlRpcValue args;
797  args.setSize(0);
798 
799  // Iterate over all of the parameter defintions and add XML-RCP data that describes the parameter and its
800  // new value.
801  //
802  QStringList changed;
803  changed.append(controllerItem_->getParameterChangedHeading());
804
