371 Log::MakeObj(const std::string& name, const std::string& fullName, Log* parent, bool hidden)
372 {
373  LogMap& map(RuntimeData::Singleton().logMap());
374 
375  // Search for the Log object in the map. If found, just return.
376  //
377  std::unique_ptr<Log> obj(map.find(fullName));
378  if (obj) return obj.release();
379 
380  // Create new Log object and install in map.
381  //
382  if (parent) {
383  obj.reset(new Log(name, fullName, parent, Priority::kError, hidden));
384  }
385  else {
386  obj.reset(new Log("root", "root", nullptr, Priority::kError, true));
387  obj->addWriter(Writers::Stream::Make(Formatters::Verbose::Make(), std::cerr, true));
388  }
389 
390  if (! map.add(obj.get())) {
391 
392  // Assume that in a mad race with another thread to add a new object, that we lost. Try the find again.
393  // If *that* fails then we throw an exception.
394  //
395  obj.reset(map.find(fullName));
396  if (! obj) {
397  Utils::Exception ex("Log::FetchObj: failed to insert new Log " "object for key: ");
398  ex << fullName;
399  std::cerr << ex.err() << std::endl;
400  throw ex;
401  }
402  }
403 
404  return obj.release();
405 }

416 Log*
417 Log::FindObj(const std::string& name, bool hidden)
418 {
419  std::string fullName("");
420  std::string objName(name);
421  if (objName.size() == 0) objName = "root";
422  Log* parent = nullptr;
423 
424  // See if we are trying to create the top-level object. Look for some aliases.
425  //
426  if (objName == "root") {
427  fullName = "root";
428  }
429  else {
430 
431  // Non-root object name. See if we have a full-path. If not, use `root' as the parent.
432  //
433  std::string::size_type lastPeriod(objName.rfind('.'));
434  if (lastPeriod > objName.size()) {
435  parent = FindObj("root", false);
436  }
437  else {
438  fullName = objName.substr(0, lastPeriod);
439  if (fullName.size() == 0) {
440  fullName = "root";
441  }
442 
443  objName = objName.substr(lastPeriod + 1, std::string::npos);
444  if (objName.size() == 0) {
445  std::cerr << "invalid parent name" << std::endl;
446  throw std::invalid_argument(name);
447  }
448 
449  parent = FindObj(fullName, false);
450  }
451 
452  assert(parent && parent->name_.size());
453  fullName = MakeFullName(parent->fullName(), objName);
454  }
455 
456  // Look for the requested object. If not found, create it.
457  //
458  return MakeObj(objName, fullName, parent, hidden);
459 }
