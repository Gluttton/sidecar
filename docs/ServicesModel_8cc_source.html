311  endRemoveRows();
312  }

315  beginRemoveRows(getModelIndex(configurationItem), row,

323  }

363 void
364 ServicesModel::updateStatus(const QList<QByteArray>& statusReports)
365 {
366  static Logger::ProcLog log("updateStatus", Log());
367 
368  foreach (QByteArray statusReport, statusReports) {
369 
370  // Verify that we have a valid XML-RPC value. For some reason I've seen corrupted XML-RPC messages.
371  // Still don't know why...
372  //
373  int offset = 0;
374  XmlRpc::XmlRpcValue xmlValue(statusReport.data(), &offset);
375  if (! xmlValue.valid() ||
376  xmlValue.getType() != XmlRpc::XmlRpcValue::TypeArray) {
377  LOGERROR << "invalid type from runner: " << xmlValue.getType()
378  << " size: " << statusReport.size() << std::endl;
379  continue;
380  }
381 
382  // Lame attempt at versioning XML-RPC messages. Throw it out?
383  //
384  Runner::RunnerStatus status(xmlValue);
385  if (! status.isLatestVersion()) {
386  LOGWARNING << "skipping status report with version "
387  << status.getVersion() << std::endl;
388  continue;
389  }
390 
391  QString serviceName(
392  QString::fromStdString(status.getServiceName()));
393  LOGDEBUG << "serviceName: " << serviceName << std::endl;
394 
395  // See if there is a service entry for the runner. It may be that it quit right after sending a status
396  // message, and we are processing a ghost message.
397  //
398  ServiceEntry* serviceEntry = browser_->getServiceEntry(serviceName);
399  if (! serviceEntry)
400  continue;
401 
402  // Obtain the configuration item that contains the runner sending us status.
403  //
404  QString configName(QString::fromStdString(status.getConfigName()));
405  LOGDEBUG << "configName: " << configName << std::endl;
406  ConfigurationItem* configurationItem = rootItem_->find(configName);
407  if (! configurationItem) {
408 
409  // Does not exist. Create an object to represent the configuration and update the view.
410  //
411  configurationItem = new ConfigurationItem(status, rootItem_);
412  int row = rootItem_->getInsertionPoint(configName);
413  beginInsertRows(QModelIndex(), row, row);
414  rootItem_->insertChild(row, configurationItem);
415  endInsertRows();
416  }
417 
418  // Locate the runner item that represents the runner sending us status.
419  //
420  RunnerItem* runnerItem = configurationItem->findService(serviceName);
421  if (! runnerItem) {
422 
423  // Does not exist. Create an object to represent the runner and update the view.
424  //
425  runnerItem = new RunnerItem(status, configurationItem);
426 
427  int row = configurationItem->getInsertionPoint(
428  runnerItem->getName());
429 
430  LOGDEBUG << "beginInsertRows row: " << row << ' ' << serviceName
431  << std::endl;
432  beginInsertRows(getModelIndex(configurationItem), row, row);
433  configurationItem->insertChild(row, runnerItem);
434  endInsertRows();
435 
436  emit runnerAdded(runnerItem);
437  emit statusUpdated();
438  }
439  else {
440 
441  // Just update the status of the existing runner and if necessary, update the status display.
442  //
443  if (configurationItem->update(status, runnerItem)) {
444  emit dataChanged(getModelIndex(runnerItem, kState),
445  getModelIndex(runnerItem,
446  kNumColumns - 1));
447  emit statusUpdated();
448  }
449  }
450 
451  // If we don't have a service entry for the object, try and set one.
452  //
453  if (! runnerItem->getServiceEntry()) {
454  ServiceEntry* serviceEntry = browser_->getServiceEntry(
455  serviceName);
456  LOGDEBUG << "serviceName: " << serviceName << " serviceEntry: "
457  << serviceEntry << std::endl;
458  if (serviceEntry) {
459  if (serviceEntry->isResolved()) {
460  LOGDEBUG << "resolved" << std::endl;
461  runnerItem->setServiceEntry(serviceEntry);
462  }
463  }
464  }
465  }
466 }
