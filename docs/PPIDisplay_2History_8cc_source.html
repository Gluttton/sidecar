27 calculateCapacity(size_t size)
28 {
29  // Find the next power-of-ten multiple of a given value. First, determine the largest power-of-ten that the
30  // given value is greater than. Reduce the value by that scaling factor, round up, and rescale to get the
31  // new value.
32  //
33  double scale = ::pow(10, int(::log10(size)));
34  return size_t(::ceil(size / scale) * scale);

48 PRIMessage::Ref
49 History::Entry::findByAzimuth(const MessageVector& data, double azimuth)
50  const
51 {
52  static const int kMinDelta = 10;
53 
54  if (data.empty())
55  return PRIMessage::Ref();
56 
57  // Obtain shaft encoding for the given azimuth value.
58  //
59  int wanted = static_cast<int>(
60  ::rint(azimuth * (RadarConfig::GetShaftEncodingMax() + 1) /
61  (M_PI * 2.0)));
62 
63  MessageVector::const_iterator low = data.begin();
64  int lowShaft = (*low)->getShaftEncoding();
65  if (wanted < lowShaft)
66  wanted += (RadarConfig::GetShaftEncodingMax() + 1);
67 
68  MessageVector::const_iterator high = data.end() - 1;
69  int highShaft = (*high)->getShaftEncoding();
70  if (highShaft < wanted)
71  highShaft += (RadarConfig::GetShaftEncodingMax() + 1);
72 
73  while (low < high) {
74 
75  MessageVector::const_iterator mid = low + (high - low) / 2;
76  const PRIMessage::Ref& msg(*mid);
77  int midShaft = msg->getShaftEncoding();
78 
79  if (midShaft < lowShaft)
80  midShaft += (RadarConfig::GetShaftEncodingMax() + 1);
81 
82  if (wanted == midShaft) {
83  return msg;
84  break;
85  }
86 
87  if (mid == low) {
88 
89  if (wanted == highShaft)
90  return (*high);
91 
92  int deltaLow = wanted - lowShaft;
93  int deltaHigh = highShaft - wanted;
94  if (deltaLow < deltaHigh) {
95  if (deltaLow < kMinDelta)
96  return *low;
97  }
98  else {
99  if (deltaHigh < kMinDelta)
100  return *high;
101  }
102  break;
103  }
104 
105  if (wanted < midShaft) {
106  high = mid;
107  highShaft = midShaft;
108  }
109  else {
110  low = mid;
111  lowShaft = midShaft;
112  }
113  }
114 
115  return PRIMessage::Ref();
116 }

396 void
397 History::setRetentionSize(int newSize)
398 {
399  if (enabled_)
400  enabled_ = newSize != 0;
401 
402  retentionSize_ = newSize;
403 
404  // First normalize the buffer so that the oldest entry is first, followed by newer ones.
405  //
406  size_t size = retained_.size();
407  if (size) {
408  if (oldestRetained_) {
409  std::rotate(retained_.begin(), retained_.begin() + oldestRetained_,
410  retained_.end());
411  oldestRetained_ = 0;
412  }
413  }
414 
415  // If reducing the number of retained entries, purge the oldest ones from the beginning.
416  //
417  if (size_t(newSize) < size) {
418  size_t reduction = size - newSize;
419  for (size_t index = 0; index < reduction; ++index) {
420  if (viewedEntryIsRetained_ &&
421  retained_[index] == viewedEntry_) {
422  viewedEntryIsRetained_ = false;
423  }
424  else {
425  delete retained_[index];
426  }
427  }
428 
429  retained_.erase(retained_.begin(), retained_.begin() + reduction);
430 
431  emit retainedCountChanged(newSize);
432  }
433 }

504  }

509  }
510  }
