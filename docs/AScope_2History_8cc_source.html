33 History::allocateSlot()
34 {
35  Logger::ProcLog log("allocateSlot", Log());
36  LOGINFO << "existing: " << slotMapping_.size() << std::endl;
37 
38  // Obtain a unique slot index for a data channel. The slot index stays constant durng the lifetime of the
39  // data channel. First, try looking for an available entry in the slot map.
40  //
41  for (int index = 0; index < slotMapping_.size(); ++index) {
42  if (slotMapping_[index] == kSlotUnallocated) {
43  slotMapping_[index] = kSlotNeedsIndex;
44  LOGDEBUG << "allocated " << index << std::endl;
45  return index;
46  }
47  }
48 
49  // No available entry found above.
50  //
51  slotMapping_.push_back(kSlotNeedsIndex);
52  LOGDEBUG << "allocated " << (slotMapping_.size() - 1) << std::endl;
53  return slotMapping_.size() - 1;

57 History::releaseSlot(int slot)
58 {
59  static Logger::ProcLog log("releaseSlot", Log());
60  LOGINFO << "slot: " << slot << " buffers.size: " << buffers_.size()
61  << " slotMapping.size: " << slotMapping_.size() << std::endl;
62 
63  // If the slot was assigned a buffer for past data, we need to remove the buffer and possibly update the
64  // slot map.
65  //
66  int oldIndex = slotMapping_[slot];
67  if (oldIndex >= 0) {
68 
69  // Forget the history buffer assigned to the slot and recalculate the overall size.
70  //
71  delete buffers_.takeAt(oldIndex);
72  frameCount_ = 0;
73  foreach (MessageBuffer* buffer, buffers_) {
74  if (buffer->size() > frameCount_)
75  frameCount_ = buffer->size();
76  }
77 
78  // Need to adjust entries in the slot map with indices smaller than the one we just removed.
79  //
80  for (int index = 0; index < slotMapping_.size(); ++index) {
81  if (slotMapping_[index] >= oldIndex)
82  slotMapping_[index] -= 1;
83  }
84 
85  // We were using an entry in the live frame, but not any more.
86  //
87  liveFrame_.shrink();
88  }
89 
90  // Remove the index from the slot mapping. If removing the last entry in the map, try collapsing the vector
91  // by removing other entries with kSlotUnallocated values.
92  //
93  if (slot == slotMapping_.size() - 1) {
94  do {
95  slotMapping_.pop_back();
96  }
97  while (! slotMapping_.empty() &&
98  slotMapping_.back() == kSlotUnallocated);
99  }
100  else {
101  slotMapping_[slot] = kSlotUnallocated;
102  }
103 }

105 void
106 History::update(int slot, const MessageList& data)
107 {
108  static Logger::ProcLog log("update", Log());
109  LOGINFO << "slot: " << slot << " count: " << data.size() << std::endl;
110 
111  // Obtain an index into the frames from the slot value assigned by allocateSlot(). Since this value may
112  // change depending on future releaseSlot() calls, it should not be held outside of this routine.
113  //
114  int bufferIndex = slotMapping_[slot];
115  if (bufferIndex == kSlotNeedsIndex) {
116  bufferIndex = buffers_.size();
117  slotMapping_[slot] = bufferIndex;
118  buffers_.append(new MessageBuffer);
119  liveFrame_.expand();
120  }
121 
122  LOGDEBUG << "bufferIndex: " << bufferIndex << std::endl;
123 
124  Messages::PRIMessage::Ref msg =
125  boost::dynamic_pointer_cast<Messages::PRIMessage>(data.back());
126 
127  liveFrame_.update(bufferIndex, msg);
128  emit liveFrameChanged();
129 
130  if (! enabled_)
131  return;
132 
133  if (pastFrozen_)
134  return;
135 
136  // Prune old messages from all of the message buffers. Take out messages that are older than the time of the
137  // last message - duration.
138  //
139  frameCount_ = 0;
140  Time::TimeStamp limit = msg->getEmittedTimeStamp();
141  limit -= duration_;
142  MessageBuffer* buffer;
143  foreach (buffer, buffers_) {
144 
145  // The beginning of the buffer contains the oldest message. Locate the first message from the start that
146  // is wanted.
147  //
148  MessageBuffer::iterator pos = buffer->begin();
149  while (pos != buffer->end() &&
150  (*pos)->getEmittedTimeStamp() < limit) {
151  pos->reset();
152  ++pos;
153  }
154 
155  // Remove them.
156  //
157  if (pos != buffer->begin()) {
158  buffer->erase(buffer->begin(), pos);
159  }
160 
161  // Keep track of the largest buffer size.
162  //
163  if (frameCount_ < buffer->size())
164  frameCount_ = buffer->size();
165  }
166 
167  // Append the incoming messages to the proper buffer.
168  //
169  buffer = buffers_[bufferIndex];
170  for (size_t index = 0; index < data.size(); ++index) {
171  msg = boost::dynamic_pointer_cast<Messages::PRIMessage>(
172  data[index]);
173  buffer->append(msg);
174  }
175 
176  if (frameCount_ < buffer->size())
177  frameCount_ = buffer->size();
178 }
