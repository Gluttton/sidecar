640 {
641  Logger::ProcLog log("adjustColumnSizes", Log());
642  LOGINFO << "begin" << std::endl;
643 
644  // Approach: fetch the width of the view port, subtract the column width for all columns but the first one
645  // (name), and allocate the remaining space to the first one if it is greater than its minimum width.
646  //
647  QTableView* configurations = mainWindow_.configurations_;
648  configurations->resizeColumnsToContents();
649  int minimum = configurations->columnWidth(0);
650  int available(configurations->viewport()->width());

652  available -= configurations->columnWidth(column);
653  if (available >= minimum)
