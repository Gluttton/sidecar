176 bool
177 HistoryPosition::checkAzimuthLatch()
178 {
179  static Logger::ProcLog log("checkAzimuthLatch", Log());
180 
181  if (! latchChannel_) return true;
182 
183  int slot = latchChannel_->getHistorySlot();
184  if (slot == -1)
185  return true;
186 
187  Messages::PRIMessage::Ref msg(history_.getLiveMessage(slot));
188 
189  double azimuth = msg->getAzimuthStart();
190 
191  if (! azLatchEnabled_ || lastAzimuth_ < 0.0) {
192  lastAzimuth_ = azimuth;
193  return true;
194  }
195 
196  bool caught = false;
197 
198  if (azimuth == lastAzimuth_) {
199  return ! caught_;
200  }
201 
202  if (azimuth > lastAzimuth_) {
203  caught = lastAzimuth_ < latch_ && azimuth >= latch_;
204  LOGDEBUG << "test 1: " << caught << std::endl;
205  }
206  else {
207  caught = (lastAzimuth_ < latch_ &&
208  azimuth + Utils::kCircleRadians >= latch_) ||
209  (lastAzimuth_ - Utils::kCircleRadians < latch_ &&
210  azimuth >= latch_);
211  LOGDEBUG << "test 2: " << caught << std::endl;
212  }
213 
214  lastAzimuth_ = azimuth;
215 
216  if (caught) {
217 
218  LOGDEBUG << "caught " << azimuth << ' ' << msg->getSequenceCounter()
219  << std::endl;
220 
221  // Caught an azimuth.
222  //
223  if (! caught_) {
224  caught_ = true;
225  if (infoReporter_)
226  azimuthLatch_->latch();
227  return true;
228  }
229 
230  // If relatching is enabled, let this one thru.
231  //
232  if (azLatchRelatch_) {
233  LOGDEBUG << "relatched" << std::endl;
234  return true;
235  }
236  }
237 
238  return ! caught_;
239 }
