59 ChannelBuffer::pruneToSequenceCounter(uint32_t sequenceCounter)
60 {
61  static const uint32_t kWrapped = 1 << 16;
62 
63  // Skip entries until we have a message with a sequenceCounter that is not older than the given one.
64  //
65  while (! isEmpty() &&
66  ((sequenceCounter > nextSequenceCounter_) ||
67  (sequenceCounter < nextSequenceCounter_ && nextSequenceCounter_ - sequenceCounter > kWrapped))) {
68  popFrontInternal();
69  }
70  return ! isEmpty() && nextSequenceCounter_ == sequenceCounter;
71 }
