580 Time::TimeStamp
581 MainWindow::createTimeStamp(QString spec)
582  const
583 {
584  Logger::ProcLog log("createTimeStamp", Log());
585  LOGINFO << "spec: " << spec << std::endl;
586 
587  bool isOffset = false;
588  bool ok = false;
589  double value;
590 
591  if (spec[0] == '+' || spec[0] == '-') {
592 
593  // Relative time given in floating-point hours, minutes or seconds, depending on optional suffix
594  // character.
595  //
596  double factor = 1.0;
597  char c = spec.at(spec.length() - 1).toLatin1();
598  if (c == 'h' || c == 'H') factor = 3600.0;
599  else if (c == 'm' || c == 'M') factor = 60.0;
600  if (factor != 1.0 || c == 's' || c == 'S')
601  spec.remove(spec.length() - 1, 1);
602  isOffset = true;
603  value = spec.toDouble(&ok) * factor;
604  LOGDEBUG << "offset value: " << value << std::endl;
605  }
606  else {
607 
608  // Absolute time given in HH:MM:SS.sss or floating-point seconds since midnight (0-86400)
609  //
610  if (spec.length() > 7 &&
611  (spec[2] == '.' || spec[2] == ':') &&
612  (spec[5] == '.' || spec[5] == ':')) {
613  int hrs = spec[0].digitValue() * 10 + spec[1].digitValue();
614  int min = spec[3].digitValue() * 10 + spec[4].digitValue();
615  spec.remove(0, 6);
616 
617  double secs = spec.toDouble(&ok);
618  if (! ok) {
619  secs = spec.toInt(&ok);
620  if (! ok) {
621  secs = 0.0;
622  ok = true;
623  }
624  }
625  value = hrs * 3600.0 + min * 60.0 + secs;
626  }
627  else {
628  value = spec.toDouble(&ok);
629  if (! ok)
630  value = spec.toInt(&ok);
631  }
632 
633  LOGDEBUG << "absolute value: " << value << std::endl;
634  }
635 
636  if (! ok)
637  return Time::TimeStamp::Min();
638 
639  Time::TimeStamp when;
640  if (isOffset) {
641  when = clock_->getPlaybackClock();
642  when += value;
643  }
644  else {
645  when = zero_;
646  when += value;
647  }
648 
649  LOGDEBUG << "when: " << when << std::endl;
650  return when;
651 }
