106 void
107 App::initializeRealTime(const QString& scheduler)
108 {
109  Logger::ProcLog log("initializeRealTime", Log());
110  LOGINFO << "scheduler: " << scheduler.toStdString() << std::endl;
111 
112  int policy;
113  if (scheduler == "SCHED_FIFO") {
114  policy = ACE_SCHED_FIFO;
115  }
116  else if (scheduler == "SCHED_RR") {
117  policy = ACE_SCHED_RR;
118  }
119  else if (scheduler == "SCHED_OTHER") {
120  policy = ACE_SCHED_OTHER;
121  }
122  else {
123  Utils::Exception ex("invalid RT scheduler policy specified - ");
124  ex << scheduler.toStdString();
125  log.thrower(ex);
126  }
127 
128  int priority = runnerConfig_->getPriority().toInt();
129  if (priority == 0) {
130  Utils::Exception ex("invalid scheduler priority specified - ");
131  ex << priority;
132  log.thrower(ex);
133  }
134 
135  LOGWARNING << "configuring scheduler" << std::endl;
136  ACE_Sched_Params schedParams(policy, priority, ACE_SCOPE_PROCESS);
137 
138  if (ACE_OS::sched_params(schedParams) != 0) {
139  Utils::Exception ex("");
140  if (ACE_OS::last_error () == EPERM)
141  ex << "User is not priviledged to modify scheduler";
142  else
143  ex << "Failed to set scheduler params";
144  log.thrower(ex);
145  }
146 
147  LOGINFO << "Policy is = " << schedParams.policy()
148  << " (0=OTHER, 1=FIFO, 2=RR) " << std::endl;
149  LOGINFO << "Priority MIN = "
150  << ACE_Sched_Params::priority_min(policy, ACE_SCOPE_PROCESS)
151  << std::endl;
152  LOGINFO << "Priority MAX = "
153  << ACE_Sched_Params::priority_max(policy, ACE_SCOPE_PROCESS)
154  << std::endl;
155  LOGINFO << "Priority is = " << schedParams.priority() << std::endl;
156 
157  // Do this before we call mlockall, at least according to mlockall(2) NOTES section.
158  //
159  LOGWARNING << "prefaulting the stack." << std::endl;
160  stackPrefault();
161 
162  LOGWARNING << "calling mlockall to prevent page faults" << std::endl;
163  if (::mlockall(MCL_CURRENT | MCL_FUTURE) == -1) {
164  LOGERROR << "mlockall failed" << std::endl;
165  }
166 }

337 void
338 App::fillStatus(XmlRpc::XmlRpcValue& status)
339 {
340  static Logger::ProcLog log("fillStatus", Log());
341  LOGINFO << streams_.size() << std::endl;
342 
343  std::unique_ptr<XmlRpc::XmlRpcValue::ValueArray> streamStatusArray(new XmlRpc::XmlRpcValue::ValueArray);
344  for (auto s: streams_) {
345  IO::StreamStatus streamStatus(s->getName());
346  s->fillStatus(streamStatus);
347  streamStatusArray->push_back(streamStatus.getXMLData());
348  }
349 
350  double memoryUsed = 0;
351 
352 #ifdef linux
353 
354  // Fetch data from /proc/PID/statm, where PID is our process ID.
355  //
356  std::ifstream fs(statmPath_.c_str());
357  int size, resident, share, text, lib, data, dt;
358  fs >> size >> resident >> share >> text >> lib >> data >> dt;
359 
360  // According to gnome-system-monitor, the amount of writable memory (allocated data) available to a runner
361  // is equal to resident - shared. Since these are page counts, multiply by our page size to get byte counts.
362  // Let the RunnerItem class in the Master application convert the byte count into a suitable display format.
363  //
364  memoryUsed = (resident - share) * 4096.0;
365 
366 #endif
367 
368 #ifdef darwin
369  malloc_statistics_t t = { 0, 0, 0, 0 };
370  malloc_zone_statistics(0, &t);
371  memoryUsed = t.size_allocated;
372 #endif
373 
374  RunnerStatus::Make(status, *runnerConfig_, std::move(streamStatusArray), std::move(logCollector_->dump()),
375  memoryUsed);
376 }
