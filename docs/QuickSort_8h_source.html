20  // Randomly calculate a pivot to use for the partitioning.
21  //
22  ValueType pivot = low[::random() % (high - low)];
23 
24  // Walk the iterators towards each other, looking for out-of-order elements with respect to the pivot value.
25  // We are looking for the point at which we can swap two elements.
26  //
27  RandomAccessIterator nl = low;
28  RandomAccessIterator nh = high - 1;
29  do {
30

32  while (*nh > pivot) --nh;

34  if (nl <= nh) {

38  //

40  *nl++ = *nh;

46  //
