111 void
112 MagnifierWindow::setBounds(double xMin, double yMin, double width,
113  double height, double xScale, double yScale)
114 {
115  const int kMaxViewWidth = 700;
116  const int kMaxViewHeight = 700;
117 
118  // Calculate scale transforms from pixels to real-world. The two axis are separate because they are in
119  // different units (radians vs. kilometers).
120  //
121  double nw = width / xScale;
122  double nh = height / yScale;
123 
124  if (nw > nh) {
125 
126  // Use kMaxViewWidth for view widget width. Use scaled kMaxViewHeight for view widget height.
127  //
128  viewWidth_ = kMaxViewWidth;
129  viewHeight_ = int(::rint(kMaxViewWidth * nh / nw));
130 
131  // Recalculate real-world height to account for rounding of widget height.
132  //
133  height = viewHeight_ * nw * yScale / viewWidth_;
134  }
135  else {
136 
137  // Use kMaxViewHeight for view widget height. Use scaled kMaxViewWidth for view widget width.
138  //
139  viewHeight_ = kMaxViewHeight;
140  viewWidth_ = int(::rint(kMaxViewHeight * nw / nh));
141 
142  // Recalculate real-world width to account for rounding of widget width.
143  //
144  width = viewWidth_ * nh * xScale / viewHeight_;
145  }
146 
147  view_->setBounds(xMin + width / 2.0, yMin + height / 2.0,
148  width / viewWidth_, height / viewHeight_);
149 }
