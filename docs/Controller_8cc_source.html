197 
198  Super::close(flags);
199 
200  if (flags) {
201 
202  // If timerThread is running, shut it down
203  //
204  LOGDEBUG << "task " << algorithmName_ << " checking to see if timer thread is alive" << std::endl;
205  if (timerThread_.joinable()){
206  timerThread_.interrupt();
207  timerThread_.join();
208  }
209  else {
210  LOGDEBUG << "task " << algorithmName_ << " timer thread not alive" << std::endl;

214  // routine and terminate.
215  //

240  //

676 Controller::alarmTimerProc()
677 {
678  Logger::ProcLog log("alarmTimerProc", Log());
679  LOGINFO << getTaskName() << std::endl;
680 
681  boost::posix_time::seconds period(getTimerSecs());
682  boost::system_time now = boost::get_system_time();
683 
684  while (1) {
685 
686  // Wait until it is time to wakeup.
687  //
688  boost::system_time wakeupTime = now + period;
689  boost::thread::sleep(wakeupTime);
690 
691  // Record time here to minimize timer drift from message queue latency.
692  //
693  now = boost::get_system_time();
694 
695  // Send control message to Task to let it know to invoke its alarm handler function in a thread-safe
696  // manner.
697  //
698  ACE_Message_Block* data = IO::MessageManager::MakeControlMessage(IO::ControlMessage::kTimeout, 0);
699  if (put(data, 0) == -1) {
700  LOGINFO << getTaskName() << " failed to post control message" << std::endl;
701  break;
702  }
703  }
704 
705  LOGINFO << getTaskName() << " thread exiting" << std::endl;
706 }
