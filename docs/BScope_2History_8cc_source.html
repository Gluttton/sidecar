35 History::findByAzimuth(const MessageVector& data, double azimuth) const
36 {
37  static const int kMinDelta = 10;
38 
39  if (data.empty())
40  return PRIMessage::Ref();
41 
42  // Obtain shaft encoding for the given azimuth value.
43  //
44  int wanted = static_cast<int>(
45  ::rint(azimuth * (RadarConfig::GetShaftEncodingMax() + 1) /
46  (M_PI * 2.0)));
47 
48  MessageVector::const_iterator low = data.begin();
49  int lowShaft = (*low)->getShaftEncoding();
50  if (wanted < lowShaft)
51  wanted += (RadarConfig::GetShaftEncodingMax() + 1);
52 
53  MessageVector::const_iterator high = data.end() - 1;
54  int highShaft = (*high)->getShaftEncoding();
55  if (highShaft < wanted)
56  highShaft += (RadarConfig::GetShaftEncodingMax() + 1);
57 
58  while (low < high) {
59 
60  MessageVector::const_iterator mid = low + (high - low) / 2;
61  const PRIMessage::Ref& msg(*mid);
62  int midShaft = msg->getShaftEncoding();
63 
64  if (midShaft < lowShaft)
65  midShaft += (RadarConfig::GetShaftEncodingMax() + 1);
66 
67  if (wanted == midShaft) {
68  return msg;
69  break;
70  }
71 
72  if (mid == low) {
73 
74  if (wanted == highShaft)
75  return (*high);
76 
77  int deltaLow = wanted - lowShaft;
78  int deltaHigh = highShaft - wanted;
79  if (deltaLow < deltaHigh) {
80  if (deltaLow < kMinDelta)
81  return *low;
82  }
83  else {
84  if (deltaHigh < kMinDelta)
85  return *high;
86  }
87  break;
88  }
89 
90  if (wanted < midShaft) {
91  high = mid;
92  highShaft = midShaft;
93  }
94  else {
95  low = mid;
96  lowShaft = midShaft;
97  }
98  }

101 }
