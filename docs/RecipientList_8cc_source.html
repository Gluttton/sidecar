53  }
54 
55  auto ok = true;
56  if (entries_.size() > 1) {
57 
58  // The goal here is to minimize the number of 'duplicate' calls against data, even if they are
59  // lightweight. We use tmp to keep track of any unused duplicate message, being careful to free it if it
60  // is not used at the end of the loop.
61  //
62  ACE_Message_Block* tmp = 0;
63  for (size_t index = entries_.size() - 1; index; --index) {
64 
65  // Perform a delivery?
66  //
67  if (! entries_[index].task_->isUsingData()) continue;
68 
69  // Need new copy?
70  //
71  if (! tmp) tmp = data->duplicate();
72 
73  // Embed the recipient's input channel index into the message block so that it can determine where
74  // it came from (all put() operations use the same message queue)
75  //
76  tmp->msg_priority(entries_[index].channelIndex_);
77  if (entries_[index].task_->put(tmp, 0) == 0) {
78  tmp = 0; // put() OK
79  }
80  else {
81  ok = false; // put() failed
82  }
83  }
84 
85  // It is possible that we might have a non-NULL tmp if the last put() call above failed.
86  //
87  if (tmp) tmp->release();
88  }
89 
90  // Finally, use the original message for the first recipient.
91  //
92  if (entries_[0].task_->isUsingData()) {
93 
94  // Embed the recipient's input channel index into the message block so that it can determine where it
95  // came from (all put() operations use the same message queue)
96  //
97  data->msg_priority(entries_[0].channelIndex_);
98  if (entries_[0].task_->put(data, 0) == -1) {
99 
100  // put() failed
