26 void
27 GLFont::DetermineTextureType()
28 {
29  Logger::ProcLog log("DetermineTextureType", Log());
30  LOGERROR << std::endl;
31 
32  // Basic goal is to use a non-power of two texture type since it will save texture memory and improve
33  // performance. First, check OpenGL version. If version 2.0 or higher than 2D textures are not constrained
34  // to power of two dimensions so use them.
35  //
36  QString versionText(
37  reinterpret_cast<const char*>(glGetString(GL_VERSION)));
38  LOGERROR << "versionText: " << versionText << std::endl;
39 
40  bool ok = true;
41  float version = versionText.section(' ', 0, 0).toFloat(&ok);
42  LOGERROR << "version: " << version << std::endl;
43  if (ok && version > 2.0) {
44  LOGERROR << "using GL_TEXTURE_2D (non-power-of-2)" << std::endl;
45  textureType_ = GL_TEXTURE_2D;
46  powerOfTwoRequired_ = false;
47  return;
48  }
49 
50  QString extensionsText(
51  reinterpret_cast<const char*>(glGetString(GL_EXTENSIONS)));
52  LOGERROR << "extensions: " << extensionsText << std::endl;
53 
54  QStringList extensions = extensionsText.split(' ');
55 
56 #ifdef GL_TEXTURE_RECTANGLE_ARB
57  if (extensions.contains("GL_ARB_texture_rectangle")) {
58  LOGERROR << "usng GL_TEXTURE_RECTANGLE_ARB" << std::endl;
59  textureType_ = GL_TEXTURE_RECTANGLE_ARB;
60  powerOfTwoRequired_ = false;
61  return;
62  }
63 #endif
64 
65 #ifdef GL_TEXTURE_RECTANGLE_EXT
66  if (extensions.contains("GL_EXT_texture_rectangle")) {
67  LOGERROR << "usng GL_TEXTURE_RECTANGLE_EXT" << std::endl;
68  textureType_ = GL_TEXTURE_RECTANGLE_EXT;
69  powerOfTwoRequired_ = false;
70  return;
71  }
72 #endif
73 
74  LOGERROR << "usng GL_TEXTURE_2D (power-of-2)" << std::endl;
75  textureType_ = GL_TEXTURE_2D;
76  powerOfTwoRequired_ = true;
77 }

105 void
106 GLFont::render(const VertexColorTagArray& tags)
107 {
108  static Logger::ProcLog log("render", Log());
109 
110  // Prepare to draw the character textures. NOTE: expects the position values in the given
111  // VertexColorTagArray to be in window coordinates, and it calls glLoadIdentity() in the current matrix mode
112  // prior to moving to the tag position.
113  //
114  glPushAttrib(GL_ENABLE_BIT);
115  glEnable(textureType_);
116  glEnable(GL_BLEND);
117 
118  // If this is enabled, then overlapping tags don't look right.
119  //
120  glDisable(GL_STENCIL_TEST);
121 
122  for (size_t index = 0; index < tags.size(); ++index) {
123  const VertexColorTag& tag(tags[index]);
124 
125  LOGDEBUG << "xy: " << tag.vertex.x << ' ' << tag.vertex.y
126  << " tag: " << tag.tag << std::endl;
127 
128  // Check to see that all of the characters in the given text string have OpenGL textures for rendering.
129  //
130  analyzeText(tag.tag);
131 
132  // Move to the location of the first character, and begin drawing them. The display list for each
133  // character ends with a glTranslatef call that moves to the next drawing location.
134  //
135  glPushMatrix();
136  glTranslatef(tag.vertex.x, tag.vertex.y, 0.0);
137 
138  // Set the vertex color to the given tag color.
139  //
140  tag.color.use();
141 
142  // Apply the character textures that make up the tag.
143  //
144  foreach (QChar character, tag.tag) {
145  CharacterRenderInfoHash::const_iterator pos =
146  characterRenderInfos_.find(character);
147  if (pos != characterRenderInfos_.end()) {
148  glCallList(pos.value().displayList_);
149  }
150  else {
151  LOGERROR << "no render info for character '"
152  << QString(character) << "'" << std::endl;
153  }
154  }
155 
156  glPopMatrix();
157  }
158 
159  // Restore the saved OpenGL environment.
160  //
161  glPopAttrib();
162 }

164 void
165 GLFont::analyzeText(const QString& text)
166 {
167  static Logger::ProcLog log("analyzeText", Log());
168 
169  // Visit each character in the string.
170  //
171  foreach (QChar character, text) {
172 
173  // See if the character already has texture data for it
174  //
175  CharacterRenderInfoHash::const_iterator pos =
176  characterRenderInfos_.find(character);
177  if (pos != characterRenderInfos_.end())
178  continue;
179 
180  LOGDEBUG << "generating texture for character '" << QString(character)
181  << "'" << std::endl;
182 
183  // Calculate character dimensions
184  //
185  int width = fontMetrics_.width(character);
186  int lb = fontMetrics_.leftBearing(character);
187  int rb = fontMetrics_.rightBearing(character);
188  if (lb < 0) width -= lb;
189  if (rb < 0) width -= rb;
190  if (width <= 0)
191  continue;
192 
193  int height = fontMetrics_.height();
194  LOGDEBUG << "width: " << width << " height: " << height << std::endl;
195 
196  // Create a QPixmap to draw into. Draw with white so that when the texture is used, it will use all of
197  // the glColor4f() value.
198  //
199  QPixmap pixmap(width, height);
200  pixmap.fill(Qt::transparent);
201 
202  QPainter painter;
203  painter.begin(&pixmap);
204  painter.setFont(font_);
205  painter.setPen(Qt::white);
206 
207  int x = 0;
208  if (lb < 0)
209  x -= lb;
210 
211  painter.drawText(x, fontMetrics_.ascent(), character);
212  painter.end();
213 
214  // Convert pixmap to an image to use as the basis for the texture.
215  //
216  QImage image = pixmap.toImage();
217  if (image.depth() != 32)
218  image = image.convertToFormat(QImage::Format_ARGB32);
219 
220  // Construct padded image to hold the texture data.
221  //
222  int paddedWidth = width;
223  int paddedHeight = height;
224  if (powerOfTwoRequired_) {
225  paddedWidth = int(::pow(2, ::ceil(::log2(paddedWidth))));
226  paddedHeight = int(::pow(2, ::ceil(::log2(paddedHeight))));
227  }
228 
229  LOGDEBUG << "paddedWidth: " << paddedWidth
230  << " paddedHeight: " << paddedHeight << std::endl;
231 
232  QImage paddedImage = QImage(paddedWidth, paddedHeight,
233  QImage::Format_ARGB32);
234  paddedImage.fill(Qt::transparent);
235 
236  // Copy over the bytes into the padded QImage.
237  //
238  if (QSysInfo::ByteOrder == QSysInfo::BigEndian) {
239  LOGERROR << "big-endian" << std::endl;
240  for (int y = 0; y < image.height(); ++y) {
241  uint32_t* imageRgb = (uint32_t*)image.scanLine(y);
242  //Mirror y-coordinates for OpenGL
243  uint32_t* paddedImageRgb = (uint32_t*)paddedImage.scanLine(
244  paddedHeight - 1 - y);
245  uint32_t* end = imageRgb + image.width();
246  while (imageRgb < end) {
247 
248  // Use color channels for alpha and set all color channels to white (Qt) ARGB -> (OpenGL)
249  // RGBA
250  //
251  *paddedImageRgb++ = (*imageRgb++ & 0xFF) | 0xFFFFFF00;
252  }
253  }
254  }
255  else {
256  LOGERROR << "little-endian" << std::endl;
257  for (int y = 0; y < image.height(); ++y) {
258  uint32_t* imageRgb = (uint32_t*)image.scanLine(y);
259  //Mirror y-coordinates for OpenGL
260  uint32_t* paddedImageRgb = (uint32_t*)paddedImage.scanLine(
261  paddedHeight - 1 - y);
262  uint32_t* end = imageRgb + image.width();
263  while (imageRgb < end) {
264  // Use color channels for alpha and set all color channels to white (Qt) ARGB -> (OpenGL)
265  // ABGR
266  //
267  *paddedImageRgb++ = ((*imageRgb++ << 24) & 0xFF000000) |
268  0x00FFFFFF;
269  }
270  }
271  }
272 
273  // Create new RenderInfo entry for the character.
274  //
275  RenderInfo renderInfo;
276  glGenTextures(1, &renderInfo.texture_);
277  renderInfo.displayList_ = glGenLists(1);
278  characterRenderInfos_[character] = renderInfo;
279 
280  // Create the OpenGL texture.
281  //
282  glBindTexture(textureType_, renderInfo.texture_);
283  glTexParameteri(textureType_, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
284  glTexParameteri(textureType_, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
285  glTexImage2D(textureType_, 0, GL_ALPHA, paddedImage.width(),
286  paddedImage.height(), 0, GL_RGBA, GL_UNSIGNED_BYTE,
287  paddedImage.bits());
288 
289  // Calculate the proper positions for rendering the texture.
290  //
291  GLfloat minX(0), minY(0), maxX(0), maxY(0);
292  switch(textureType_) {
293  case GL_TEXTURE_2D:
294  minX = 0;
295  maxX = float(width) / paddedWidth;
296  minY = 1 - float(height) / paddedHeight;
297  maxY = 1;
298  break;
299 
300  case GL_TEXTURE_RECTANGLE_ARB:
301  minX = 0;
302  minY = 0;
303  maxX = width;
304  maxY = height;
305  break;
306 
307  default:
308  break;
309  }
310 
311  LOGERROR << "minX: " << minX << " minY: " << minY << " maxX: " << maxX
312  << " maxY: " << maxY << std::endl;
313 
314  // Create the display list for using the texture
315  //
316  glNewList(renderInfo.displayList_, GL_COMPILE);
317  {
318  if (lb < 0)
319  glTranslatef(lb, 0, 0);
320 
321  glBindTexture(textureType_, renderInfo.texture_);
322  glBegin(GL_QUADS);
323  {
324  glTexCoord2f(minX, minY);
325  glVertex2f(0, 0);
326  glTexCoord2f(maxX, minY);
327  glVertex2f(width, 0);
328  glTexCoord2f(maxX, maxY);
329  glVertex2f(width, height);
330  glTexCoord2f(minX, maxY);
331  glVertex2f(0, height);
332  }
333  glEnd();
334 
335  // Shift the horizontal position so that OpenGL is ready for the next character.
336  //
337  width = fontMetrics_.width(character);
338  if (lb < 0)
339  width -= lb;
340  glTranslatef(width, 0 , 0);
341  }
342  glEndList();
343  }
344 }
