102 App::processVideo(const MessageList& data)
103 {
104  static Logger::ProcLog log("processVideo", Log());
105  LOGINFO << std::endl;
106 
107  int available = idleWorkerThreads_.size();
108  int size = data.size();
109 
110  int index = size - available;
111  if (index < 0)
112  index = 0;
113 
114  for (;index < size; ++index) {
115 
116  Q_ASSERT(! idleWorkerThreads_.empty());
117 
118  Messages::Video::Ref msg =
119  boost::dynamic_pointer_cast<Messages::Video>(data[index]);
120 
121  // Fetch the next available WorkerThread, add data to it, and if it is ready for processing, then remove
122  // it from the set of available threads and start processing.
123  //
124  WorkerThread* workerThread = idleWorkerThreads_.back();
125  if (workerThread->getWorkRequest()->addData(msg)) {
126  LOGDEBUG << "processing" << std::endl;
127  idleWorkerThreads_.pop_back();
128  workerThread->submit();
129  }
130  }

175 void
176 App::setWorkerThreadCount(int value)
177 {
178  static Logger::ProcLog log("setWorkerThreadCount", Log());
179  LOGINFO << "value: " << value << " current: " << workerThreads_.size()
180  << std::endl;
181 
182  LOGINFO << "workerThreads size: " << workerThreads_.size()
183  << " idleWorkerThreads size: " << idleWorkerThreads_.size()
184  << std::endl;
185 
186  if (value == workerThreads_.size()) return;
187 
188  while (value < workerThreads_.size()) {
189 
190  // First, remove the thread from the workerThreads list. The threadFinished() method checks for
191  // membership in this before sending the FFT data to the display.
192  //
193  WorkerThread* workerThread = workerThreads_.takeLast();
194  LOGDEBUG << "stopping thread" << std::endl;
195  workerThread->stop();
196 
197  // Call sendPostedEvents() to flush out any pending calls to threadFinished() since a thread may have
198  // submitted an asynchronous call to that method while we have been in this one.
199  //
200  QCoreApplication::sendPostedEvents(this, 0);
201  LOGDEBUG << "workerThreads size: " << workerThreads_.size()
202  << " idleWorkerThreads_ size: " << idleWorkerThreads_.size()
203  << std::endl;
204 
205  int index = idleWorkerThreads_.indexOf(workerThread);
206  if (index != -1)
207  idleWorkerThreads_.removeAt(index);
208 
209  // Don't do an immediate delete, since there may be outstanding asynchronous submitRequest() calls.
210  //
211  workerThread->deleteLater();
212  }
213 
214  if (value == 0) return;
215 
216  while (value > workerThreads_.size()) {
217  WorkRequest* workRequest = workerThreads_.empty() ?
218  new WorkRequest(*weightWindow_) :
219  new WorkRequest(*workerThreads_[0]->getWorkRequest());
220  WorkerThread* workerThread = new WorkerThread(workRequest);
221  workerThreads_.append(workerThread);
222  connect(workerThread, SIGNAL(finished()),
223  SLOT(threadFinished()), Qt::QueuedConnection);
224  idleWorkerThreads_.append(workerThread);
225  }
226 }
