172 bool
173 NCIntegrate::process(Video::Ref msg)
174 {
175  static Logger::ProcLog log("process", getLog());
176  LOGINFO << std::endl;
177  LOGDEBUG << *msg.get() << std::endl;
178 
179  if (! enabled_->getValue() || numPulses_->getValue() == 1) {
180  bool rc = send(msg);
181  LOGDEBUG << "rc: " << rc << std::endl;
182  return rc;
183  }
184 
185  Video::Ref gone;
186  if (in_.full()) {
187  gone = in_.back();
188  }
189 
190  in_.add(msg);
191 
192  if (in_.size() == 1) {
193 
194  // This is the first message, just copy over its sample values into our running average buffer.
195  //
196  LOGDEBUG << "first time" << std::endl;
197  if (iqValues_->getValue()) {
198  Video::const_iterator pos = msg->begin();
199  Video::const_iterator end = msg->end();
200  while (pos < end) {
201  RunningAverageVector::value_type newI = *pos++;
202  RunningAverageVector::value_type newQ = *pos++;
203  runningAverage_.push_back(newI * newI + newQ * newQ);
204  }
205  }
206  else {
207  std::copy(msg->begin(), msg->end(), std::back_inserter<>(runningAverage_));
208  }
209  LOGDEBUG << runningAverage_ << std::endl;
210  return true;
211  }
212 
213  // Make sure we don't address out-of-bounds values.
214  //
215  size_t limit = runningAverage_.size();
216  if (msg->size() < limit) {
217  limit = msg->size();
218  if (iqValues_->getValue()) {
219  limit /= 2;
220  }
221  }
222  else if (msg->size() > limit) {
223 
224  // Expand our runningAverage_ to the largest message ever seen.
225  //
226  limit = msg->size();
227  if (iqValues_->getValue()) {
228  limit /= 2;
229  }
230  runningAverage_.resize(limit, 0);
231  }
232 
233  if (! in_.full()) {
234 
235  // Just update runningAverage_ by adding the new message to it.
236  //
237  if (iqValues_->getValue()) {
238  RunningAverageVector::iterator pos = runningAverage_.begin();
239  RunningAverageVector::iterator end = runningAverage_.begin() + limit;
240  Video::const_iterator vit = msg->begin();
241  while (pos != end) {
242  RunningAverageVector::value_type i = *vit++;
243  RunningAverageVector::value_type q = *vit++;
244  *pos++ += i * i + q * q;
245  }
246  }
247  else {
248  std::transform(runningAverage_.begin(), runningAverage_.begin() + limit, msg->begin(),
249  runningAverage_.begin(), std::plus<int32_t>());
250  }
251  LOGDEBUG << runningAverage_ << std::endl;
252  return true;
253  }
254 
255  // Base our outgoing message on the middle message in our retention queue.
256  //
257  Video::Ref midPoint(*(in_.begin() + in_.size() / 2));
258  Video::Ref out(Video::Make(getName(), midPoint));
259 
260  // Make sure we only allocate once for the samples we are about to generate.
261  //
262  out->reserve(runningAverage_.size());
263 
264  if (gone) {
265 
266  // Update runningAverage by adding to it the newest input message and subtracting the oldest input message, and
267  // fill the output message with the calculated running average.
268  //
269  if (iqValues_->getValue()) {
270  std::transform(runningAverage_.begin(), runningAverage_.end(), runningAverage_.begin(),
271  Average3IQ(in_.size(), msg->begin(), gone->begin(), out->getData()));
272  }
273  else {
274  std::transform(runningAverage_.begin(), runningAverage_.end(), runningAverage_.begin(),
275  Average3(in_.size(), msg->begin(), gone->begin(), out->getData()));
276  }
277  }
278  else {
279 
280  // Update runningAverage by adding to it the newest input message, and fill the output message with the
281  // calculated running average.
282  //
283  if (iqValues_->getValue()) {
284  std::transform(runningAverage_.begin(), runningAverage_.end(), runningAverage_.begin(),
285  Average2IQ(in_.size(), msg->begin(), out->getData()));
286  }
287  else {
288  std::transform(runningAverage_.begin(), runningAverage_.end(), runningAverage_.begin(),
289  Average2(in_.size(), msg->begin(), out->getData()));
290  }
291  }
292 
293  bool rc = send(out);
294  LOGDEBUG << "rc: " << rc << std::endl;
295 
296  return rc;
297 }
