96 {
97  // Enable verbose debugging if desired
98  //
99  if (cla_.hasOpt("debug"))
100  Logger::Log::Root().setPriorityLimit(Logger::Priority::kDebug);
101 
102  // Set the number of messages to process.
103  //
104  std::string value;
105  if (cla_.hasOpt("count", value)) {
106  value >> count_;
107  }
108 
109  // Open the input file.
110  //

112  if (! inputPath.exists()) {
113  std::cerr << "*** file '" << inputPath << "' does not exists\n";
114  return false;
115  }
116 
117  reader_ = IO::FileReaderTask::Make();
118  reader_->next(&inputTask_);
119  if (! reader_->openAndInit("", inputPath)) {
120  std::cerr << "*** failed to open reader on file '" << inputPath << "'"
121  << std::endl;
122  return false;
123  }
124 
125  reader_->start();
126 
127  // Create output path
128  //
129  std::string outputPath;
130  if (cla_.hasArg(1, outputPath)) {
131  ofs_.reset(new std::ofstream(outputPath.c_str()));
132  os_ = ofs_.get();
133  }
134 
135  // Fetch the first message so we can use its time stamp as the zero for the
136  // Time::TimeStamp::ParseSpecification call.
137  //
138  Time::TimeStamp timeStamp;
139  Messages::Header::Ref msg = readNext(timeStamp);
140  if (! msg) return false;
141 
142  std::string spec;
143  if (cla_.hasOpt("t0", spec)) {
144  t0_ = Time::TimeStamp::ParseSpecification(spec, timeStamp);
145  }
146 
147  if (cla_.hasOpt("t1", spec)) {
148  t1_ = Time::TimeStamp::ParseSpecification(spec, t0_);
149  }
150 
151  std::string lastProducer;
152  std::string lastType;
153 
154  // Output XML file header.
155  //
156  *os_ << "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
157  << "<!DOCTYPE file>\n"
158  << "<file>\n";
159 
160  do {
161 
162  // See if we are within the time frame set by the user.
163  //
164  if (timeStamp < t0_)
165  continue;
166  else if (timeStamp > t1_)
167  break;
168 
169  // See if this message has a different producer name or message type.
170  //
171  std::string producer(msg->getGloballyUniqueID().getProducerName());
172  std::string type(msg->getMetaTypeInfo().getName());
173  bool split = false;
174 
175  if (producer != lastProducer) {
176  if (lastProducer.size())
177  std::clog << "*** warning: diffent producers found in file: "
178  << producer << " and " << lastProducer << "\n";
179  split = true;
180  }
181  else if (lastType != type) {
182  if (lastType.size())
183  std::clog << "*** warning: diffent message types found in "
184  "file: " << type << " and " << lastType << "\n";
185  split = true;
186  }
187 
188  if (split) {
189 
190  // Close the last message block.
191  //
192  if (lastProducer.size())
193  *os_ << "</group>\n";
194 
195  lastProducer = producer;
196  lastType = type;
197 
198  // Open a new message block with the new information.
199  //
200  *os_ << "<group producer=\"" << producer << "\" type=\"" << type
201  << "\">\n";
202  }
203 
204  // Write out the XML data and stop if we've written enough.
205  //
206  *os_ << msg->xmlPrinter() << '\n';
207 
208  // See if we have a counter to watch.
209  //
210  if (count_) {
211  if (--count_ == 0) {
212  break;
213  }
214  }
215 
216  } while ((msg = readNext(timeStamp)));
217

219  // trying to add another message to the InputTask's message queue. We deactivate that queue to wake up any blocked
220  // thread, and then properly shutdown the FileReaderTask and its thread.
221  //
222  inputTask_.msg_queue()->deactivate();
223  reader_->close(1);
224 
225  // Close the last message block and if writing to a file, close it too.
226  //
227  *os_ << "</group>\n</file>\n";
228  if (ofs_)
229  ofs_->close();

231  return true;

235 Processor::readNext(Time::TimeStamp& timeStamp)
236 {
237  // Read the next message from the input queue.
238  //
239  Messages::Header::Ref msg;
240  ACE_Message_Block* data;
241  if (inputTask_.getq(data, 0) == -1)
242  return msg;
243 
244  // If this is an EOF signal from the FileReaderTask then signal that we are done.
245  //
246  IO::MessageManager mgr(data);
247  if (mgr.isShutdownRequest())
248  return msg;
249 
250  // Get the native message and fetch the appropriate time stamp.
251  //
252  msg = mgr.getNative();
253  timeStamp = useEmitted_ ? msg->getEmittedTimeStamp() :
254  msg->getCreatedTimeStamp();
255 
256  return msg;
