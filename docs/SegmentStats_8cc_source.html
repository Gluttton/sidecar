42 bool
43 SegmentStats::processSegmentMessage(SegmentMessage::Ref pri)
44 {
45  // Read in the message
46  SegmentList &in=*pri->data();
47 
48  // Create a new message to send
49  SegmentMessage::Ref output(new SegmentMessage("SegmentStats",
50  pri,
51  pri->getRangeMin(),
52  pri->getRangeFactor()));
53  SegmentList &out=*output->data();
54 
55  // calculate the statistics in one pass
56  //
57 
58  /*
59  Note on handling wraparound:
60  Normally one calculates the first moment by finding sum(bin*position)/sum(bin).
61  This doesn't work when there is a discontinuity (such as at 0==2pi).
62  Instead, calculate two independent first moments, one for [0,pi) and another for [pi,2pi).
63  Then look at the angle between these two moments to determine how things should resolve.
64  Only fails if the angle exceeds pi, which won't happen for a well-formed target.
65  */
66 
67  // Calculate the total power
68  const size_t pi = (RadarConfig::GetShaftEncodingMax() + 1) / 2;
69  double powerA=0; // for [0,pi)
70  double powerB=0; // for [pi,2pi)
71 
72  // Calculate the power centroid
73  double powerAx=0;
74  double powerBx=0;
75  double powery=0; // range doesn't wrap
76 
77  // Find the peak power
78  VideoT peak=std::numeric_limits<VideoT>::min();
79  size_t peakRange = 0, peakAz = 0;
80  if(in.peakPower > peak)
81  {
82  peak=in.peakPower;
83  peakRange=in.peakRange;
84  peakAz=in.peakAzimuth;
85  }
86 
87  // The master loop
88  std::list<Segment>::const_iterator seg;
89  std::list<Segment>::const_iterator stop=in.data().end();
90  for(seg=in.data().begin(); seg!=stop; seg++)
91  {
92  out.merge(*seg); // Add this segment to the output
93 
94  size_t azimuth=seg->azimuth;
95  for(size_t i=seg->start; i < seg->stop; i++)
96  {
97  VideoT p=buffer.get(azimuth, i);
98  if(azimuth < pi)
99  {
100  powerA+=p;
101  powerAx+=p*azimuth;
102  }
103  else
104  {
105  powerB+=p;
106  powerBx+=p*azimuth;
107  }
108  powery+=p*i;
109 
110  if(p>peak)
111  {
112  peak=p;
113  peakRange=i;
114  peakAz=azimuth;
115  }
116  }
117  }
118 
119  // Resolve the power centroid
120  const double dpi = (RadarConfig::GetShaftEncodingMax() + 1) / 2;
121  double power=powerA+powerB;
122  double powerx=(powerAx+powerBx)/power;
123  if(powerA && powerB)
124  {
125  powerAx/=powerA;
126  powerBx/=powerB;
127 
128  if(powerBx-powerAx >= dpi)
129  {
130  if(powerA>=powerB)
131  {
132  powerx-=dpi;
133  }
134  else
135  {
136  powerx+=dpi;
137  }
138  }
139  }
140  powery/=power;
141 
142  out.totalPower=power;
143  out.centroidRange=powery;
144  out.centroidAzimuth=powerx;
145 
146  out.peakPower=peak;
147  out.peakRange=peakRange;
148  out.peakAzimuth=peakAz;
149 
150  // Check for sharp fall-off at the edges ?? Average over a few PRI's to avoid modulation noise ??
151  //
152 
153  size_t meanx=size_t(powerx+0.5);
154  size_t meany=size_t(powery+0.5);
155  const VideoT thresh=VideoT(power * minPower->getValue() / out.getCellCount());
156  out.distMinRange=findRangeEnd(meanx, meany, -deltaRange->getValue(), -1, thresh);
157  out.distMaxRange=findRangeEnd(meanx, meany, deltaRange->getValue(), rangeGates, thresh);
158  out.distMinAzimuth=findAzEnd(meany, meanx, -deltaAz->getValue(), -1, azimuthEncodings-1, thresh);
159  out.distMaxAzimuth=findAzEnd(meany, meanx, deltaAz->getValue(), azimuthEncodings, 0, thresh);
160 
161  return send(output);
162 }

165 int
166 SegmentStats::findRangeEnd(size_t az, size_t start, int maxDistance, int bound,
167  VideoT threshold)
168 {
169  int dir=1;
170  if(maxDistance<0)
171  dir=-1;
172  int stop=start+maxDistance;
173  int shortStop=stop;
174  if(dir*stop > dir*bound)
175  shortStop=bound;
176 
177  double maxDrop=maxRangeDrop->getValue();
178 
179  VideoT p0=buffer.get(az, start);
180  int range;
181  for(range=start; range!=shortStop; range+=dir)
182  {
183  // check
184  VideoT p=buffer.get(az, range);
185  if(p<threshold || p<maxDrop*p0)
186  return range-start;
187  p0=p;
188  }
189 
190  return range-start;
191 }

195 SegmentStats::findAzEnd(size_t range, size_t start, int maxDistance, int bound,

199  if(maxDistance<0)
200  dir=-1;
201 
202  int stop=start+maxDistance;
203  int shortStop=stop;
204  if(dir*stop > dir*bound)
205  shortStop=bound;
206 
207  double maxDrop=maxAzDrop->getValue();
208 
209  VideoT p0=buffer.get(start, range);
210  for(int az=start; az!=shortStop; az+=dir)
211  {
212  // check
213  VideoT p=buffer.get(az, range);
214  if(p<threshold || p<maxDrop*p0)
215  return az-start;
216  p0=p;
217  }
218 
219  if(shortStop!=stop)
220  {
221  int newStop=stop-shortStop;
222 
223  for(int az=restart; az!=newStop; az+=dir)
224  {
225  // check
226  VideoT p=buffer.get(az, range);
227  if(p<threshold || p<maxDrop*p0)
228  return (az-restart)+(shortStop-start);
229  p0=p;
230  }
231  }
232 
233  return maxDistance;
234 }
