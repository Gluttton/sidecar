60 PeakBarRenderer::calculateGeometry(const QMatrix& transform,
61  const Messages::PRIMessage::Ref& msg,
62  bool showRanges, bool showVoltages)
63 {
64  static Logger::ProcLog log("calculateGeometry", Log());
65  LOGINFO << msg << std::endl;
66 
67  double x = 0.0;
68  double width = width_;
69 
70  if (showRanges) {
71  x = msg->getRangeMin();
72  width = msg->getRangeFactor() * width;
73  }
74 
75  // See if we need to recalculate bar dimensions due to transform or bar width changes.
76  //
77  if (transformDirty_) {
78  transformDirty_ = false;
79  allDirty_ = true;
80 
81  // Calculate the bar dimensions, translating from real-world values.
82  //
83  origin_ = transform.map(QPointF(0, 0));
84  qreal barWidth, barHeight;
85  transform.map(width, 0, &barWidth, &barHeight);
86  barWidth -= origin_.x();
87  barHeight = barWidth * 0.6180339888 / 2.0; // golden-ratio / 2
88 
89  // If the bar height is too small for a rectangle fill, force to be a line.
90  //
91  if (barHeight < 1.5)
92  barHeight = 1.0;
93  else if (barHeight > 4.0)
94  barHeight = 4.0;
95 
96  barSize_ = QSizeF(barWidth, barHeight);
97  }
98 
99  VideoChannel& channel(channelConnection_.getChannel());
100 
101  bool usingLines = barSize_.height() == 1.0;
102  for (int index = 0; index < cache_.size(); ++index) {
103 
104  if (allDirty_ ||
105  (! usingLines && rects_[index].isNull()) ||
106  (usingLines && lines_[index].isNull())) {
107 
108  int sample = cache_[index].getValue();
109  LOGDEBUG << "index: " << index << " sample: " << sample
110  << std::endl;
111 
112  qreal value = showVoltages ?
113  channel.getVoltageForSample(sample) : sample;
114 
115  LOGDEBUG << "value: " << value << std::endl;
116 
117  QPointF from = transform.map(QPointF(x, value));
118  if (usingLines) {
119  QPointF to(from);
120  to.rx() += barSize_.width();
121  lines_[index] = QLineF(from, to);
122  }
123  else {
124  from.ry() -= barSize_.height();
125  rects_[index] = QRectF(from, barSize_);
126  }
127  }
128 
129  x += width;
130  }
131 
132  dirty_ = false;
133  allDirty_ = false;
134 }
