323 void
324 MainWindow::simulatorTimeout()
325 {
326  double dt = timer_->interval() / 1000.0;
327  elapsedTime_ += dt;
328  elapsed_->setText(QString::number(elapsedTime_, 'f', 3));
329 
330  Segment* segment = model_->getActiveSegment(elapsedTime_);
331  if (! segment) {
332  stopTimer();
333  return;
334  }
335 
336  // Calculate the number of seconds under acceleration for this segment.
337  //
338  double t = segment->duration_ - (segment->end_ - elapsedTime_);
339  xyz_[0] += (segment->xVel_ + segment->xAcc_ * t) * dt;
340  xyz_[1] += (segment->yVel_ + segment->yAcc_ * t) * dt;
341  xyz_[2] += (segment->zVel_ + segment->zAcc_ * t) * dt;
342 
343  updateTargetPositionXYZ(true);
344 }

609 MainWindow::openFile(const QString& path)
610 {
611  Logger::ProcLog log("openFile", Log());
612  LOGINFO << "path: " << path << std::endl;
613 
614  if (timer_->isActive())
615  stopTimer();
616 
617  QFile file(path);
618  if (! file.open(QIODevice::ReadOnly | QIODevice::Text)) {
619  statusBar()->showMessage("Failed to open file!");
620  return;
621  }
622 
623  QByteArray line(file.readLine());
624  if (line.isEmpty()) {
625  statusBar()->showMessage("Empty file!");
626  return;
627  }
628 
629  QTextStream in(line);
630  int port, systemId, rangeFormat;
631  double rate, range, azimuth, elevation;
632  QString address;
633  in >> address >> port >> systemId >> rangeFormat >> range >> azimuth
634  >> elevation >> rate;
635 
636  port_->setValue(port);
637  systemId_->setValue(systemId);
638  rate_->setValue(rate);
639  rangeFormat_->setChecked(rangeFormat);
640  if (! address.isEmpty())
641  address_->setText(address);
642 
643  manualRange_->setValue(range);
644  manualAzimuth_->setValue(azimuth);
645  manualElevation_->setValue(elevation);
646 
647  model_->clear();
648  while (1) {
649  QByteArray line(file.readLine());
650  if (line.isEmpty())
651  break;
652 
653  QTextStream in(line);
654 
655  Segment* segment = new Segment;
656  in >> segment->xVel_ >> segment->xAcc_
657  >> segment->yVel_ >> segment->yAcc_
658  >> segment->zVel_ >> segment->zAcc_
659  >> segment->duration_;
660 
661  model_->addRow(segment);
662  }
663 
664  lastFile_ = path;
665  setWindowTitle(QString("RangeTruthEmitter - %1[*]").arg(
666  strippedName(path)));
667 
668  QSettings settings;
669  QStringList files = settings.value("RecentFileList").toStringList();
670  files.removeAll(path);
671  files.prepend(path);
672  while (files.size() > kMaxRecentFiles)
673  files.removeLast();
674 
675  settings.setValue("RecentFileList", files);
676  updateRecentFileActions();
677  setWindowModified(false);
