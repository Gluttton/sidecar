207 int
208 StateEmitter::Private::svc()
209 {
210  Logger::ProcLog log("svc", Log());
211 
212  ACE_Message_Block* data = 0;
213  Destination* dest = 0;
214 
215  // Fetch data blocks from our processing queue. Keep going until our queue returns an error because it has
216  // been deactivated.
217  //
218  while (getq(data) != -1) {
219  switch(data->msg_type()) {
220  case ACE_Message_Block::MB_DATA:
221 
222  // Emit the current status to all active destinations.
223  //
224  for (auto d: destinations_) sendTo(d, data);
225  break;
226 
227  case ACE_Message_Block::MB_START:
228 
229  // New Destination object to add. First make sure that it points to a unique location.
230  //
231  dest = reinterpret_cast<Destination*>(data->base());
232  for (size_t index = 0; index < destinations_.size(); ++index) {
233  if (destinations_[index]->serviceEntry_->getName() ==
234  dest->serviceEntry_->getName()) {
235  delete dest;
236  dest = 0;
237  break;
238  }
239  }
240 
241  if (dest) {
242  destinations_.push_back(dest);
243  bringUpToDate(dest);
244  }
245  break;
246 
247  case ACE_Message_Block::MB_STOP:
248 
249  // Destination object to remove. Scan the Destination vector for the entry to remove and then remove
250  // it.
251  //
252  dest = reinterpret_cast<Destination*>(data->base());
253  for (size_t index = 0; index < destinations_.size(); ++index) {
254  if (destinations_[index] == dest) {
255  destinations_.erase(destinations_.begin() + index);
256  delete dest;
257  break;
258  }
259  }
260  break;
261 
262  default:
263  break;
264  }
265 
266  data->release();
267  data = 0;
268  }
269 
270  return 0;
271 }

285 {
286  static Logger::ProcLog log("sendTo", Log());
287  Zeroconf::ServiceEntry::Ref serviceEntry = destination->serviceEntry_;
288 
289  LOGINFO << serviceEntry->getName() << " port: " << destination->address_.get_port_number() << ' '

291 
292  // If we've received more than 10 errors in a row, stop sending to this destination.
293  //

297  //

299  if (rc != ssize_t(data->size() - 1)) {

301  LOGERROR << serviceEntry->getName() << " failed send to "

304  }
