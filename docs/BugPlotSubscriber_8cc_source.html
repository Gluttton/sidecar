30 BugPlotReader::openAndInit(const Zeroconf::ServiceEntry::Ref& service)
31 {
32  Logger::ProcLog log("BugPlotReader::openAndInit", bugCollector_.getLog());
33  serviceName_ = service->getName();
34 
35  const Zeroconf::ResolvedEntry& resolved = service->getResolvedEntry();
36 
37  ACE_INET_Addr address;
38  if (address.set(resolved.getPort(), resolved.getHost().c_str(), 1, PF_INET) == -1) {
39  LOGERROR << "invalid address: " << resolved.getHost() << '/' << resolved.getPort() << std::endl;
40  return false;
41  }
42 
43  if (! reader_.join(address)) {
44  LOGERROR << "failed mcast join using address " << resolved.getHost() << '/' << resolved.getPort()
45  << std::endl;
46  return false;
47  }
48 
49  // Obtain the heart-beat port for us to send heart-beat messages to.
50  //
51  std::istringstream is(resolved.getTextEntry("HeartBeatPort"));
52  uint16_t port;
53  is >> port;
54  heartBeatAddress_.set(port, resolved.getNativeHost().c_str(), 1, PF_INET);
55 
56  if (reactor()->register_handler(this, ACE_Event_Handler::READ_MASK) == -1) {
57  LOGERROR << "failed to register input handler" << std::endl;
58  return false;
59  }
60 
61  sendHeartBeat();
62 
63  return true;
64 }
