131 void
132 TrackMaintainer::checkDatabase()
133 {
134  static Logger::ProcLog log("checkDatabase", getLog());
135 
136  LOGINFO << "track database has " << trackDatabase_.size() << " entries"
137  << std::endl;
138 
139  double dropLimit = RadarConfig::GetRotationDuration() *
140  missesBeforeDrop_->getValue();
141  LOGDEBUG << "drop duration " << dropLimit << std::endl;
142 
143  // Loop through the data base.
144  //
145  Mapping::iterator it = trackDatabase_.begin();
146  while (it != trackDatabase_.end()) {
147 
148  LOGDEBUG << "track database entry " << it->first
149  << std::endl;
150 
151  const TrackMsgVector& tracks(it->second);
152  const Messages::Track::Ref& track(tracks.back());
153 
154  Messages::Track::Ref report;
155 
156  // Check to see if track is promotable
157  //
158  if (track->getType() == Track::kTentative) {
159  LOGDEBUG << "tentative track with " << tracks.size()
160  << " msgs. Threshold= " << hitsBeforePromote_->getValue()
161  << std::endl;
162  if (tracks.size() >= hitsBeforePromote_->getValue()) {
163  LOGDEBUG << "track should be promoted " << std::endl;
164  report = Track::Make("TrackMaintainer", track);
165  report->setFlags(Track::kPromoted);
166  }
167  }
168 
169  // Check to see if track should be dropped
170  //
171  double now = Time::TimeStamp::Now().asDouble();
172  if ((now - (epoch_ + track->getExtractionTime())) > dropLimit) {
173 
174  LOGWARNING << "dropping track " << it->first << std::endl;
175 
176  report = Track::Make("TrackMaintainer", track);
177  report->setFlags(Track::kDropping);
178 
179  // NOTE: erasing the current item only affects the current iterator but no others, so we
180  // post-increment so that we have a valid iterator to the next element after the erasure.
181  //
182  trackDatabase_.erase(it++);
183  }
184  else {
185 
186  // Not deleting track, so safe to advance the iterator to the next one.
187  //
188  ++it;
189  }
190 
191  // If we created a report above, send it out.
192  //
193  if (report) {
194  send(report);
195 
196  std::string flag;
197  switch (report->getFlags()) {
198  case Track::kDropping: flag = " dropping"; break;
199  case Track::kNew: flag = "new"; break;
200  case Track::kPromoted: flag = "promoted"; break;
201  case Track::kNeedsPrediction: flag = "needs prediction"; break;
202  case Track::kNeedsCorrection: flag = "needs correction"; break;
203  case Track::kPredicted: flag = "predicted"; break;
204  case Track::kCorrected: flag = "corrected"; break;
205  default: break;
206  };
207 
208  std::string type;
209  switch (report->getType()) {
210  case Track::kTentative: type = "tentative"; break;
211  case Track::kConfirmed: type = "confirmed"; break;
212  default: break;
213  };
214 
215  LOGDEBUG << "maintained track: " << report->getTrackNumber()
216  << " flag: " << flag << " type: " << type
217  << std::endl;
218  }
219  }
220 }
