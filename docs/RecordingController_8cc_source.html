114 RecordingController::restore(const QString& path)
115 {
116  Logger::ProcLog log("restore", Log());
117  LOGINFO << "path: " << path << std::endl;
118 
119  QStringList nameFilters;
120  nameFilters << QDateTime::currentDateTime().toUTC().toString(
121  "yyyyMMdd-*");
122  LOGDEBUG << "filter: " << nameFilters[0] << std::endl;
123 
124  QDir recordingDir(path);
125  recordingDir.setNameFilters(nameFilters);
126 
127  QFileInfoList entries = recordingDir.entryInfoList(QDir::NoFilter,
128  QDir::Name);
129  LOGDEBUG << "num entries: " << entries.size() << std::endl;
130 
131  QStringList failed;
132  foreach (QFileInfo fileInfo, entries) {
133  LOGDEBUG << "recordingDir: " << fileInfo.fileName() << std::endl;
134  QDir dir(fileInfo.filePath());
135  RecordingInfo* info = model_->getModelData(dir.dirName());
136  if (info) {
137  info->addRecordingDirectory(dir);
138  }
139  else {
140  info = new RecordingInfo(*this, dir);
141  model_->add(info);
142  if (info->getElapsedTime().isEmpty()) {
143  failed << info->getName();
144  }
145 
146  // Get a sequence ID from the directory name, which is formatted as YYYYMMDD-HHMMSS-ID. Note that
147  // the sequence ID may not exist for old-style recording names so play safe.
148  //
149  QStringList bits = dir.dirName().split("-");
150  if (bits.size() > 2) {
151  bool ok = false;
152  int sequenceId = bits[2].toInt(&ok);
153  if (ok && sequenceId >=
154  mainWindow_.recordingSequenceId_->value())
155  mainWindow_.recordingSequenceId_->setValue(
156  sequenceId + 1);
157  }
158  }
159  }

258 RecordingController::Status
259 RecordingController::start(const QStringList& configNames)
260 {
261  Logger::ProcLog log("start", Log());
262  LOGINFO << std::endl;
263 
264  if (configNames.empty()) {
265  LOGERROR << "no configurations" << std::endl;
266  return kNoLoadedConfig;
267  }
268 
269  QStringList doneDirs;
270  QStringList recordingPaths;
271 
272  // Create the name of the new recording. Our recording names have the follwoing format:
273  // YYYYMMDD-HHMMSS-ID[-CAL] where ID is an increasing sequence ID and CAL is an optional tag that
274  // indicates the recording is a calibration. NOTE: if you change the formatting of the recording names,
275  // you must change the Playback application as well since it relies on this formatting.
276  //
277  QString recordingName(
278  QDateTime::currentDateTime().toUTC().toString("yyyyMMdd-hhmmss"));
279  recordingName += QString("-%1").arg(
280  mainWindow_.recordingSequenceId_->value());
281 
282  // Scan the configurations to see if any of them are in the calibration state. We only want to append 'CAL'
283  // once.
284  //
285  foreach (QString configName, configNames) {
286  if (mainWindow_.isCalibrating(configName)) {
287  recordingName += "-CAL";
288  break;
289  }
290  }
291 
292  QString recordingPath;
293  QDir recordingDir;
294  activeRecording_ = 0;
295 
296  // Prepare for a new recording. We visit each of the recordable configurations since each one may have a
297  // different destination for recording.
298  //
299  foreach (QString configName, configNames) {
300  ConfigurationInfo* configuration =
301  mainWindow_.getConfigurationInfo(configName);
302 
303  // Get the recording directory to use.
304  //
305  QString parentDir(configuration->getRecordingsDirectory());
306 
307  // Only process the directory once, no matter how many configurations use it.
308  //
309  if (! doneDirs.contains(parentDir)) {
310 
311  // Attempt to create a new recording directory.
312  //
313  QFileInfo filePath(parentDir, recordingName);
314  recordingPath = filePath.filePath();
315  LOGDEBUG << "recordingPath: " << recordingPath << std::endl;
316 
317  QDir dir(parentDir);
318  if (! dir.mkdir(recordingName)) {
319  LOGERROR << "failed to create new recording directory"
320  << std::endl;
321  return kFailedCreateDirectory;
322  }
323 
324  recordingDir.setPath(recordingPath);
325 
326  // Allow everyone to read/write to the new directory.
327  //
328  if (QProcess::execute(QString("chmod a+rw %1")
329  .arg(recordingPath)) != 0)
330  return kFailedRecordingSetup;
331 
332  // Remove any old 'last' soft link.
333  //
334  if (QProcess::execute(
335  QString("rm -f %1/last").arg(parentDir)) != 0)
336  return kFailedRecordingSetup;
337 
338  // Create 'last' soft link to the new recording directory
339  //
340  if (QProcess::execute(
341  QString("ln -s %1 %2/last").arg(recordingPath)
342  .arg(parentDir)) != 0)
343  return kFailedRecordingSetup;
344 
345  if (! activeRecording_) {
346 
347  // Create a new RecordingInfo object to represent the pending recording.
348  //
349  QTime duration;
350  if (mainWindow_.recordingDurationEnable_->isChecked())
351  duration = mainWindow_.recordingDuration_->time();
352  activeRecording_ = new RecordingInfo(*this, recordingName,
353  duration);
354  }
355 
356  activeRecording_->addRecordingDirectory(recordingDir);
357  doneDirs.append(parentDir);
358  }
359 
360  // Copy over the configuration file to the new recording directory.
361  //
362  QFileInfo filePath(configuration->getPath());
363  if (! QFile::copy(filePath.absoluteFilePath(),
364  recordingDir.absoluteFilePath(
365  filePath.fileName())))
366  return kFailedRecordingSetup;
367 
368  // Save the config name and recording path for use later when we command the runners to begin recording.
369  //
370  recordingPaths.append(recordingPath);
371  activeRecording_->addConfiguration(configName);
372  }
373 
374  // Fetch any runtime parameter deviations from startup/configured values and add them to the notes window
375  // for historical purposes.
376  //
377  QStringList parameterChanges;
378  mainWindow_.getChangedParameters(configNames, parameterChanges);
379  activeRecording_->start(parameterChanges);
380 
381  // Finally, command the tasks to start recording
382  //
383  if (! mainWindow_.startRecording(configNames, recordingPaths)) {
384  LOGERROR << "failed to command runners to start recording" << std::endl;
385  return kFailedPostRecordingStateChange;
386  }
387 
388  activeRecording_->showNotesWindow();
389  lastRecording_ = activeRecording_;
390 
391  model_->add(activeRecording_);
392  mainWindow_.recordings_->scrollToBottom();
393 
394  // Update GUI to reflect the change in recording state.
395  //
396  mainWindow_.recordingStartStop_->setText(kStop);
397  mainWindow_.actionRecordingStartStop_->setText(kStop);
398  mainWindow_.setRecordingElapsed(
399  activeRecording_->getFormattedElapsedTime(),
400  activeRecording_->getRemainingTime());
401  mainWindow_.recordingDurationEnable_->setEnabled(false);
402  mainWindow_.actionRecordingLimitDuration_->setEnabled(false);
403  mainWindow_.recordingDuration_->setEnabled(false);
404  mainWindow_.recordingSequenceId_->stepUp();
405 
406  LOGINFO << "started recording" << std::endl;
407 
408  emit statusChanged(configNames, true);
409 
410  return kOK;
411 }

493 void
494 RecordingController::deleteSelectedRecording()
495 {
496  // Obtain the current selection
497  //
498  QItemSelectionModel* selectionModel =
499  mainWindow_.recordings_->selectionModel();
500  QModelIndex index = selectionModel->currentIndex();
501  if (! index.isValid()) return;
502 
503  // Only allow deletion if it is not recording.
504  //
505  RecordingInfo* obj = model_->getModelData(index.row());
506  if (! obj->isDone()) return;
507 
508  // Make sure the user really wants to do this!
509  //
510  if (QMessageBox::question(
511  qApp->activeWindow(), "Delete Recording",
512  QString("<p>Delete Recording</p>"
513  "<p>Are you sure you want to delete "
514  "the recording '%1' and all of the "
515  "data files it contains?</p>")
516  .arg(obj->getName()), QMessageBox::No | QMessageBox::Yes,
517  QMessageBox::No) == QMessageBox::No) return;
518 
519  QApplication::setOverrideCursor(Qt::WaitCursor);
520  if (obj == lastRecording_)
521  lastRecording_ = 0;
522 
523  // Get the list of directories we need to remove. There may be more than one since multiple configuration
524  // files may have been involved in the recording with different recording base paths. Remove the
525  // directories.
526  //
527  QStringList dirPaths(obj->getRecordingDirectories());
528  foreach (QDir dir, dirPaths)
529  RemoveDirectory(dir);
530 
531  // Remove the item from the model and attempt to select another entry.
532  //
533  selectionModel->clear();
534  model_->remove(index);
535  if (model_->rowCount() > 0) {
536  if (model_->rowCount() <= index.row())
537  index = model_->index(model_->rowCount() - 1, 0);
538  selectionModel->setCurrentIndex(
539  index, QItemSelectionModel::SelectCurrent |
540  QItemSelectionModel::Rows);
541  }
542 
543  QApplication::restoreOverrideCursor();
544 }

585 RecordingController::adjustColumnSizes()
586 {
587  Logger::ProcLog log("adjustColumnSizes", Log());
588  LOGINFO << "begin" << std::endl;
589 
590  // Approach: fetch the width of the view port, subtract the column width for all columns but the first one
591  // (name), and allocate the remaining space to the first one if it is greater than its minimum width.
592  //
593  QTableView* recordings = mainWindow_.recordings_;
594  recordings->resizeColumnsToContents();
595  int minimum = recordings->columnWidth(RecordingModel::kConfig);
596  int available(recordings->viewport()->width());
597  for (int column = 0; column < model_->columnCount(); ++column)
598  if (column != RecordingModel::kConfig)
599  available -= recordings->columnWidth(column);
600  if (available >= minimum)
601  recordings->setColumnWidth(RecordingModel::kConfig, available);
602 }
