92 int
93 Recorder::svc()
94 {
95  // !!! Running in a separate thread here -- use protection...
96  //
97  Logger::ProcLog log("svc", Log());
98  LOGINFO << "starting" << std::endl;
99 
100  // Create a gather writer that will write out in 32K bursts. Should be a configurable parameter to allow for
101  // testing.
102  //
103  IO::GatherWriter gatherWriter(writer_);
104  gatherWriter.setSizeLimit(32 * 1024);
105 
106  // Main processing loop. Fetch from the input queue and add to a GatherWriter object. A failure to fetch
107  // means that the queue has been deactivated by the producer thread as a signal for us to quit. A failure of
108  // the GatherWriter probably means we ran out of space.
109  //
110  ACE_Message_Block* data = 0;
111  while (gatherWriter.isOK() && getq(data) != -1) {
112  LOGDEBUG << "msg: " << data << std::endl;
113  IO::MessageManager mgr(data);
114  if (! gatherWriter.add(mgr.getEncoded())) {
115  msg_queue()->deactivate();
116  }
117  }
118 
119  // If our GatherWriter is still OK, then the producer thread told to quit. Continue writing out anything in
120  // our queue.
121  //
122  if (gatherWriter.isOK()) {
123 
124  // Flush anything remaining in the queue to disk. Its OK to reactivate now since at this point the
125  // producer thread is waiting for us to exit and will not add anything to the queue.
126  //
127  if (! msg_queue()->is_empty()) {
128  msg_queue()->activate();
129  while (1) {
130  int remaining = getq(data);
131  IO::MessageManager mgr(data);
132  gatherWriter.add(mgr.getEncoded());
133  if (! remaining) break;
134  }
135  msg_queue()->deactivate();
136  }
137  gatherWriter.flush();
138  }
139  else {
140 
141  // Our gatherWriter has failed, so deactivate our queue so future puts will fail, which alert the
142  // producer thread that something bad has happened.
143  //
144  msg_queue()->deactivate();
145  }
146 
147  // Safe now to close the file connection. Use ::fsync to make sure we write everything to disk just to be
148  // safe. This is to be POSIX compliant.
149  //
150  ACE_OS::fsync(writer_.getDevice().get_handle());
151  writer_.getDevice().close();
152 
153  LOGINFO << "terminating" << std::endl;
154  return 0;
155 }
