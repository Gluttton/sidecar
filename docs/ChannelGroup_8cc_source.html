62  // it as we load the new service names.
63  //

73  LOGDEBUG << "current: " << current << std::endl;
74 
75  found_->clear();
76 
77  // Get the found keys and sort then alphabetically.
78  //
79  int active = 0;

81  qSort(keys.begin(), keys.end());
82 
83  // Add to the widget, remembering which one matches the one we would like to be current.
84  //
85  found_->addItem(" ");
86  for (int index = 0; index < keys.size(); ++index) {
87  if (keys[index] == current) active = index + 1;
88  found_->addItem(keys[index]);

90
