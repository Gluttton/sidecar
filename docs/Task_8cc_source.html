230 bool
231 Task::enterProcessingState(ProcessingState::Value goalState)
232 {
233  static Logger::ProcLog log("enterProcessingState", Log());
234 
235  LOGINFO << "task: " << taskName_ << " current: " << ProcessingState::GetName(processingState_)
236  << '/' << processingState_ << " goal: " << ProcessingState::GetName(goalState) << '/' << goalState
237  << std::endl;
238 
239  // Remember the new processing state for use with enterLastProcessingState()
240  //
241  if (ProcessingState::IsNormal(goalState)) {
242  lastProcessingState_ = goalState;
243  }
244 
245  // Keep looping as long as we have not encountered a failure, until we've reached the desired end state.
246  //
247  bool ok = true;
248  while (goalState != processingState_) {
249 
250  LOGDEBUG << "goalState: " << goalState << " processingState: " << processingState_ << std::endl;
251 
252  ProcessingState::Value nextState = transitions_[goalState][processingState_];
253 
254  LOGDEBUG << "current: " << ProcessingState::GetName(processingState_) << " next: "
255  << ProcessingState::GetName(nextState) << std::endl;
256 
257  switch (nextState) {
258  case ProcessingState::kInitialize: ok = enterInitializeState(); break;
259  case ProcessingState::kAutoDiagnostic: ok = enterAutoDiagnosticsState(); break;
260  case ProcessingState::kCalibrate: ok = enterCalibrateState(); break;
261  case ProcessingState::kRun: ok = enterRunState(); break;
262  case ProcessingState::kStop: ok = enterStopState(); break;
263  default: // Covers kInvalid and kFailure
264  ok = false;
265  break;
266  }
267 
268  if (! ok) break;
269 
270  processingState_ = nextState;
271  }
272 
273  if (! ok) {
274 
275  // We failed somewhere above. Enter the failure state.
276  //
277  LOGDEBUG << "entering FAILURE state" << std::endl;
278  if (! hasError()) setError("Unknown");
279  enterFailureState();
280  }
281 
282  // Update the runtime parameter value with any changes. Only a subset of the values are valid for the user
283  // to pick from.
284  //
285  if (processingState_ < ProcessingStateEnumTraits::GetMinValue() ||
286  processingState_ > ProcessingStateEnumTraits::GetMaxValue()) {
287  processingStateParameter_->setValue(ProcessingStateEnumDef::None());
288  }
289  else {
290  processingStateParameter_->setValue(processingState_);
291  }
292 
293  LOGDEBUG << "new state: " << ProcessingState::GetName(processingState_) << std::endl;
294 
295  updateUsingDataValue();
296 
297  return true;
298 }

391 int
392 Task::put(ACE_Message_Block* data, ACE_Time_Value* timeout)
393 {
394  static Logger::ProcLog log("put", Log());
395  LOGINFO << taskName_ << " msg_type: " << data->msg_type() << std::endl;
396 
397  if (MessageManager::IsControlMessage(data)) {
398 
399  // Forward any control messages down the processing chain. Create a duplicate message because we want to
400  // keep ownership of the original. Do not do this for timeout messages, since those are local to this
401  // task.
402  //
403  if (MessageManager::GetControlMessageType(data) != ControlMessage::kTimeout){
404  ACE_Message_Block* dup = data->duplicate();
405  if (put_next(dup) == -1) {
406  dup->release();
407  if (next()) LOGWARNING << "failed put_next() for control message" << std::endl;
408  }
409  }
410 
411  // Now deliver the original to ourselves.
412  //
413  return deliverControlMessage(data, timeout) ? 0 : -1;
414  }
415  else if (MessageManager::IsDataMessage(data)) {
416 
417  // Data message delivery. Extract the channel index from the message and use to update the channel
418  // stats.
419  //
420  MessageManager mgr(data->duplicate());
421  size_t channelIndex = data->msg_priority();
422  Header::Ref msg = mgr.getNative();
423  updateInputStats(channelIndex, msg->getSize(), msg->getMessageSequenceNumber());
424 
425  // Attempt to deliver the message to the task. For threaded tasks, this will insert the message into a
426  // work queue for the thread to process.
427  //
428  if (! deliverDataMessage(data, timeout)) {
429  LOGWARNING << "failed deliverDataMessage()" << std::endl;
430  setError("Failed to deliver message to task");
431  return -1;
432  }
433 
434  data = 0;
435  }
436  else {
437  LOGERROR << "unknown messsage type - " << data->msg_type() << std::endl;
438  }
439 
440  if (data) data->release();
441 
442  return 0;
443 }

500 bool
501 Task::processControlMessage(ACE_Message_Block* data)
502 {
503  static Logger::ProcLog log("processControlMessage", Log());
504  MessageManager mgr(data->duplicate());
505  LOGTIN << taskName_ << " control message: " << mgr.getControlMessageType() << std::endl;
506 
507  // Dispatch on control message type
508  //
509  bool ok = false;
510  switch (mgr.getControlMessageType()) {
511  case ControlMessage::kParametersChange:
512  ok = doParametersChange(mgr.getControlMessage<ParametersChangeRequest>());
513  break;
514 
515  case ControlMessage::kProcessingStateChange:
516  ok = doProcessingStateChange(mgr.getControlMessage<ProcessingStateChangeRequest>());
517  break;
518 
519  case ControlMessage::kRecordingStateChange:
520  ok = doRecordingStateChange(mgr.getControlMessage<RecordingStateChangeRequest>());
521  break;
522 
523  case ControlMessage::kShutdown:
524  ok = doShutdownRequest();
525  break;
526 
527  case ControlMessage::kClearStats:
528  ok = doClearStatsRequest();
529  break;
530 
531  case ControlMessage::kTimeout:
532  ok = doTimeout();
533  break;
534 
535  default:
536  LOGERROR << "unknown control message type - " << mgr.getControlMessageType() << std::endl;
537  setError("Unknown control message type");
538  break;
539  }
540 
541  if (ok) {
542 
543  // Obey the SideCar programming contract for message passing. If we return success, we own the incoming
544  // message.
545  //
546  data->release();
547  }
548 
549  return ok;
550 }

552 bool
553 Task::doParametersChange(const IO::ParametersChangeRequest& msg)
554 {
555  static Logger::ProcLog log("doParametersChange", Log());
556  LOGINFO << std::endl;
557 
558  XmlRpc::XmlRpcValue params(msg.getValue());
559  for (int index = 0; index < params.size();) {
560 
561  // Fetch the name of the parameter to change, then the XML-RPC value to apply.
562  //
563  std::string name(params[index++]);
564  LOGDEBUG << "name: " << name << std::endl;
565  if (index == params.size()) {
566  LOGERROR << "missing value for parameter '" << name << "'" << std::endl;
567  return true;
568  }
569 
570  XmlRpc::XmlRpcValue value = params[index++];
571 
572  // Locate the parameter with the given name
573  //
574  ParameterMap::const_iterator pos = parameterMap_.find(name);
575  if (pos != parameterMap_.end()) {
576  try {
577  if (msg.hasOriginalValues()) {
578  parameterVector_[pos->second]->setXMLOriginal(value);
579  LOGWARNING << taskName_ << " changed parameter " << name << " to " << value << std::endl;
580  }
581  else {
582  parameterVector_[pos->second]->setXMLValue(value);
583  LOGWARNING << taskName_ << ' ' << name << " = " << value << std::endl;
584  }
585  }
586  catch (Parameter::InvalidValue& err) {
587  LOGERROR << err.what() << std::endl;
588  setError(err.what());
589  return true;
590  }
591  }
592  else {
593  LOGERROR << "Unknown parameter name '" << name << "'" << std::endl;
594  }
595  }
596 
597  updateUsingDataValue();
598 
599  return true;
600 }

642 bool
643 Task::sendManaged(MessageManager& manager, size_t channelIndex)
644 {
645  bool ok = true;
646 
647  // If there is nothing after us, just eat the message.
648  //
649  if (! next())
650  return true;
651 
652  // If the channel is a valid index, then have its recipient list handle delivery.
653  //
654  if (channelIndex < outputs_.size()) {
655  ok = outputs_.getChannel(channelIndex).deliver(
656  manager.getMessage());
657  }
658 
659  // !!! Hack to keep some old unit tests running. Since Runner validates task connections, this should never
660  // happen from well-formed XML configurations.
661  //
662  else if (channelIndex == 0) {

664  if (data) {
665  if (put_next(data, 0) == -1) {
666  data->release();
667  ok = false;
668  }
669  }
670  }
671

675 void
676 Task::fillStatus(StatusBase& status)
677 {
678  static Logger::ProcLog log("fillStatus", Log());
679 
680  size_t pendingQueueCount = msg_queue()->message_count();
681  if (pendingQueueCount) {
682  LOGDEBUG << "input queue size: " << pendingQueueCount << std::endl;
683  }
684 
685  status.setSlot(TaskStatus::kProcessingState, processingState_);
686  status.setSlot(TaskStatus::kError, error_);
687  status.setSlot(TaskStatus::kConnectionInfo, connectionInfo_);
688  status.setSlot(TaskStatus::kPendingQueueCount, int(pendingQueueCount));
689 
690  // Only allow parameter editing if there are more than two (there is always the editingEnabled_ parameter,
691  // but users cannot change it via XML-RPC), and there is the processingState parameter, but we only allow
692  // editing of it for algorithms
693  //
694  status.setSlot(TaskStatus::kHasParameters, parameterVector_.size() > taskParameterCount_);
695  status.setSlot(TaskStatus::kUsingData, usingData_);
696 
697  // Calculate message counts and rates from the Stats object associated with each input.
698  //
699  size_t messageCount = 0;
700  size_t byteRate = 0;
701  size_t messageRate = 0;
702  size_t dropCount = 0;
703  size_t dupeCount = 0;
704  size_t inputCount = inputStats_.size();
705  for (size_t index = 0; index < inputCount; ++index) {
706  Stats& s(inputStats_[index]);
707  s.calculateRates();
708  messageCount += s.getMessageCount();
709  byteRate += s.getByteRate();
710  messageRate += s.getMessageRate();
711  dropCount += s.getDropCount();
712  dupeCount += s.getDupeCount();
713  }
714 
715  status.setSlot(TaskStatus::kMessageCount, int(messageCount / inputCount));
716  status.setSlot(TaskStatus::kByteRate, int(byteRate / inputCount));
717  status.setSlot(TaskStatus::kMessageRate, int(messageRate / inputCount));
718  status.setSlot(TaskStatus::kDropCount, int(dropCount / inputCount));
719  status.setSlot(TaskStatus::kDupeCount, int(dupeCount / inputCount));
720 }
