170 void
171 SamplePlotWidget::paintEvent(QPaintEvent* event)
172 {
173  QPainter painter(this);
174  painter.fillRect(0, 0, width(), height(), Qt::black);
175 
176  // Default scaling will be expected_ +/- expectedVariance. However, if our min or max value is outside this
177  // range, then the scaling will expand to to it.
178  //
179  double span = expectedVariance_ * 2;
180  double min = expected_ - expectedVariance_;
181  if (plotStats_) {
182  double valueMax = plotStats_->getMaximumValue();
183  if (expected_ > valueMax) valueMax = expected_;
184  double valueMin = plotStats_->getMinimumValue();
185  if (expected_ < valueMin) valueMin = expected_;
186  double valueSpan = plotStats_->getMaximumValue() -
187  plotStats_->getMinimumValue();
188  if (valueSpan > span) {
189  span = valueSpan;
190  min = plotStats_->getMinimumValue();
191  }
192  }
193 
194  // Calculate conversion parameters, scale sample values into pixel Y positions, and draw lines connecting
195  // the sample values.
196  //
197  double offset = (height() - 1);
198  double scale = offset / span;
199  offset += min * scale;
200  size_t count = plotDataPoints_.size();
201  for (size_t index = 0; index < count; ++index) {
202  double value = offset - scale * plotDataPoints_[index].sample;
203  samplePoints_[index].setY(static_cast<int>(::rint(value)));
204  }
205 
206  // Draw a line showing the expected (nominal) value in the plot.
207  //
208  int expected = static_cast<int>(::rint(offset - scale * expected_));
209  painter.setPen(Qt::gray);
210  painter.drawLine(0, expected, width() - 1, expected);
211 
212  // Generate tick marks to show elapsed time in seconds on the plot.
213  //
214  painter.setFont(font());
215  int x = kTickSpan;
216  Time::TimeStamp now = Time::TimeStamp::Now();
217  int numDataPoints = plotDataPoints_.size();
218  while (x < width() && x < numDataPoints) {
219  double seconds = (now - plotDataPoints_[x].when).asDouble();
220  QString text = QString("%1s").arg(int(::rint(seconds)));
221  QRect rect(painter.boundingRect(0, 0, width(), height(), 0, text));
222  int v = height() - rect.height() + 1;
223  painter.drawText(x - rect.width() / 2, v, rect.width(), rect.height(),
224  0, text);
225  painter.drawLine(x, expected + 10, x, expected - 10);
226  x += kTickSpan;
227  }
228 
229  painter.setPen(samplesColor_);
230  painter.drawPolyline(&samplePoints_[0], count);
231 }
