234 TargetPlotImaging::render(QWidget* widget, const TargetPlotList& targets)
235 {
236  static Logger::ProcLog log("render", Log());
237 
238  if (! plotPositionFunctor_)
239  return;

241  // Fetch the matrices involved in converting from model coordinates into window positions. This should be
242  // very light-weight, but if not, then create a new class that contains these values and provide them as an
243  // input into the render() method. We cannot cache these values within this class because it may be used by
244  // more than one view.
245  //
246  GLdouble modelMatrix[16];
247  glGetDoublev(GL_MODELVIEW_MATRIX, modelMatrix);
248  GLdouble projectionMatrix[16];
249  glGetDoublev(GL_PROJECTION_MATRIX, projectionMatrix);
250  GLint viewPort[4];
251  glGetIntegerv(GL_VIEWPORT, viewPort);
252 
253  // All of our points window coordinates. Save the current transformation matrix and set up an identity
254  // transform.
255  //
256  glMatrixMode(GL_PROJECTION);
257  glPushMatrix();
258  glLoadIdentity();
259  // gluOrtho2D(0, widget->width(), 0, widget->height());
260  glOrtho(0, widget->width(), 0, widget->height(), -1.0, 1.0);
261  glLineWidth(getLineWidth());
262 
263  if (symbolPoints_.empty())
264  scaleSymbolPoints();
265 
266  int kind = GetSymbolInfo(getSymbolType()).kind;
267  VertexColorTagArray tags;
268 
269  // Process oldest target first so that newer targets are not obscured by the older ones.
270  //
271  for (int targetIndex = 0; targetIndex < targets.size(); ++targetIndex) {
272 
273  const TargetPlot& target(targets[targetIndex]);
274 
275  // Obtained target plot color, faded by time if enabled.
276  //
277  Color color(getColor());
278  if (getFadeEnabled())
279  fadeColor(color, target.getAge());
280 
281  // Calculate the position of the target in window coordinates. This allows us to draw the symbols at the
282  // same size, regardless of the view settings.
283  //
284  Vertex center = plotPositionFunctor_->getPosition(target);
285  GLdouble x = center.x;
286  GLdouble y = center.y;
287 
288  GLdouble winX, winY, winZ;
289  //gluProject(x, y, 0.0, modelMatrix, projectionMatrix, viewPort,
290  //&winX, &winY, &winZ);
291  ProjectPoint(x, y, 0.0, modelMatrix, projectionMatrix, viewPort, &winX, &winY, &winZ);
292  LOGDEBUG << "xy: " << x << ' ' << y << " winXY: " << winX << ' ' << winY
293  << std::endl;
294 
295  center.x = winX;
296  center.y = winY;
297  for (size_t symbolIndex = 0; symbolIndex < symbolPoints_.size();
298  ++symbolIndex) {
299  points_.push_back(symbolPoints_[symbolIndex] + center, color);
300  }
301 
302  // Due to the definition for GL_LINE_LOOP, we cannot batch them up in the points_ vector and draw them
303  // all at once. We must do each individually.
304  //
305  if (kind == GL_LINE_LOOP) {
306  points_.draw(kind);
307  points_.clear();
308  }
309 
310  // Draw the target's alphanumeric tag if available and enabled.
311  //
312  if (showTags_ && showTags_->getValue() && target.getTag().size()) {
313  center.x += getExtent2();
314  center.y += getExtent2();
315  tags.push_back(center, color, target.getTag());
316  LOGDEBUG << "tag: " << target.getTag() << std::endl;
317  }
318  }
319 
320  // Draw the plot symbols.
321  //
322  if (! points_.empty()) {
323  points_.draw(kind);
324  points_.clear();
325  }
326 
327  // Now draw any tags.
328  //
329  if (! tags.empty())
330  plotPositionFunctor_->renderTags(tagSize_->getValue(), tags);
331 
332  glPopMatrix();
333 }

497  break;
498 
499  case GL_LINE_LOOP:

501  break;
502 
503  default:
504  break;
505  }
506 
507  painter.end();
508
