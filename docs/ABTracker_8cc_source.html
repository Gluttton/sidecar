104 bool
105 ABTracker::updateTracks(const Messages::Extraction& plot)
106 {
107  Logger::ProcLog log("updateTracks", getLog());
108  double when = plot.getWhen().asDouble();
109 
110  LOGINFO << "when: " << when
111  << " range: " << plot.getRange()
112  << " az: " << plot.getAzimuth()
113  << std::endl;
114 
115  Track* found = 0;
116  if (plot.getRange() >= minRange_->getValue()) {
117 
118  // We only associate with something if it has a proximity value smaller
119  // than associationRadius. We use the squared variants to save us from
120  // the costly square root.
121  //
122  double best = associationRadius2_;
123  Geometry::Vector v(plot.getX(), plot.getY(), plot.getElevation());
124 
125  // Visit all of the existing Track objects to find the one that best
126  // associates with the given Extraction plot.
127  //
128  TrackList::iterator pos = tracks_.begin();
129  TrackList::iterator end = tracks_.end();
130  while (pos != end) {
131  Track* track = *pos++;
132  double distance = track->getProximityTo(when, v);
133  if (distance < best) {
134  found = track;
135  best = distance;
136  }
137  }
138 
139  // None found, so create a new Track object that starts in the
140  // kInitiating state.
141  //
142  if (! found) {
143  found = new Track(*this, ++trackIdGenerator_, when , v);
144  LOGINFO << "created new track - " << found->getId() << std::endl;
145  tracks_.push_back(found);
146  ++trackCount_;
147  }
148  else {
149  LOGINFO << "found existing track - " << found->getId() << std::endl;
150 
151  // Apply the Extraction plot to the found track
152  //
153  found->updatePosition(when, v);
154  }
155  }
156 
157  // Revisit all of the tracks, emitting TSPI messages for those that are
158  // dropping and the one that was updated above.
159  //
160  bool ok = true;
161  TrackList::iterator pos = tracks_.begin();
162  TrackList::iterator end = tracks_.end();
163  while (pos != end) {
164  Track* track = *pos;
165  if (track == found || track->isDropping()) {
166  if (! track->emitPosition())
167  ok = false;
168  if (track->isDropping()) {
169  LOGINFO << "dropping track " << track->getId() << std::endl;
170  delete track;
171  pos = tracks_.erase(pos);
172  --trackCount_;
173  continue;
174  }
175  }
176  ++pos;
177  }
178 
179  return ok;
180 }
