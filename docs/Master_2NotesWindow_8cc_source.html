202 void
203 NotesWindow::loadFromFile(QFile& file)
204 {
205  Logger::ProcLog log("loadFromFile", Log());
206 
207  gui_->recordingInfo_->setVisible(false);
208  gui_->actionStop_Recording->setEnabled(false);
209 
210  // Radar Transmitting
211  //
212  QByteArray data = file.readLine().trimmed();
213  QList<QByteArray> bits = data.split(' ');
214  if (bits.count() == 7) {
215  gui_->radarTransmitting_->setChecked(true);
216  gui_->radarFrequency_->setValue(bits[5].toDouble());
217  }
218  else {
219  gui_->radarTransmitting_->setChecked(false);
220  }
221 
222  // Radar Rotating
223  //
224  data = file.readLine().trimmed();
225  bits = data.split(' ');
226  if (bits.count() == 7) {
227  gui_->radarRotating_->setChecked(true);
228  gui_->radarRotationRate_->setValue(bits[5].toDouble());
229  } else {
230  gui_->radarRotating_->setChecked(false);
231  }
232 
233  // Using DRFM
234  //
235  data = file.readLine().trimmed();
236  bits = data.split(' ');
237  gui_->drfmOn_->setChecked(false);
238  if (bits.count() > 3) {
239  gui_->drfmOn_->setChecked(true);
240  QString tmp(bits[5]);
241  for (int index = 6; index < bits.size(); ++index) {
242  tmp.append(' ');
243  tmp.append(bits[index]);
244  }
245  gui_->drfmConfig_->setText(tmp.trimmed());
246  }
247  else {
248  gui_->drfmOn_->setChecked(false);
249  }
250 
251  file.readLine().trimmed();
252  gui_->notes_->document()->setPlainText(file.readAll());
253 
254  statsChanged();
255 }

289  .append(" rpm");

292  .append(gui_->drfmOn_->isChecked() ? "YES" : "NO");
293 
294  if (gui_->drfmOn_->isChecked())
295  data.append(" - Configuration: ")
296  .append(gui_->drfmConfig_->text());
297 
298  data.append("\n\n");

300 
301  foreach (QString path, info_.getRecordingDirectories()) {
302  QDir dir(path);
303  QFile file(dir.filePath("notes.txt"));

305  file.write(data);
