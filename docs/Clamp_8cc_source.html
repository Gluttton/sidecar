51 bool
52 Clamp::processInput(const Messages::Video::Ref& msg)
53 {
54  static Logger::ProcLog log("processInput", getLog());
55 
56  // If not enabled, simply pass message.
57  //
58  if (! enabled_->getValue()) {
59  return send(msg);
60  }
61 
62  // Create a new message to hold the output of what we do. Note that although we pass in the input message,
63  // the new message does not contain any data.
64  //
65  Messages::Video::Ref out(Messages::Video::Make("Clamp", msg));
66 
67  // Use the STL transform function to convert values from the input message into clamped values placed in the
68  // output message. The std::back_inserter method creates an output iterator that uses push_back() when
69  // inserting values.
70  //
71  using DatumType = Messages::Video::DatumType;
72  DatumType minValue = min_->getValue();
73  DatumType maxValue = max_->getValue();
74  std::transform(msg->begin(), msg->end(), std::back_inserter(out->getData()),
75  [minValue, maxValue](auto value) {
76  return std::max(minValue, std::min(maxValue, value));
77  });
78 
79  // Send out on the default output device, and return the result to our Controller. NOTE: for multichannel
80  // output, one must give a channel index to the send() method. Use getOutputChannelIndex() to obtain the
81  // index for an output channel with a given name.
82  //
83  bool rc = send(out);
84  LOGDEBUG << "rc: " << rc << std::endl;
85  return rc;
86 }
