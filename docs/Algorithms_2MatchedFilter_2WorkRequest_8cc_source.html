35  // We use an ACE_Message_Block to hold a WorkRequest object so that we may use ACE's thread-safe message

112 WorkRequest::process()
113 {
114  static Logger::ProcLog log("process", Log());
115 
116  LOGINFO << "offset: " << offset_ << std::endl;
117 
118  // Create complex values from our input message.
119  //
120  Messages::Video::const_iterator inputPos = input_->begin() + offset_;
121  Messages::Video::const_iterator inputEnd = input_->end();
122  size_t inputSize = inputEnd - inputPos;
123  if (inputSize > fftSize_ * 2)
124  inputEnd = inputPos + fftSize_ * 2;
125 
126  int index = 0;
127  while (inputPos < inputEnd) {
128  float i = *inputPos++;
129  float q = *inputPos++;
130  rxVec_->put(index++, ComplexType(i, q));
131  }
132 
133  // If necessary, pad the end with zeros.
134  //
135  while (index < fftSize_) {
136  rxVec_->put(index++, ComplexType(0.0, 0.0));
137  }
138 
139  // Filter the data in-place.
140  //
141  (*rxVec_) = (*invFFT_)((*txPulseVec_) * (*fwdFFT_)((*rxVec_)));
142 
143  // Replace the appropriate output message samples with the complex component values from the above filtering
144  // operation.
145  //
146  Messages::Video::iterator outputPos = output_->begin() + offset_;
147  Messages::Video::iterator outputEnd = output_->end();
148  size_t outputSize = outputEnd - outputPos;
149  if (outputSize > fftSize_ * 2)
150  outputEnd = outputPos + fftSize_ * 2;
151 
152  index = 0;
153  while (outputPos < outputEnd) {
154  *outputPos++ = Messages::Video::DatumType(
155  rxVec_->get(index).real());
156  *outputPos++ = Messages::Video::DatumType(
157  rxVec_->get(index).imag());
158  ++index;
159  }
160 }
