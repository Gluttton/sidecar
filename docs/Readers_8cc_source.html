95 bool
96 StreamReader::fetchInput()
97 {
98  static Logger::ProcLog log("fetchInput", Log());
99 
100  LOGDEBUG << "remaining: " << remaining_ << std::endl;
101 
102  while (remaining_) {
103 
104  // Fetch data from the device, storing at the end of the message block.
105  //
106  LOGDEBUG << "before fetchFromDevice" << std::endl;
107  ssize_t fetched = fetchFromDevice(building_->wr_ptr(), remaining_);
108  LOGDEBUG << "fetched: " << fetched << std::endl;
109 
110  switch (fetched) {
111  case -1: // device err
112  switch (errno) {
113  case EWOULDBLOCK:
114  case ETIME:
115  LOGDEBUG << "nothing available - " << Utils::showErrno()
116  << std::endl;
117  return true;
118  }
119  LOGERROR << "failed to fetch data - " << Utils::showErrno()
120  << std::endl;
121  return false;
122  break;
123 
124  case 0: // device EOF
125  return false;
126  break;
127  };
128 
129  // Adjust the write pointer by the number of bytes we read in.
130  //
131  building_->wr_ptr(fetched);
132  remaining_ -= fetched;
133  LOGDEBUG << "new remaining: " << remaining_ << std::endl;;
134 
135  // If we are have already fetched a Preamble or we don't have enough data to build a message, then stop
136  // looping.
137  //
138  LOGDEBUG << "count so far: " << building_->length() << std::endl;
139 
140  if (needed_ > Preamble::kCDRStreamSize || remaining_ > 0)
141  break;
142 
143  // If we are looking for the SYNCH word, see if we found it.
144  //
145  if (needed_ == sizeof(int16_t) * 2) {
146  int16_t* ptr = reinterpret_cast<int16_t*>(building_->rd_ptr());
147  int16_t magic = *ptr++;
148  int16_t byteOrder = *ptr++;
149  if (magic != int16_t(Preamble::kMagicTag) ||
150  (byteOrder != int16_t(0) &&
151  byteOrder != int16_t(0xFFFF))) {
152  if (! needSynch_) {
153  LOGERROR << "missing SYNCH" << std::endl;
154  needSynch_ = true;
155  }
156  building_->reset();
157  ACE_CDR::mb_align(building_);
158  }
159  else {
160  if (needSynch_) {
161  LOGERROR << "found SYNCH" << std::endl;
162  needSynch_ = false;
163  }
164  needed_ = Preamble::kCDRStreamSize;
165  }
166 
167  remaining_ = needed_;
168  continue;
169  }
170 
171  // Decode a Preamble and update how many bytes we need for a valid message. If the preamble says 0, then
172  // we are done.
173  //
174  Decoder decoder(building_->duplicate());
175  size_t messageSize = decoder.getMessageSize();
176  LOGDEBUG << "message size: " << messageSize << std::endl;
177  if (messageSize == 0) break;
178 
179  // Resize buffer to account for entire message, and recalculate number of bytes we need for a complete
180  // message.
181  //
182  needed_ = Preamble::kCDRStreamSize + messageSize;
183  LOGDEBUG << "needed: " << needed_ << " so far: " << building_->length()
184  << std::endl;
185  remaining_ = needed_ - building_->length();
186  LOGDEBUG << "remaining: " << remaining_ << std::endl;
187  if (remaining_ > building_->space()) {
188  LOGDEBUG << "expanding buffer - " << remaining_ << " space: "
189  << building_->space() << std::endl;
190  ACE_Message_Block* old = building_;
191  building_ = MessageManager::MakeMessageBlock(
192  needed_ + ACE_CDR::MAX_ALIGNMENT);
193  ACE_CDR::mb_align(building_);
194  building_->copy(old->rd_ptr(), old->length());
195  old->release();
196  assert(remaining_ <= building_->space());
197  }
198 
199  LOGDEBUG << "remaining: " << remaining_ << std::endl;
200  }
201 
202  if (remaining_ == 0) {
203  setAvailable(building_);
204  makeIncomingBuffer(building_->length() + 128);
205  }
206 
207  LOGDEBUG << "EXIT - remaining: " << remaining_ << std::endl;
208  return true;
209 }
