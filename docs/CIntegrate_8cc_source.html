81 CIntegrate::processInput(const Messages::Video::Ref& msg)
82 {
83  static Logger::ProcLog log("processInput", getLog());
84 
85  if (! enabled_->getValue()) {
86  return true;
87  }
88 
89  // Check the need to resize buffers. We just handle growing
90  //
91  size_t msgSize = msg->size() / 2;
92  if (msgSize > maxSpan_) {
93  maxSpan_ = msgSize;
94  LOGDEBUG << "new maxSpan: " << maxSpan_ << std::endl;
95  dimensionsChanged();
96  }
97 
98  // Add this new message to the message queue
99  //
100  buffer_.push_back(msg);
101  if (buffer_.size() != N_) {
102 
103  // Not enough messages to process, so bail out
104  //
105  return true;
106  }
107 
108  // Populate the input matrix
109  //
110  for (size_t rowIndex = 0; rowIndex < N_; ++rowIndex) {
111  Messages::Video::Ref pri = buffer_[rowIndex];
112  size_t M = pri->size() / 2;
113  Messages::Video::const_iterator pos = pri->begin();
114  VsipComplexMatrix::row_type row(PRIs_->row(rowIndex));
115  size_t colIndex = 0;
116  for (; colIndex < M; ++colIndex) {
117  float r = *pos++;
118  float i = *pos++;
119  row.put(colIndex, ComplexType(r, i));
120  }
121 
122  // Pad with zeroes if necessary
123  //
124  static const ComplexType zero(0.0, 0.0);
125  for (; colIndex < maxSpan_; ++colIndex)
126  row.put(colIndex, zero);
127  }
128 
129  // Apply FFT on *columns* of the PRIs matrix. Results go into fftOut.
130  //
131  (*fftm_)(*PRIs_, *fftOut_);
132 
133  // Return results for mid point of the buffer
134  //
135  Messages::Video::Ref mid(buffer_[N_ / 2]);
136 
137  // Do maxAmp processing in I,Q space
138  //
139  Messages::Video::Ref maxAmp(Messages::Video::Make("CIntegrate", mid));
140  processMaxAmp(maxAmp->getData());
141 
142  Messages::Video::Ref oneDop(Messages::Video::Make("CIntegrate", mid));
143  processOneDopplerBin(oneDop->getData());
144 
145  // Compute the magnitudes of the PRIs
146  //
147  *fftOut_ = vsip::mag(*fftOut_);
148 
149  bool rc = true;
150 
151  // Create a new message to hold the output of what we do. Note that although we pass in the input message,
152  // the new message does not contain any data.
153  //
154  Messages::Video::Ref out(Messages::Video::Make("CIntegrate", mid));
155  out->resize(maxSpan_);
156 
157  // Send out the FFT results, one doppler bin's worth of data for each PRI
158  //
159  Messages::Video::Container& outputData(out->getData());
160  for (size_t rowIndex = 0; rowIndex < N_ && rc; ++rowIndex) {
161  VsipComplexMatrix::row_type row(fftOut_->row(rowIndex));
162  for (size_t colIndex = 0; colIndex < maxSpan_; ++colIndex) {
163  outputData[colIndex] = ::rint(row.get(colIndex).real());
164  }
165  rc = rc && send(out, resultsIndex_);
166  }
167 
168  Messages::Video::Ref rgb(Messages::Video::Make("CIntegrate", mid));
169  if (colorEncoding_->getValue() == kRGB) {
170  processRGB(rgb->getData());
171  }
172  else {
173  processHSV(rgb->getData());
174  }
175 
176  // After CPI processing, remove W PRIs from buffer (sliding a 'window'). Send the same RGB, MaxAmp, and
177  // Doppler bin slice messages for each message removed.
178  //
179  size_t W = moveSlidingWin_->getValue();
180  for (size_t i = 0; i < W && ! buffer_.empty(); ++i){
181 
182  // Grab oldest message. Created messages will inherit its azimuth and other settings.
183  //
184  Messages::Video::Ref oldest(buffer_.front());
185  buffer_.pop_front();
186 
187  if (rc && rgbIndex_ != kInvalidChannel_) {
188  out = Messages::Video::Make("CIntegrate", oldest);
189  out->getData() = rgb->getData();
190  rc = rc && send(out, rgbIndex_);
191  }
192 
193  if (rc && maxAmpIndex_ != kInvalidChannel_) {
194  out = Messages::Video::Make("CIntegrate", oldest);
195  out->getData() = maxAmp->getData();
196  rc = rc && send(out, maxAmpIndex_);
197  }
198 
199  if (rc && dopplerBinOutIndex_ != kInvalidChannel_) {
200  out = Messages::Video::Make("CIntegrate", oldest);
201  out->getData() = oneDop->getData();
202  rc = rc && send(out, dopplerBinOutIndex_);
203  }
204  }
205 
206  LOGDEBUG << "rc: " << rc << std::endl;
207  return rc;
208 }

211 CIntegrate::processHSV(Messages::Video::Container& outputData)
212 {
213  static Logger::ProcLog log("processHSV", getLog());
214  LOGINFO << std::endl;

216  struct RGBType {
217  RGBType() : R(0.0), G(0.0), B(0.0) {}
218  RGBType(float r, float g, float b) : R(r), G(g), B(b) {}
219  void operator+=(const RGBType& rhs) {
220  R += rhs.R, G += rhs.G, B += rhs.B;
221  }
222  float R, G, B;
223  };
224 
225  struct HSVType {
226  HSVType(float h, float s, float v) : H(h), S(s), V(v) {}
227  const float H, S, V;
228  };
229 
230  for (size_t colIndex = 0; colIndex < maxSpan_; ++colIndex) {
231  VsipComplexMatrix::col_type col(fftOut_->col(colIndex));
232 
233  RGBType sum;
234  for (size_t rowIndex = 0; rowIndex < N_; ++rowIndex) {
235 
236  //
237  //scale amplitude data so values fall between 0 and 1
238  //
239  const ComplexType& z = col(rowIndex);
240  double amp = scaleAmp(z.real());
241 
242  // Come up with an hsv value for this doppler-range cell
243  //
244  HSVType hsv(H_[rowIndex], 1.0, amp);
245 
246  // Convert to RGB
247  //
248  int index = ::floor(hsv.H);
249  float f = hsv.H - index;
250  if (! (index & 1)) f = 1 - f; // if index is even
251  float m = hsv.V * (1 - hsv.S);
252  float n = hsv.V * (1 - hsv.S * f);
253 
254  switch (index) {
255  case 6:
256  case 0: sum += RGBType(hsv.V, n, m); break;
257  case 1: sum += RGBType(n, hsv.V, m); break;
258  case 2: sum += RGBType(m, hsv.V, n); break;
259  case 3: sum += RGBType(m, n, hsv.V); break;
260  case 4: sum += RGBType(n, m, hsv.V); break;
261  case 5: sum += RGBType(hsv.V, m, n); break;
262  }
263  }
264 
265  // Push and encoded RGB value into the output buffer
266  //
267  outputData.push_back(
268  Messages::Video::DatumType(EncodeRGB(sum.R, sum.G, sum.B)));
269  }

272 void
273 CIntegrate::processRGB(Messages::Video::Container& outputData)
274 {
275  static Logger::ProcLog log("processRGB", getLog());
276 
277  // Compute the RGB values for each range across the CPI
278  //
279  double span = N_ / 3.0; // 3 components to work with: R, G, B
280  double shift = span / 2.0; // The G component uses the highest and lowest
281  // bits, so everything is rotated 1/2 a span
282 
283  // Compute the summations
284  //
285  for (size_t i = 0; i < maxSpan_; ++i) {
286  VsipComplexMatrix::col_type col(fftOut_->col(i));
287 
288  // Build our red, green, and blue components using values from the FFT output. Green is centered around
289  // the 0 doppler bin and thus wraps around the buffer.
290  //
291  size_t j = 0;
292  float G = 0.0;
293  for (size_t limit = ::rint(shift); j < limit; ++j)
294  G += scaleAmp(ComplexType(col(j)).real());
295  float B = 0.0;
296  for (size_t limit = ::rint(shift + span); j < limit; ++j)
297  B += scaleAmp(ComplexType(col(j)).real());
298  float R = 0.0;
299  for (size_t limit = ::rint(shift + span * 2); j < limit; ++j)
300  R += scaleAmp(ComplexType(col(j)).real());
301  for (; j < N_; ++j)
302  G += scaleAmp(ComplexType(col(j)).real());
303 
304  // Push data into the output buffer
305  //
306  outputData.push_back(EncodeRGB(R, G, B));
307  }

368 void
369 CIntegrate::dimensionsChanged()
370 {
371  static Logger::ProcLog log("dimensionsChanged", getLog());
372  LOGINFO << std::endl;
373 
374  N_ = numPRIs_->getValue();
375 
376  LOGDEBUG << "N: " << N_ << " maxSpan: " << maxSpan_ << std::endl;
377 
378  // Only allocate when we've seen the first PRI message (which defines maxSpan_)
379  //
380  if (maxSpan_) {
381  PRIs_.reset(new VsipComplexMatrix(N_, maxSpan_));
382  fftOut_.reset(new VsipComplexMatrix(N_, maxSpan_));
383 
384  // Create FFT processor with a scaling of 1.0 / N. Although the matrix defined is N_ rows by maxSpan_
385  // range bins, the ForwardFFTM type dictates that this will perform maxSpan_ FFTs, each of N_ samples;
386  // in other words, each FFT process samples at the same range but at increasing time. NOTE: for faster
387  // processing, look into rolling the 1.0 / N scaling into another operation where we don't have to visit
388  // every sample.
389  //
390  fftm_.reset(new ForwardFFTM(vsip::Domain<2>(N_, maxSpan_),
391  1.0 / N_));
392  }
393 
394  // This is for HSV encoding. Map each doppler slice to a distinct hue, starting with RGB green (0.3333 Hue)
395  // for doppler bin 0.
396  //
397  H_.clear();
398  H_.reserve(N_);
399  double step = 1.0 / N_;
400  for (size_t i = 0; i < N_; ++i) {
401  H_.push_back(6.0 * ::fmod(1.0 / 3.0 + step * i, 1.0));
402  LOGDEBUG << "H_[" << i << "]: " << H_.back() << std::endl;
403  }
404 
405  reset();
406 }
