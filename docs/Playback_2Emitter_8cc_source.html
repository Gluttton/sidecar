61 Emitter::load(const QFileInfo& fileInfo)
62 {
63  std::string path(fileInfo.absoluteFilePath().toStdString());
64  Utils::FilePath fp(path);
65  fp.setExtension(IO::TimeIndex::GetIndexFileSuffix());
66  if (! fp.exists()) {
67  IO::IndexMaker::Status status = IO::IndexMaker::Make(
68  path, 15, boost::bind(&Emitter::indexMakerUpdate, this, _1));
69  if (status != IO::IndexMaker::kOK) {
70  return;
71  }
72  }
73 
74  timeIndex_ = new IO::TimeIndex(fp);
75  int fd = ::open(path.c_str(), O_RDONLY);
76  if (fd == -1) {
77  return;
78  }
79 
80  IO::FileReader reader;
81  reader.getDevice().set_handle(fd);
82 
83  // Fetch the first record so we can get the file's metatype. Create a dummy Header object. We give it the Video
84  // message meta-type to satisfy Header's API, but it is not relevant here; we could have given it any of the valid
85  // meta-types.
86  //
87  Messages::Header header(*Messages::MetaTypeInfo::Find(Messages::MetaTypeInfo::Value::kVideo));
88  if (! reader.fetchInput() || ! reader.isMessageAvailable()) {
89  return;
90  }
91 
92  ACE_Message_Block* data = reader.getMessage();
93  IO::Decoder decoder(data);
94  header.load(decoder);
95 
96  // Now we have a valid Header. Fetch the MetaTypeInfo object it refers to.
97  //
98  metaTypeInfo_ = Messages::MetaTypeInfo::Find(header.getGloballyUniqueID().getMessageTypeKey());
99  if (! metaTypeInfo_) {
100  return;
101  }
102 
103  // We are valid now. Assume the real emitting value.
104  //
105  ::lseek(fd, 0, SEEK_SET);
106 
107  // Connect signal handlers to the playback clock.
108  //
109  connect(clock_, SIGNAL(started()), SLOT(start()));
110  connect(clock_, SIGNAL(stopped()), SLOT(stop()));
111  connect(clock_, SIGNAL(playbackClockStartChanged(const Time::TimeStamp&)),
112  SLOT(setPlaybackClockStart(const Time::TimeStamp&)));
113 
114  // Set our reader to use the given open file descriptor. The reader is set
115  // to begin reading from the start of the file.
116  //
117  reader_.getDevice().set_handle(fd);
118 
119  // Read the first record to get its timestamp.
120  //
121  reader_.fetchInput();
122  IO::MessageManager mgr(reader_.getMessage(), metaTypeInfo_);
123  Messages::Header::Ref msg(mgr.getNative());
124  startTime_ = msg->getCreatedTimeStamp();
125 
126  // Read the last record in the data file to get its timestamp.
127  //
128  ::lseek(fd, timeIndex_->getLastEntry(), SEEK_SET);
129  if (reader_.fetchInput()) {
130  IO::MessageManager mgr(reader_.getMessage(), metaTypeInfo_);
131  Messages::Header::Ref msg(mgr.getNative());
132  endTime_ = msg->getCreatedTimeStamp();
133  }
134 
135  // Rewind to the file beginning.
136  //
137  ::lseek(fd, 0, SEEK_SET);
138 
139  // Create a message writer that will send out the messages at the appropriate time.
140  //
141  valid_ = true;
142  indexMakerUpdate(0.0);
143 }

147 {
148  static Logger::ProcLog log("makeWriter", Log());
149  LOGINFO << name_ << " restartIfRunning: " << restartIfRunning << std::endl;
150 
151  if (! valid_ || ! emitting_) {
152  return;
153  }
154 
155  // Make sure we are not running when we create a new UDPMessageWriter. Remember if we were running in case we are
156  // permitted to restart.
157  //
158  bool wasRunning = running_;
159  if (wasRunning)
160  stop();
161 
162  if (writer_) {
163  writer_->stop();
164  delete writer_;
165  writer_ = 0;
166  }
167 
168  // Create our published name.
169  //
170  QString name(name_);
171  if (! suffix_.isEmpty()) {
172  name += suffix_;
173  }
174 
175  subscriberCount_ = 0;
176  writer_ = UDPMessageWriter::Make(name, metaTypeInfo_->getName(), address_);
177  if (! writer_) {
178  LOGERROR << "failed to create writer!" << std::endl;
179  return;
180  }
181

183 
184  // Restart our own emitter thread if we were running and we can restart.
185  //
186  if (wasRunning && restartIfRunning) {
187  start();
188  }

319 void
320 Emitter::repositionAndFetch(off_t pos, const Time::TimeStamp& when)
321 {
322  static Logger::ProcLog log("repositionAndFetch", Log());
323  LOGINFO << "pos: " << pos << " when: " << when << std::endl;
324 
325  if (! valid_) return;
326 
327  if (pending_) {
328  pending_->release();
329  pending_ = 0;
330  }
331 
332  int fd = reader_.getDevice().get_handle();
333  if (fd == -1) {
334  LOGERROR << "invalid fd" << std::endl;
335  return;
336  }
337 
338  // Move to given position in the file.
339  //
340  ::lseek(fd, pos, SEEK_SET);
341 
342  // Now continue reading until we've reached the record that is beyond the time we are seeking. If our indexing is
343  // not too coarse, this should be fairly quick.
344  //
345  Messages::Header header(*metaTypeInfo_);
346  ACE_Message_Block* last = 0;
347  while (reader_.fetchInput() && reader_.isMessageAvailable()) {
348  ACE_Message_Block* data = reader_.getMessage();
349  IO::Decoder decoder(data->duplicate());
350  header.load(decoder);
351  if (header.getCreatedTimeStamp() > when) {
352  if (! last) last = data;
353  break;
354  }
355 
356  if (last) last->release();
357  last = data;
358  }
359 
360  if (last) {
361 
362  // Found something to emit. Record the message found and its timestamp.
363  //
364  IO::MessageManager mgr(last, metaTypeInfo_);
365  Messages::Header::Ref msg(mgr.getNative());
366  LOGDEBUG << "sequence: "
367  << msg->getGloballyUniqueID().getMessageSequenceNumber()
368  << " time: " << msg->getCreatedTimeStamp() << std::endl;
369  pending_ = mgr.getMessage();
370  pendingTime_ = header.getCreatedTimeStamp();
371  }
372 }

374 void
375 Emitter::run()
376 {
377  // Maximum amount of time to sleep while waiting to emit a message.
378  //
379  static const Time::TimeStamp kMaxSleep(0, Time::TimeStamp::kMicrosPerSecond / 10); // 0.1 seconds
380 
381  static Logger::ProcLog log("run", Log());
382  LOGINFO << name_ << " valid: " << valid_ << " emitting: " << emitting_ << " writer: " << writer_ << " pending: "
383  << pending_ << std::endl;
384 
385  // !!! This should never be true here
386  //
387  if (! valid_ || ! emitting_) {
388  LOGERROR << "exiting" << std::endl;
389  return;
390  }
391 
392  // Keep running while our running_ attribute is true and we have a message to emit.
393  //
394  while (running_ && pending_) {
395 
396  // Calculate amount of wall time until the next emission.
397  //
398  Time::TimeStamp duration(clock_->getWallClockDurationUntil(pendingTime_));
399  LOGDEBUG << name_ << " clock: " << clock_->getPlaybackClock()
400  << " pendingTime: " << pendingTime_
401  << " duration: " << duration << std::endl;
402 
403  // Do we need to sleep?
404  //
405  if (duration > Time::TimeStamp::Min()) {
406 
407  // Don't sleep too long in order to detect running_ changes. Since we know there will be more time to wait,
408  // we 'continue' back to beginning of the while loop once we wake up.
409  //
410  if (duration > kMaxSleep) {
411  usleep(kMaxSleep.getMicro());
412  continue;
413  }
414 
415  // Sleep for the required number of microseconds. When we awake, we will emit the pending message.
416  //
417  usleep(duration.getSeconds() * Time::TimeStamp::kMicrosPerSecond + duration.getMicro());
418  }
419 
420  // Time to emit the message.
421  //
422  LOGDEBUG << name_ << " emitting " << pendingTime_ << std::endl;
423  IO::MessageManager mgr(pending_);
424  if (subscriberCount_) {
425  writer_->writeMessage(mgr);
426  }
427  pending_ = 0;
428 
429  // Fetch the next message to emit.
430  //
431  if (reader_.fetchInput()) {
432  IO::MessageManager mgr(reader_.getMessage(), metaTypeInfo_);
433  pending_ = mgr.getMessage();
434  pendingTime_ = mgr.getNative()->getCreatedTimeStamp();
435  LOGDEBUG << name_ << " pendingTime: " << pendingTime_ << std::endl;
436  }
437  }
438 
439  // !!! Don't clear 'running_' flag when we exit. That way, the main thread will reap us with a wait() call.
440  //
441  LOGDEBUG << name_ << " thread exiting" << std::endl;
442 }
