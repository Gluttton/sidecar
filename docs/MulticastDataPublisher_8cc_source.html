51 MulticastDataPublisher::openAndInit(const std::string& key, const std::string& serviceName,
52  const std::string& multicastAddress, long threadFlags, long threadPriority)
53 {
54  Logger::ProcLog log("openAndInit", Log());
55  LOGINFO << "key: " << key << " serviceName: " << serviceName
56  << " address: " << multicastAddress
57  << " threadFlags: " << std::hex << threadFlags << std::dec
58  << " threadPriority: " << threadPriority << std::endl;
59 
60  if (! reactor()) reactor(ACE_Reactor::instance());
61 
62  threadFlags_ = threadFlags;
63  threadPriority_ = threadPriority;
64  setMetaTypeInfoKeyName(key);
65  setTransport("multicast");
66 
67  // Disallow any data into our processing queue. Activate it later when we've successfully published our
68  // connection information, just before we start our writer thread.
69  //
70  msg_queue()->deactivate();
71 
72  // Working directly with the ACE datagram (UDP) object -- this perhasps should be refactored into the
73  // UDPWriter class itself as an open() call.
74  //
75  ACE_SOCK_Dgram& device(writer_.getDevice());
76 
77  // Open the writer device to use to send out data. For opening, we use the "any" address with a zero port.
78  // The system will assign us a port number to use.
79  //
80  ACE_INET_Addr address(u_short(0), "0.0.0.0", AF_INET);
81  LOGDEBUG << "opening UDP socket with address "
82  << Utils::INETAddrToString(address) << std::endl;
83  if (device.open(address, AF_INET, 0, 1) == -1) {
84  LOGERROR << "failed to open multicast writer at " << multicastAddress << std::endl;
85  return false;
86  }
87 
88  LOGDEBUG << "opened socket " << writer_.getDevice().get_handle() << std::endl;
89 
90  // Set the time-to-live to a non-zero value so we don't break the MIT/LL network.
91  //
92  unsigned char ttl = 10;
93  if (device.set_option(IPPROTO_IP, IP_MULTICAST_TTL, &ttl, sizeof(ttl)) == -1) {
94  LOGERROR << "failed to set TTL for " << multicastAddress << std::endl;
95  return false;
96  }
97 
98  // Fetch the port assigned to us by the system.
99  //
100  device.get_local_addr(address);
101  uint16_t port = address.get_port_number();
102  LOGDEBUG << "socket device address: " << Utils::INETAddrToString(address) << std::endl;
103 
104  // Build our multicast address now that we know our port number. Tell the writer where to send the UDP
105  // messages.
106  //
107  address.set(port, multicastAddress.c_str(), 1, AF_INET);
108  LOGDEBUG << "mcast address: " << Utils::INETAddrToString(address) << std::endl;
109  writer_.setRemoteAddress(address);
110 
111  // Set our Zeroconf publisher connection info.
112  //
113  std::string serviceType(MakeZeroconfType(key));
114  LOGDEBUG << "serviceType: " << serviceType << std::endl;
115  setType(serviceType); // Our unique service ID
116  setPort(port); // Our unique port number
117  setHost(multicastAddress); // Our assigned multicast address
118 
119  // Create our heart-beat reader. First, open a new UDP socket so we can get its system-assigned port value.
120  //
121  address.set(uint16_t(0));
122  if (heartBeatReader_.open(address) == -1) {
123  LOGERROR << "failed to create heart-beat UDP reader" << std::endl;
124  return false;
125  }
126 
127  // Install ourselves in the global ACE_Reactor so that we receive notification when there is data available
128  // on the heart-beat port.
129  //
130  if (reactor()->register_handler(heartBeatReader_.get_handle(), this, ACE_Event_Handler::READ_MASK) == -1) {
131  LOGERROR << "failed to register heart-beat input handler" << std::endl;
132  return false;
133  }
134 
135  // Fetch the port assigned to us by the system.
136  //
137  heartBeatReader_.get_local_addr(address);
138  LOGDEBUG << "heartbeat address: " << Utils::INETAddrToString(address) << std::endl;
139 
140  // Convert port number to text so we can store in the published TEXT record.
141  //
142  std::ostringstream os;
143  os << address.get_port_number();
144  getConnectionPublisher()->setTextData("HeartBeatPort", os.str());
145 
146  // Now we are ready to publish our connection information.
147  //
148  if (! publish(serviceName)) {
149  LOGERROR << "failed to publish connection with name '" << serviceName << " type: " << serviceType
150  << std::endl;
151  return false;
152  }
153 
154  // Create a timer to periodically check for expired connections.
155  //
156  const ACE_Time_Value delay(5);
157  const ACE_Time_Value repeat(5);
158  timer_ = reactor()->schedule_timer(this, &timer_, delay, repeat);
159 
160  return true;
161 }

214 int
215 MulticastDataPublisher::handle_input(ACE_HANDLE handle)
216 {
217  static Logger::ProcLog log("handle_input", Log());
218  LOGINFO << std::endl;
219 
220  // Fetch the heart-beat message. Should be 'HI' or 'BYE'. Actually, this whole routine should be refactored
221  // into another class.
222  //
223  ACE_TCHAR buffer[16];
224  ACE_INET_Addr address;
225  ssize_t count = heartBeatReader_.recv(buffer, sizeof(buffer), address);
226  if (count < 1) {
227  LOGERROR << "failed recv - " << errno << ' ' << strerror(errno) << std::endl;
228  return 0;
229  }
230 
231  std::string msg(buffer);
232  LOGDEBUG << "msg: " << msg << std::endl;
233 
234  // Convert the client address into a string to be used as a key into the HeartBeatMap container.
235  //
236  std::string key(Utils::INETAddrToString(address));
237  if (key.size() == 0) {
238  LOGERROR << "failed INETAddrToString" << std::endl;
239  return 0;
240  }
241 
242  LOGDEBUG << "key: " << key << std::endl;
243  if (msg == "HI") {
244 
245  // Update the timestamp for the client address. This also works for first-time additions.
246  //
247  heartBeats_[key] = Time::TimeStamp::Now();
248  }
249  else if (msg == "BYE") {
250 
251  // Remove the entry for the client address.
252  //
253  HeartBeatMap::iterator pos = heartBeats_.find(key);
254  if (pos != heartBeats_.end()) heartBeats_.erase(pos);
255  }
256 
257  updateUsingDataValue();
258 
259  LOGDEBUG << "heartBeats_.empty(): " << heartBeats_.empty() << std::endl;
260  return 0;
261 }

263 int
264 MulticastDataPublisher::handle_timeout(const ACE_Time_Value& duration, const void* arg)
265 {
266  static Logger::ProcLog log("handle_timeout", Log());
267  LOGINFO << std::endl;
268 
269  if (arg != &timer_) return Super::handle_timeout(duration, arg);
270 
271  if (heartBeats_.empty()) return 0;
272 
273  // Calculate a time that is 60 seconds ago. If a timestamp is older than that, assume the connection is
274  // gone.
275  //
276  Time::TimeStamp limit = Time::TimeStamp::Now();
277  limit -= Time::TimeStamp(60, 0);
278 
279  HeartBeatMap::iterator pos = heartBeats_.begin();
280  HeartBeatMap::iterator end = heartBeats_.end();
281  while (pos != end) {
282  if (pos->second > limit) {
283  ++pos;
284  }
285  else {
286 
287  // !!! Erasing an item from std::map affects only affects iterators pointing to the item. So, we
288  // !!! need to advance our loop iterator before peforming the erase.
289  //
290  HeartBeatMap::iterator tmp = pos++;
291  LOGDEBUG << "forgetting " << pos->first << std::endl;
292  heartBeats_.erase(tmp);
293  }
294  }
295 
296  updateUsingDataValue();
297 
298  return 0;
