53  }

57  double getValue()
58  {
59  if (values_.empty()) {
60  int counter = offset_;
61  for (int index = 0; index < gatesPerCycle_; ++index) {
62  if (counter == gatesPerCycle_)
63  counter = 0;
64  values_.push_back(::sin(counter++ * gateChange_) *
65  scaledAmplitude_ +
66  scaledDC_);
67  }
68  }
69 
70  if (gate_ == gatesPerCycle_)
71  gate_ = 0;
72  return values_[gate_++];
73  }
