39 SimpleOp::processChannels()

42  LOGINFO << std::endl;
43 
44  // If we are not enabled, just copy the message from the first enabled channel to the output.
45  //
46  if (! isEnabled()) {
47  LOGDEBUG << "disabled" << std::endl;
48  for (size_t index = 0; index < getChannelCount(); ++index) {
49  VideoChannelBuffer* channel = getChannelBuffer<Messages::Video>(index);
50  if (channel->isEnabled()) {
51  Messages::Video::Ref msg(channel->popFront());
52  Messages::Video::Ref out(Messages::Video::Make(getName(), msg));
53  out->getData() = msg->getData();
54  LOGDEBUG << "sending from channel " << channel->getChannelIndex() << std::endl;
55  return send(out);
56  }
57  }
58 
59  LOGDEBUG << "nothing to output" << std::endl;
60  return true;
61  }

63  using IteratorVector = std::vector<Messages::Video::const_iterator>;
64  IteratorVector iterators;
65  size_t size = 0;
66  Messages::Video::Ref out;
67 
68  // Obtain iterators to the samples of the messages in the input buffers. NOTE: do not call channel->popFront()
69  // since we need to keep a reference around for the message. Call popFront after we've built our output message.
70  //
71  for (size_t index = 0; index < getChannelCount(); ++index) {
72  VideoChannelBuffer* channel = getChannelBuffer<Messages::Video>(index);
73  if (channel->isEnabled()) {
74  Messages::Video::Ref msg(channel->getFront());
75  if (! out) {
76  out = Messages::Video::Make(getName(), msg);
77  size = msg->size();
78  }
79  else if (size > msg->size()) {
80  size = msg->size();
81  }
82  iterators.push_back(msg->begin());
83  }
84  }
85 
86  if (! out) {
87  LOGWARNING << "enabled but no channels enabled" << std::endl;
88  return true;
89  }
90 
91  Operator op = operator_->getValue();
92  LOGDEBUG << "operation " << op << ' ' << kOperatorNames[op] << std::endl;
93 
94  while (size-- != 0) {
95  IteratorVector::iterator pos = iterators.begin();
96  Messages::Video::DatumType value = *(*pos++)++;
97  IteratorVector::iterator end = iterators.end();
98  switch (op) {
99  case kSumOp: // A + B
100  while (pos != end) value += *(*pos++)++;
101  break;
102 
103  case kDiffOp: // A - B
104  while (pos != end) value -= *(*pos++)++;
105  break;
106 
107  case kProdOp: // A * B
108  while (pos != end) value *= *(*pos++)++;
109  break;
110 
111  case kMinOp: // MIN(A, B)
112  while (pos != end) value = std::min(value, *(*pos++)++);
113  break;
114 
115  case kMaxOp: // MAX(A, B)
116  while (pos != end) value = std::max(value, *(*pos++)++);
117  break;
118 
119  default: break;
120  }
121 
122  out->push_back(value);
123  }
124 
125  LOGDEBUG << out->headerPrinter() << std::endl;
126  LOGDEBUG << out->dataPrinter() << std::endl;
127 
128  // Remove the message from the buffers.
129  //
130  for (size_t index = 0; index < getChannelCount(); ++index) {
131  VideoChannelBuffer* channel = getChannelBuffer<Messages::Video>(index);
132  if (channel->isEnabled()) {
133  Messages::Video::Ref msg(channel->popFront());
134  }
135  }
136 
137  return send(out);
