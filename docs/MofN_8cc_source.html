74 bool
75 MofN::process(const Messages::BinaryVideo::Ref& video)
76 {
77  static Logger::ProcLog log("process", getLog());
78  LOGDEBUG << std::endl;
79 
80  if (! enabled_->getValue()) {
81  return send(video);
82  }
83 
84  // Detect mismatch between current runningCounts_.size() and sample count in the message. Grow
85  // runningCounts_.size() if necessary, but leave the message alone if it is smaller.
86  //
87  size_t priLen = video->size();
88  if (runningCounts_.size() < priLen) {
89  runningCounts_.resize(priLen, 0);
90  }
91  else if (runningCounts_.size() > priLen) {
92  LOGWARNING << "incoming message too small - expected: " << runningCounts_.size() << " got: " << priLen
93  << std::endl;
94  }
95 
96  // We save the result of counting in the range dimension so that we won't have to repeat the calculation for
97  // subsequent PRIs. We will ultimately keep around vectors for the previous numPRIs PRIs
98  //
99  size_t numPRIs = numPRIs_->getValue();
100  retained_[newestIndex_].video = video;
101 
102  // Use the oldest + 1 slot to hold our sample and detection counts. We need the oldest detection counts below to
103  // keep our running count up-to-date so we cannot use that slot yet.
104  //
105  DetectionCountVector& detections(retained_[newestIndex_++].detectionCounts);
106  if (newestIndex_ == retained_.size()) {
107  newestIndex_ = 0;
108  }
109 
110  detections.resize(runningCounts_.size(), 0);
111 
112  // The first stage in the calculation is to count the number of detections within a "numGates" sized window window
113  // around each range cell.
114  //
115  Messages::BinaryVideo::const_iterator videoIter(video->begin());
116  size_t numGates = numGates_->getValue();
117  size_t halfWindow = numGates / 2;
118  size_t gateIndex = 0;
119 
120  // Calculate partial windows at the start of the PRI. Update the running counter, but don't record anything get.
121  //
122  DetectionCountType count = 0;
123  while (gateIndex++ < halfWindow) {
124  count += (*videoIter++ ? 1 : 0);
125  }
126 
127  // Keep updating the counter, but now we can store the counter results.
128  //
129  DetectionCountVector::iterator detectionsIter(detections.begin());
130  while (gateIndex++ < numGates) {
131  count += (*videoIter++ ? 1 : 0);
132  *detectionsIter++ = count;
133  }
134 
135  // Even better -- we can store and remove the oldest sample.
136  //
137  Messages::BinaryVideo::const_iterator oldestIter(video->begin());
138  while (gateIndex++ < priLen) {
139  count += (*videoIter++ ? 1 : 0);
140  *detectionsIter++ = count;
141  count -= (*oldestIter++ ? 1 : 0);
142  }
143 
144  // Reached the end of the PRI message. Go back and finish calculating windowed counts. Be careful not to move
145  // beyond the sample data which could happen if priLen < runningCounts_.size()
146  //
147  Messages::BinaryVideo::const_iterator videoEnd(video->end());
148  while (oldestIter < videoEnd) {
149  *detectionsIter++ = count;
150  count -= (*oldestIter++ ? 1 : 0);
151  }
152 
153  // Finally, handle the case where priLen < runningCounts_.size(). Just set to zero.
154  //
155  DetectionCountVector::iterator detectionsEnd(detections.end());
156  while (detectionsIter < detectionsEnd) {
157  *detectionsIter++ = 0;
158  }
159 
160  // Update the running counts vector with the detections we just calculated. If we don't have a desired number of
161  // PRI messages, then just add the calculated counts for each gate to the running count tally.
162  //
163  if (retainedCount_ < numPRIs) {
164  ++retainedCount_;
165  std::transform(runningCounts_.begin(), runningCounts_.end(), detections.begin(), runningCounts_.begin(),
166  std::plus<DetectionCountType>());
167  return true;
168  }
169 
170  // Create a new BinaryVideo message with gate values that are true/false depending on whether the MofN window
171  // surrounding the gate has enough ON values to pass a threshold value. Base our outgoing message on the middle
172  // message in our MofN window.
173  //
174  size_t index = (oldestIndex_ + numPRIs / 2);
175  if (index >= retained_.size()) {
176  index -= retained_.size();
177  }
178 
179  Messages::BinaryVideo::Ref midPoint((retained_.begin() + index)->video);
180  Messages::BinaryVideo::Ref out(Messages::BinaryVideo::Make(getName(), midPoint));
181  out->reserve(runningCounts_.size());
182 
183  // Add the new counts vector to and subtract the oldest counts vector from the running count. A side-effect of this
184  // is that the output message gets the boolean values based on whether the updated sample count values match or
185  // pass thresholdValue_.
186  //
187  std::transform(runningCounts_.begin(), runningCounts_.end(), runningCounts_.begin(),
188  Updater(thresholdValue_, retained_[oldestIndex_].detectionCounts.begin(), detections.begin(),
189  out->getData()));
190  ++oldestIndex_;
191  if (oldestIndex_ == retained_.size()) {
192  oldestIndex_ = 0;
193  }
194 
195  LOGDEBUG << *out.get() << std::endl;
196  bool rc = send(out);
197  LOGDEBUG << "rc: " << rc << std::endl;
198 
199  return rc;
200 }
