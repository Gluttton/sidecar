69 PresetManager::restoreAllPresets()
70 {
71  Logger::ProcLog log("restoreAllPresets", Log());
72  LOGINFO << std::endl;
73 
74  restoring_ = true;
75  file_.beginGroup(prefix_);
76  {
77  // Load in the global settings.
78  //
79  file_.beginGroup(kGlobals);
80  {
81  for (int index = 0; index < globals_.size(); ++index) {
82  Setting* setting = globals_[index];
83  QVariant value = file_.value(setting->getName(),
84  setting->getOpaqueValue());
85  if (isOldVersion())
86  value = upgradeSetting(setting->getName(), value);
87  setting->restoreValue(value);
88  }
89  }
90  file_.endGroup();
91 
92  // Remove any existing named presets (but not the first entry)
93  //
94  while (presets_.size())
95  delete presets_.takeLast();
96 
97  // Load in the names of defined presets. This will create a new Preset object for each name that was
98  // last saved.
99  //
100  int size = file_.beginReadArray(kPresetNames);
101  for (int index = 0; index < size; ++index) {
102  file_.setArrayIndex(index);
103  QString name = file_.value(kName).toString();
104  if (! name.isEmpty())
105  presets_.append(new Preset(name, this));
106  }
107  file_.endArray();
108 
109  // Temporary hack for old config file versions - if the first entry is NOT the 'default' one, create one
110  // and place at the beginning of the list.
111  //
112  if (presets_.empty() || presets_[0]->getName() != kDefault) {
113  presets_.insert(0, new Preset(kDefault, this));
114  }
115 
116  emit presetNamesChanged(getPresetNames());
117 
118  // Restore the index of the last preset.
119  //
120  activePresetIndex_ = file_.value(kLastActive, 0).toInt();
121 
122  // Load in the preset setting values. If restoring the active preset, have it also apply the value to
123  // the Setting object.
124  //
125  file_.beginGroup(kPresetData);
126  {
127  for (int index = 0; index < presets_.size(); ++index) {
128  presets_[index]->restore(file_, settings_,
129  index == activePresetIndex_);
130  if (index)
131  emit presetDirtyStateChanged(index, false);
132  }
133  }
134  file_.endGroup(); // kPresetData
135 
136  emit activePresetChanged(activePresetIndex_);
137  }
138 
139  file_.endGroup(); // prefix_
140  restoring_ = false;
141  restored_ = true;
142 }

164  //

173  //

175  {

180  if (index)

183  }
184  }
185  file_.endGroup(); // kPresetData
186 
187  // Save the index of the active preset.
188  //

413 PresetManager::removePreset(int index)
414 {
415  // Don't allow deleting the 'default' preset.
416  //
417  if (index < 1 || index > presets_.size() - 1)
418  return false;
419 
420  delete presets_.takeAt(index);
421 
422  file_.beginGroup(prefix_);
423 
424  // Write the array of preset names to the configuration file.
425  //
426  file_.beginWriteArray(kPresetNames, presets_.size());
427  for (int nameIndex = 0; nameIndex < presets_.size(); ++nameIndex) {
428  file_.setArrayIndex(nameIndex);
429  file_.setValue(kName, presets_[nameIndex]->getName());
430  }
431  file_.endArray();
432 
433  // Are we removing the active preset?
434  //
435  if (activePresetIndex_ != index) {
436 
437  // No. If removing an entry prior to the active one, just adjust the active index value and record it,
438  // but do not announce any change.
439  //
440  if (index < activePresetIndex_) {
441  --activePresetIndex_;
442  file_.setValue(kLastActive, activePresetIndex_);
443  }
444  }
445  else {
446 
447  // Yes. Make the 'default' preset the active one.
448  //
449  activePresetIndex_ = 0;
450  presets_[activePresetIndex_]->apply(settings_);
451  file_.setValue(kLastActive, activePresetIndex_);
452  emit activePresetChanged(activePresetIndex_);
453  }
454 
455  file_.endGroup(); // prefix_
456 
457  emit presetNamesChanged(getPresetNames());
458 
459  return true;
460 }
