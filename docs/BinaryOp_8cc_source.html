78 bool
79 BinaryOp::processChannels()
80 {
81  static Logger::ProcLog log("processInput", getLog());
82  LOGINFO << std::endl;
83 
84  using MsgVector = std::vector<BinaryVideo::Ref>;
85  MsgVector inputs;
86  size_t minSize = fetchEnabledChannelMessages<BinaryVideo>(inputs);
87 
88  // If we are not enabled, just copy the message from the first enabled channel to the output.
89  //
90  if (! isEnabled()) {
91  LOGDEBUG << "disabled" << std::endl;
92  if (! inputs.empty()) {
93  BinaryVideo::Ref msg(inputs.front());
94  BinaryVideo::Ref out(BinaryVideo::Make(getName(), msg));
95  out->getData() = msg->getData();
96  return send(out);
97  }
98 
99  LOGDEBUG << "nothing to output" << std::endl;
100  return true;
101  }
102  else if (inputs.empty()) {
103  LOGWARNING << "enabled but no channels enabled" << std::endl;
104  return true;
105  }
106 
107  // Create an output message using the first input message as its basis.
108  //
109  BinaryVideo::Ref out(BinaryVideo::Make(getName(), inputs.front()));
110  out->resize(minSize);
111 
112  Operator op = operator_->getValue();
113  LOGDEBUG << "operation " << op << ' ' << kOperatorNames[op] << std::endl;
114 
115  // We support configurations with more than two inputs. To do so, we perform the binary operation on the
116  // first two inputs and store the result in the output message. For subsequent passes, we perform the binary
117  // operation on the output message and the next input message, storing the result back in the output
118  // message. This works fine, except that we must hold off performing the NOT operation for the NAND, NOR,
119  // NXOR operations until we've processed all operations.
120  //
121  BinaryVideo::Ref a = inputs.front();
122  if (op != kNotOp) {
123  for (size_t index = 1; index < inputs.size(); ++index) {
124  BinaryVideo::Ref b = inputs[index];
125  switch (operator_->getValue()) {
126  case kAndOp:
127  case kNotAndOp:
128  std::transform(a->begin(), a->begin() + minSize, b->begin(), out->begin(), AndOp());
129  break;
130 
131  case kOrOp:
132  case kNotOrOp:
133  std::transform(a->begin(), a->begin() + minSize, b->begin(), out->begin(), OrOp());
134  break;
135 
136  case kXorOp:
137  case kNotXorOp:
138  std::transform(a->begin(), a->begin() + minSize, b->begin(), out->begin(), ExclusiveOrOp());
139  break;
140 
141  default:
142  break;
143  }
144 
145  // Make the next 'a' input be the results of the last binary operation.
146  //
147  a = out;
148  }
149  }
150 
151  // Need to perform a NOT operation on the result? If we've gone through the above loop at least once, 'a'
152  // will alias 'out'.
153  //
154  if (op >= kNotOp) {
155  std::transform(a->begin(), a->begin() + minSize, out->begin(), NotOp());
156  }
157 
158  LOGDEBUG << out->dataPrinter() << std::endl;
159 
160  return send(out);
161 }
