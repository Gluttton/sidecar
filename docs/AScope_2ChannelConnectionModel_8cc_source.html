75 ChannelConnectionModel::setVisualizer(Visualizer* visualizer)
76 {
77  static Logger::ProcLog log("setVisualizer", Log());
78  LOGINFO << "old: " << visualizer_ << " new: " << visualizer << std::endl;
79 
80  // CAREFUL! The rowCount() method relies on visualizer_ to get its value. Make sure we set visualizer_ only
81  // after we are done with the old value.
82  //
83 
84  // Fetch the number of rows in the model for the current visualizer_ value.
85  //
86  int oldRows = rowCount();
87  if (oldRows) {
88 
89  // Tell views that these rows are going away.
90  //
91  beginRemoveRows(QModelIndex(), 0, oldRows - 1);
92 
93  // Zap our visualizer_ value so any future rowCount() calls will return zero.
94  //
95  visualizer_ = 0;
96  reset();
97  endRemoveRows();
98  }
99 
100  // Now assign the new visualzer_ value, and fetch the number of entries it has.
101  //
102  int newRows = visualizer ? visualizer->getNumChannelConnections() : 0;
103  if (newRows)
104  beginInsertRows(QModelIndex(), 0, newRows - 1);
105 
106  visualizer_ = visualizer;
107  if (newRows)
108  endInsertRows();
109 
110  QSet<QString> unconnected = available_;
111  LOGDEBUG << "available: " << available_ << std::endl;
112 
113  if (newRows)
114  unconnected -= visualizer_->getConnectionNames();
115 
116  LOGDEBUG << "unconnected: " << unconnected << std::endl;
117  QStringList tmp = unconnected.toList();
118  qSort(tmp);
119  LOGDEBUG << "tmp: " << tmp << std::endl;
120 
121  getParent()->setUnconnected(tmp);
122 }

242 ChannelConnectionModel::data(const QModelIndex& pos, int role) const
243 {
244  static Logger::ProcLog log("data", Log());
245  if (! visualizer_ || ! pos.isValid() || pos.row() >= rowCount())
246  return QVariant();
247 
248  ChannelConnection* connection = getChannelConnection(pos.row());
249  const VideoChannel& channel = connection->getChannel();
250 
251  QVariant value;
252  if (role == Qt::TextAlignmentRole) {
253  if (pos.column() > kColor)
254  return int(Qt::AlignRight | Qt::AlignVCenter);
255  return value;
256  }
257  else if (role == Qt::ForegroundRole) {
258  return channel.isConnected() ? Qt::black : Qt::red;
259  }
260 
261  switch (pos.column()) {
262  case kName:
263 
264  // The values returned for the kName column are either the name text or the display font. The font
265  // italicized if the channel is not connected.
266  //
267  switch (role) {
268  case Qt::DisplayRole:
269  value = channel.getName(); break;
270  case Qt::ToolTipRole:
271  value = "Double-click to change sample min/max values"; break;
272  default:
273  break;
274  }
275  break;
276 
277  case kColor:
278 
279  // The value returned for the kColor column are the QColor value stored in the configuration.
280  //
281  if (role == Qt::BackgroundColorRole || role == Qt::EditRole)
282  value = channel.getColor();
283  else if (role == Qt::ToolTipRole)
284  value = "Double-click to change the color";
285  break;
286 
287  case kSampleMin:
288  if (role == Qt::DisplayRole)
289  value = channel.getSampleMin();
290  else if (role == Qt::ToolTipRole)
291  value = "Double-click to change sample min/max values";
292  break;
293 
294  case kSampleMax:
295  if (role == Qt::DisplayRole)
296  value = channel.getSampleMax();
297  else if (role == Qt::ToolTipRole)
298  value = "Double-click to change sample min/max values";
299  break;
300 
301  case kVoltageMin:
302  if (role == Qt::DisplayRole)
303  value = channel.getVoltageMin();
304  else if (role == Qt::ToolTipRole)
305  value = "Double-click to change sample min/max values";
306  break;
307 
308  case kVoltageMax:
309  if (role == Qt::DisplayRole)
310  value = channel.getVoltageMax();
311  else if (role == Qt::ToolTipRole)
312  value = "Double-click to change sample min/max values";
313  break;
314 
315  case kVisible:
316 
317  // The visibility state is represented as a checkbox which is either checked or unchecked.
318  //
319  if (role == Qt::CheckStateRole || role == Qt::EditRole)
320  value = connection->isVisible() ? Qt::Checked : Qt::Unchecked;
321  else if (role == Qt::ToolTipRole)
322  value = connection->isVisible() ?
323  "Click to stop plotting channel data" :
324  "Click to begin plotting channel data";
325 
326  break;
327 
328  case kFrozen:
329 
330  // The frozen state is represented as a checkbox which is either checked or unchecked.
331  //
332  if (role == Qt::CheckStateRole || role == Qt::EditRole)
333  value = connection->isFrozen() ? Qt::Checked : Qt::Unchecked;
334  else if (role == Qt::ToolTipRole)
335  value = connection->isFrozen() ?
336  "Click to resume updating channel data" :
337  "Click to freeze channel data";
338  break;
339 
340  case kShowPeakBars:
341 
342  // The showPeakBars state is represented as a checkbox which is either checked or unchecked.
343  //
344  if (role == Qt::CheckStateRole || role == Qt::EditRole)
345  value = connection->isShowingPeakBars() ?
346  Qt::Checked : Qt::Unchecked;
347  else if (role == Qt::ToolTipRole)
348  value = connection->isShowingPeakBars() ?
349  "Click to hide peak bars" :
350  "Click to show peak bars";
351  break;
352 
353  case kDisplayCount:
354  if (role == Qt::DisplayRole)
355  value = uint(channel.getDisplayCount());
356  break;
357 
358  default:
359  break;
360  }
361 
362  return value;
363 }
