179 void
180 AppBase::restoreWindows()
181 {
182  Logger::ProcLog log("restoreWindows", Log());
183  LOGINFO << std::endl;
184 
185  // Restore all previously-saved MainWindowBase objects.
186  //
187  QSettings settings;
188  int restored = 0;
189  int count = settings.beginReadArray("Windows");
190  LOGDEBUG << "restoring " << count << " windows" << std::endl;
191  for (int index = 0; index < count; ++index) {
192  settings.setArrayIndex(index);
193  QString objectName = settings.value(kObjectName).toString();
194  LOGDEBUG << "restoring window " << index << ' ' << objectName
195  << std::endl;
196  MainWindowBase* window = makeAndInitializeNewMainWindow(objectName);
197  if (window) {
198  window->showAndRaise();
199  ++restored;
200  }
201  }
202 
203  settings.endArray();
204 
205  // If this is the first time the application has been run, create a new main window now.
206  //
207  if (restored == 0) {
208  LOGDEBUG << "none saved" << std::endl;
209  MainWindowBase* window = makeAndInitializeNewMainWindow("");
210  if (window) {
211  window->showAndRaise();
212  }
213  }
214 
215  // Now restore all registered tool windows.
216  //
217  emit restoreToolWindows();
218 }

288 void
289 AppBase::applicationQuit()
290 {
291  Logger::ProcLog log("applicationQuit", Log());
292 
293  // If we are already in 'quit' sequence, don't continue.
294  //
295  if (quitting_) {
296  return;
297  }
298 
299  quitting_ = true;
300  LOGWARNING << "*** USER QUITTING ***" << std::endl;
301 
302  // We are about to quit. Give all of the windows a chance to save their configurations. Record the logging
303  // configuration. Broadcast a "shutdown" signal.
304  //
305  windowManager_->saveWindows();
306  QSettings settings;
307  settings.setValue(kLogConfigurationFile, lastConfigurationFile_);
308 
309  // Close all of the open windows. For MainWindowBase objects, this will invoke unmanageMainWindow(), which
310  // will reduce the number of managed windows. A MainWindowBase object may reject a close via its
311  // closeEvent() event handler. We detect when that occurs and stop the quit action.
312  //
313  closeAllWindows();
314  if (getNumManagedWindows() != 0) {
315  quitting_ = false;
316  LOGWARNING << "*** NOT QUITTING *** - " << getNumManagedWindows() << std::endl;
317  return;
318  }
319 
320  // Tell interested parties that we are really shutting down.
321  //
322  emit shutdown();
323 
324  quit();
325 }
