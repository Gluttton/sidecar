30 void
31 RunningMedian::setWindowSize(size_t windowSize, double initialValue)
32 {
33  Logger::ProcLog log("setWindowSize", Log());
34  LOGINFO << "windowSize: " << windowSize << " initialValue: " << initialValue << std::endl;
35 
36  if (windowSize == 0) {
37  throw Utils::Exception("windowSize must be non-zero");
38  }
39 
40  // If no change in size, just reset to the initial value.
41  //
42  if (windowSize_ == windowSize) {
43  clear(initialValue);
44  return;
45  }
46 
47  // Release any existing Node objects and prepare to create new ones for the new window size.
48  //
49  if (oldest_) {
50  releaseNodes();
51  }
52 
53  windowSize_ = windowSize;
54  median_ = initialValue;
55  sum_ = initialValue * windowSize;
56  minValue_ = maxValue_ = initialValue;
57  isOdd_ = windowSize & 1;
58 
59  // Calculate how many Node objects there are between finger Nodes.
60  //
61  fingerSpan_ = size_t(::sqrt(windowSize));
62  size_t fingerIndex = 0;
63 
64  LOGDEBUG << "fingerSpan: " << fingerSpan_ << std::endl;
65 
66  // Calculate the finger index where the median value would be found. TODO: change to locate the closest
67  // finger in distance.
68  //
69  size_t midPoint;
70  midPoint = (windowSize - 1) / 2;
71 
72  // Calculate which finger Nodes to use for accessing the median entry, and how many Nodes we must traverse
73  // before we reach it.
74  //
75  medianFingerIndex_ = size_t(midPoint / fingerSpan_);
76  medianFingerOffset_ = midPoint - medianFingerIndex_ * fingerSpan_;
77  LOGDEBUG << "midPoint: " << midPoint << " medianFingerIndex: " << medianFingerIndex_
78  << " medianFingerOffset: " << medianFingerOffset_ << std::endl;
79 
80  // Create our value Nodes, initialized to the given value and properly linked up.
81  //
82  for (size_t index = 0; index < windowSize; ++index) {
83 
84  Node* node = new Node;
85  node->value_ = initialValue;
86  node->younger_ = 0;
87  node->bigger_ = 0;
88 
89  if (! oldest_) {
90  oldest_ = node;
91  }
92  else {
93  youngest_->bigger_ = node;
94  youngest_->younger_ = node;
95  }
96 
97  node->smaller_ = youngest_;
98  youngest_ = node;
99 
100  // Periodically create a finger entry. NOTE: since ::sqrt(1) is 1, we will always have at least one
101  // finger Node object.
102  //
103  if (index == fingerIndex) {
104  fingers_.push_back(node);
105  fingerIndex += fingerSpan_;
106  }
107  }
108 }

124 RunningMedian::Node*
125 RunningMedian::storeValue(double value)
126 {
127  // Unlink the oldest Node, and relink as the youngest.
128  //
129  Node* tmp = oldest_;
130 
131  if (value != tmp->value_) {
132  sum_ -= tmp->value_;
133  sum_ += value;
134 
135  if (tmp->value_ == maxValue_ && ! tmp->bigger_ && tmp->smaller_)
136  maxValue_ = tmp->smaller_->value_;
137 
138  if (tmp->value_ == minValue_ && ! tmp->smaller_ && tmp->bigger_)
139  minValue_ = tmp->bigger_->value_;
140 
141  tmp->value_ = value;
142  }
143 
144  if (tmp->younger_) {
145  oldest_ = tmp->younger_;
146  tmp->younger_ = 0;
147  youngest_->younger_ = tmp;
148  youngest_ = tmp;
149  }
150 
151  return tmp;
152 }

154 double
155 RunningMedian::addValue(double newValue)
156 {
157  static Logger::ProcLog log("addValue", Log());
158  LOGINFO << "newValue: " << newValue << std::endl;
159 
160  Node* left = 0;
161  Node* right = 0;
162 
163  // Check if we can insert before the first Node.
164  //
165  if (newValue <= fingers_.front()->value_) {
166  right = fingers_.front();
167  }
168  else {
169 
170  // Going backwards, locate the first finger Node that has a value less than the new one.
171  //
172  size_t fingerIndex = findLargestFingerSmallerThan(newValue);
173 
174  // Now walk forwards from the found finger Node
175  //
176  left = fingers_[fingerIndex];
177  right = left->bigger_;
178  while (right && newValue > right->value_) {
179  left = right;
180  right = right->bigger_;
181  }
182  }
183 
184  // At this point, left or right may be NULL, but never both at the same time.
185  //
186  double oldestValue = oldest_->value_;
187  LOGDEBUG << "left: " << left << " right: " << right << " oldest: " << oldest_ << " oldestValue: "
188  << oldestValue << std::endl;
189 
190  LOGDEBUG3 << "BEFORE" << std::endl;
191  if (log.showsDebug3()) dump();
192 
193  if (left == oldest_ || right == oldest_ || oldestValue == newValue) {
194 
195  // We want to insert at a point where we would reuse the oldest Node. Since it is already in the proper
196  // ordered linked-list position, just make it the youngest node and update its value and we're done.
197  //
198  LOGDEBUG << "reusing oldest entry" << std::endl;
199  storeValue(newValue);
200 
201  // Update min/max values, taking care of corner cases.
202  //
203  if (left == oldest_) {
204  maxValue_ = newValue;
205  if (windowSize_ == 1)
206  minValue_ = newValue;
207  }
208 
209  if (right == oldest_) {
210  minValue_ = newValue;
211  if (windowSize_ == 1)
212  maxValue_ = newValue;
213  }
214  }
215  else {
216 
217  bool shiftBigger = false;
218  std::vector<size_t> shifts;
219  if (oldestValue < newValue) {
220  shiftBigger = true;
221  size_t oldFingerIndex = findFingerIndexAtOrBeforeOldest();
222  LOGDEBUG << "oldestValue < newValue - oldFingerIndex: " << oldFingerIndex << std::endl;
223 
224  for (; oldFingerIndex < fingers_.size() && fingers_[oldFingerIndex]->value_ < newValue;
225  ++oldFingerIndex) {
226  Node* tmp = fingers_[oldFingerIndex];
227  shifts.push_back(oldFingerIndex);
228  LOGDEBUG << "will shift " << oldFingerIndex << " higher - tmp: " << tmp << " (" << tmp->value_
229  << ") oldest: " << oldest_ << " (" << oldestValue << ")" << std::endl;
230  }
231  }
232  else if (oldestValue > newValue) {
233  shiftBigger = false;
234  size_t oldFingerIndex = findFingerIndexAtOrAfterOldest();
235  LOGDEBUG << "oldestValue > newValue - oldFingerIndex: " << oldFingerIndex << std::endl;
236 
237  while (fingers_[oldFingerIndex]->value_ >= newValue) {
238  Node* tmp = fingers_[oldFingerIndex];
239  shifts.push_back(oldFingerIndex);
240  LOGDEBUG << "will shift " << oldFingerIndex << " lower - tmp: " << tmp << " (" << tmp->value_
241  << ") oldest: " << oldest_ << " (" << oldestValue << ")" << std::endl;
242  if (oldFingerIndex == 0)
243  break;
244  --oldFingerIndex;
245  }
246  }
247 
248  // Now resuse the oldest node and prepare to move it to its new location in the ordered list.
249  //
250  Node* tmp = storeValue(newValue);
251  if (left == tmp) left = left->smaller_;
252  if (right == tmp) right = right->bigger_;
253 
254  Node* smaller = tmp->smaller_;
255  Node* bigger = tmp->bigger_;
256 
257  // Unlink the Node from the ordered chain. WARNING: the operations from here to the end of this scope
258  // are very order dependent; change at your peril.
259  //
260  if (! smaller) {
261  bigger->smaller_ = 0;
262  }
263  else if (! bigger) {
264  smaller->bigger_ = 0;
265  }
266  else {
267  smaller->bigger_ = bigger;
268  bigger->smaller_ = smaller;
269  }
270 
271  LOGDEBUG << "tmp: " << tmp << " left: " << left << " right: " << right << " min: " << minValue_
272  << " max: " << maxValue_ << std::endl;
273 
274  // Begin to relink the Node into the ordered chain. However, DO NOT update the new Node's ordered links
275  // until after we apply any shifts to the finger Nodes, since the finger Nodes could point to the old
276  // node.
277  //
278  if (left) left->bigger_ = tmp;
279  if (right) right->smaller_ = tmp;
280 
281  // Now, apply the shifts.
282  //
283  if (shiftBigger) {
284  std::for_each(shifts.begin(), shifts.end(), boost::bind(&RunningMedian::shiftFingerBigger,
285  this, _1));
286  }
287  else {
288  std::for_each(shifts.begin(), shifts.end(), boost::bind(&RunningMedian::shiftFingerSmaller,
289  this, _1));
290  }
291 
292  // Finally, complete the relinking of the old node into its proper place in the ordered linked-list.
293  //
294  tmp->bigger_ = right;
295  if (! right) maxValue_ = newValue;
296 
297  tmp->smaller_ = left;
298  if (! left) minValue_ = newValue;
299  }
300 
301  LOGDEBUG3 << "AFTER" << std::endl;
302  if (log.showsDebug3()) dump();
303 
304  // Update the median value. Locate the proper node.
305  //
306  Node* tmp = fingers_[medianFingerIndex_];
307  for (size_t index = 0; index < medianFingerOffset_; ++index) {
308  tmp = tmp->bigger_;
309  }
310 
311  if (isOdd_) {
312  median_ = tmp->value_;
313  }
314  else {
315  median_ = (tmp->value_ + tmp->bigger_->value_) / 2;
316  }
317 
318  LOGDEBUG << "median: " << median_ << std::endl;
319 
320  return median_;
321 }

395 RunningMedian::findFingerIndexAtOrBeforeOldest() const
396 {
397  // Locate the first finger whose value is less than or equal to the oldest value in the set. If we found the
398  // oldest value, just return the index of the finger.
399  //
400  double oldestValue = oldest_->value_;
401  size_t index = fingers_.size();
402  while (index > 0) {
403  --index;
404  if (fingers_[index]->value_ < oldestValue)
405  break;
406  if (fingers_[index] == oldest_)
407  return index;
408  }
409 
410  // We need to walk forward to until we locate the oldest node, incrementing our index value each time we
411  // come to a finger node.
412  //
413  Node* node = fingers_[index++]->bigger_;
414  while (node != oldest_ && index < fingers_.size() - 1) {
415  node = node->bigger_;
416  if (node == fingers_[index])
417  ++index;
418  }
419 
420  return index;
421 }
