63 {
64  // Search for an extension, but don't treat any '.' found before a '/' character.
65  //
66  auto dotPos = filePath_.rfind('.');
67  auto slashPos = filePath_.rfind('/');
68  if (slashPos != std::string::npos && dotPos != std::string::npos && dotPos < slashPos) {
69  dotPos = std::string::npos;
70  }
71 
72  if (ext.size() == 0) {
73 
74  // Removing any existing extension.
75  //

100 void
101 FilePath::expandEnvVars()
102 {
103  auto pos = 0;
104  do {
105  pos = filePath_.find_first_of("$~", pos, 2);
106  if (pos == std::string::npos) break;
107 
108  auto name = pos;
109  std::string var;
110  const char* value = nullptr;
111 
112  // Treat '~' as equivalent of $HOME.
113  //
114  if (filePath_[pos++] == '~') {
115  var = "HOME";
116  }
117  else {
118 
119  // Find location of first character that does not belong to a shell variable name.
120  //
121  while (pos < filePath_.size()) {
122  auto c = filePath_[pos];
123  if (c == '_' || c == '{' || c == '}' || isalnum(c)) {
124  ++pos;
125  if (c != '}') continue;
126  }
127  break;
128  }
129 
130  // Extract the name of the environment variable to fetch. Properly account for optional '{' '}'
131  // characters.
132  //
133  if (filePath_[name + 1] == '{') {
134  var = filePath_.substr(name + 2, pos - name - 3);
135  }
136  else {
137  var = filePath_.substr(name + 1, pos - name - 1);
138  }
139  }
140 
141  // Fetch the value of the variable and if OK, replace the variable name with its value.
142  //
143  value = ::getenv(var.c_str());
144  if (! value) {
145  std::cerr << "FilePath::expandEnvVars: environment variable '" << var << "' does not exist\n";
146  value = "";
147  }
148 
149  filePath_.replace(name, pos - name, value);
150 
151  // Start the next search near the beginning of the previous match so we can expand variables in previous
152  // expansions. Note that we do not handle embedded epansions like ${FOO_${BAR}}. C'est la vie.
153  //
154  pos = name + 1;
155 
156  } while (1);
157 }
