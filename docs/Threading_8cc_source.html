54 static void
55 fillTimespec(double offset, bool abs, timespec& ts) throw(std::out_of_range)
56 {
57  static const long kNanosecondsPerSecond = 1000000000L;
58 
59  // Get current time in seconds/microseconds.
60  //
61  if (abs) {
62  timeval now;
63  gettimeofday(&now, NULL);
64  ts.tv_sec = now.tv_sec;
65  ts.tv_nsec = now.tv_usec * 1000;
66  }
67  else {
68  ts.tv_sec = 0;
69  ts.tv_nsec = 0;
70  }
71 
72  // Now add in offset to make a time in the future.
73  //
74  double tmp = std::floor(offset);
75  if (tmp > std::numeric_limits<long>::max()) {
76  throw std::out_of_range("fillTimespec");
77  }
78 
79  // Extract the number of whole seconds from given value.
80  //
81  ts.tv_sec += static_cast<long>(tmp);
82  tmp = offset - tmp;
83 
84  // If the offset value still has something in it (> 0.0 && < 1.0) then use as a multiplier to obtain a
85  // fraction of nanoseconds. Add the fraction to the timespec field, but detect and massage any overflow.
86  //
87  ts.tv_nsec += static_cast<long>(tmp * kNanosecondsPerSecond);
88  if (ts.tv_nsec >= kNanosecondsPerSecond) {
89  ts.tv_sec += ts.tv_nsec / kNanosecondsPerSecond;
90  ts.tv_nsec %= kNanosecondsPerSecond;
91  }
92 }
