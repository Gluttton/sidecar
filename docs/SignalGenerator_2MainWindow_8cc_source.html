397 void
398 MainWindow::generateMessages()
399 {
400  Logger::ProcLog log("generateMessages", Log());
401 
402  if (! p_->initialized_) return;
403 
404  p_->vme_.header.msgDesc = (Messages::VMEHeader::kIRIGValidMask |
405  Messages::VMEHeader::kAzimuthValidMask |
406  Messages::VMEHeader::kPRIValidMask);
407 
408  p_->doComplex_ = p_->gui_->doComplex_->isChecked();
409  if (p_->doComplex_) {
410  p_->vme_.header.msgDesc |= (Messages::VMEHeader::kPackedIQ << 16);
411  }
412  else {
413  p_->vme_.header.msgDesc |= (Messages::VMEHeader::kPackedReal << 16);
414  }
415 
416  p_->priRate_ = 1.0 / getEmitterFrequency();
417 
418  p_->vme_.header.pri = 0;
419  p_->vme_.header.timeStamp = 0;
420  p_->vme_.header.irigTime = 0;
421  p_->vme_.header.azimuth = 0;
422  p_->vme_.rangeMin = p_->gui_->rangeMin_->value();
423  p_->vme_.rangeFactor = (p_->gui_->rangeMax_->value() -
424  p_->gui_->rangeMin_->value()) /
425  p_->gui_->sampleCount_->value();
426 
427  p_->bufferSize_ = p_->gui_->sampleCount_->value();
428  if (p_->doComplex_) p_->bufferSize_ *= 2;
429 
430  p_->accumulator_.resize(p_->bufferSize_, 0.0);
431  p_->clock_ = Time::TimeStamp(0.0);
432 
433  int valueMin = p_->gui_->valueMin_->value();
434  int valueMax = p_->gui_->valueMax_->value();
435 
436  int activeCount = 0;
437  for (int index = 0; index < p_->model_->rowCount(); ++index) {
438  GeneratorConfiguration* cfg = p_->model_->getConfiguration(index);
439  if (cfg->isEnabled()) {
440  ++activeCount;
441  cfg->reset();
442  }
443  }
444 
445  // Calculate the coefficients in the linear transform from amplitudes to sample values. Amplitudes values
446  // span [-1 * rowCount, 1 * rowCount], while sample values span [valueMin, valueMax]: alpha = (Vmax -
447  // Vmin) / (Amax - Amin) = (Vmax - Vmin) / 2 V = (A / rowCount - Amin) * alpha + Vmin Simplifying
448  // gets us: beta = -Amin * alpha + Vmin = alpha + Vmin V = A * alpha / rowCount + beta
449  //
450  p_->alpha_ = (valueMax - valueMin) / 2.0;
451  p_->beta_ = valueMin + p_->alpha_;
452  p_->alpha_ /= activeCount;
453  p_->shaftMovePerMessage_ = 65536.0 / p_->gui_->radialCount_->value();
454 
455  LOGDEBUG << "alpha: " << p_->alpha_ << " beta: " << p_->beta_ << std::endl;
456 
457  p_->emitter_->clear();
458  if (! p_->timer_->isActive()) {
459  p_->timer_->start();
460  }
461 
462  generateOneMessage();
463 }

465 void
466 MainWindow::generateOneMessage()
467 {
468  static Logger::ProcLog log("generateOneMessage", Log());
469 
470  if (p_->doComplex_) {
471  for (int index = 0; index < p_->model_->rowCount(); ++index) {
472  GeneratorConfiguration* cfg = p_->model_->getConfiguration(index);
473  if (cfg->isEnabled())
474  cfg->complexAddTo(p_->accumulator_);
475  }
476  }
477  else {
478  for (int index = 0; index < p_->model_->rowCount(); ++index) {
479  GeneratorConfiguration* cfg = p_->model_->getConfiguration(index);
480  if (cfg->isEnabled())
481  cfg->normalAddTo(p_->accumulator_);
482  }
483  }
484 
485  double azimuth = p_->shaftMovePerMessage_ * p_->vme_.header.pri;
486  p_->vme_.header.azimuth = uint32_t(::rint(azimuth)) % 65536;
487 
488  ++p_->vme_.header.pri;
489  Messages::Video::Ref msg = Messages::Video::Make("SignalGenerator",
490  p_->vme_,
491  p_->bufferSize_);
492  msg->setCreatedTimeStamp(p_->clock_);
493  p_->clock_ += p_->priRate_;
494  p_->vme_.header.irigTime += p_->priRate_;
495 
496  for (size_t index = 0; index < p_->bufferSize_; ++index) {
497  double value = p_->accumulator_[index] * p_->alpha_ + p_->beta_;
498  msg->push_back(int(::rint(value)));
499  p_->accumulator_[index] = 0.0;
500  }
501 
502  p_->emitter_->addMessage(msg);
503 
504  if ((p_->vme_.header.pri % 50) == 0)
505  statusBar()->showMessage(
506  QString("Generated message %1").arg(p_->vme_.header.pri));
507 
508  if (p_->vme_.header.pri == uint32_t(p_->gui_->messageCount_->value())) {
509  statusBar()->showMessage("Done.", 5000);
510  p_->timer_->stop();
511  }
512 }
