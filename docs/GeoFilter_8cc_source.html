117  Logger::ProcLog log("loadConfig", getLog());
118 
119  QFile file(QString::fromStdString(path));

122  return false;
123  }

137  static const QString kFilterEntity("filter");
138 
139  QDomElement top = dom.namedItem(kRootEntity).toElement();
140  if (top.isNull()) {
141  LOGERROR << "missing <geofilter> root entity" << std::endl;
142  return false;
143  }

145  static const QRegExp trueRE("(t(rue)?)|(y(es)?)|1",
146  Qt::CaseInsensitive);
147  static const QRegExp falseRE("(f(alse)?)|(n(o)?)|0",
148  Qt::CaseInsensitive);
149 
150  QDomElement filterSpec = top.firstChildElement(kFilterEntity);
151  while (! filterSpec.isNull()) {
152  Filter filter;
153  bool enabled;
154 
155  filter.name = filterSpec.attribute("name", "").trimmed();
156  if (filter.name.isEmpty()) {
157  LOGERROR << "missing name attribute" << std::endl;
158  return false;
159  }
160 
161  QString value = filterSpec.attribute("enabled", "true").trimmed();
162  if (trueRE.exactMatch(value)) {
163  enabled = true;
164  }
165  else if (falseRE.exactMatch(value)) {
166  enabled = false;
167  }
168  else {
169  LOGERROR << "invalid/missing enabled attribute value - "
170  << value << std::endl;
171  return false;
172  }
173 
174  QDomNodeList children = filterSpec.childNodes();
175  for (int index = 0; index < children.count(); ++index) {
176 
177  QDomElement element(children.item(index).toElement());
178  if (element.isNull()) {
179  LOGERROR << "expected filter entity" << std::endl;
180  return false;
181  }
182 
183  QString tag = element.tagName();
184  if (tag == "range") {
185  QStringList values = element.text().split(
186  ' ', QString::SkipEmptyParts);
187  if (values.size() != 2) {
188  LOGERROR << "invalid range values" << std::endl;
189  return false;
190  }
191  bool ok = false;
192  filter.rangeMin = values[0].toDouble(&ok);
193  if (! ok) {
194  LOGERROR << "invalid range min value" << std::endl;
195  return false;
196  }
197  ok = false;
198  filter.rangeMax = values[1].toDouble(&ok);
199  if (! ok) {
200  LOGERROR << "invalid range max value" << std::endl;
201  return false;
202  }
203  }
204  else if (tag == "azimuth") {
205  QStringList values = element.text().split(
206  ' ', QString::SkipEmptyParts);
207  if (values.size() != 2) {
208  LOGERROR << "invalid azimuth values" << std::endl;
209  return false;
210  }
211  bool ok = false;
212  filter.azMin =
213  Utils::normalizeRadians(
214  Utils::degreesToRadians(values[0].toDouble(&ok)));
215  if (! ok) {
216  LOGERROR << "invalid azimuth min value" << std::endl;
217  return false;
218  }
219  ok = false;
220  filter.azMax =
221  Utils::normalizeRadians(
222  Utils::degreesToRadians(values[1].toDouble(&ok)));
223  if (! ok) {
224  LOGERROR << "invalid azimuth max value" << std::endl;
225  return false;
226  }
227  }
228  else if (tag == "attenuation") {
229  bool ok = false;
230  filter.attenuation = element.text().toDouble(&ok);
231  if (! ok) {
232  LOGERROR << "invalid attenuation value" << std::endl;
233  return false;
234  }
235  }
236  else if (tag == "offset") {
237  bool ok = false;
238  filter.offset = element.text().toDouble(&ok);
239  if (! ok) {
240  LOGERROR << "invalid offset value" << std::endl;
241  return false;
242  }
243  }
244  else if (tag == "clamp") {
245  QStringList values = element.text().split(
246  ' ', QString::SkipEmptyParts);
247  if (values.size() != 2) {
248  LOGERROR << "invalid clamp values" << std::endl;
249  return false;
250  }
251  bool ok = false;
252  filter.clampMin = values[0].toDouble(&ok);
253  if (! ok) {
254  LOGERROR << "invalid clamp min value" << std::endl;
255  return false;
256  }
257  ok = false;
258  filter.clampMax = values[1].toDouble(&ok);
259  if (! ok) {
260  LOGERROR << "invalid clamp max value" << std::endl;
261  return false;
262  }
263  }
264  else {
265  LOGERROR << "unknown entity - " << tag << std::endl;
266  return false;
267  }
268  }
269 
270  if (enabled) {
271  LOGWARNING << "adding filter - " << filter.name
272  << " az: "
273  << Utils::radiansToDegrees(filter.azMin) << '/'
274  << Utils::radiansToDegrees(filter.azMax)
275  << " range: "
276  << filter.rangeMin << '/' << filter.rangeMax
277  << " attenuation: " << filter.attenuation
278  << " offset: " << filter.offset
279  << " clamp: "

281  << std::endl;
282  p_->filters_.push_back(filter);
283  }
284 
285  filterSpec = filterSpec.nextSiblingElement("filter");
286  }

291 bool
292 GeoFilter::processInput(const Messages::Video::Ref& inMsg)
293 {
294  static Logger::ProcLog log("processInput", getLog());
295 
296  // Create a new message to hold the output of what we do. Note that although we pass in the input message, the new
297  // message does not contain any data.
298  //
299  Messages::Video::Ref outMsg(Messages::Video::Make("GeoFilter", inMsg));
300  outMsg->getData() = inMsg->getData();
301  if (! enabled_->getValue())
302  return send(outMsg);
303 
304  double azimuth = inMsg->getAzimuthStart();
305 
306  const std::vector<Filter>& filters(p_->filters_);
307  for (size_t index = 0; index < filters.size(); ++index) {
308  const Filter& filter(filters[index]);
309  if (filter.azMin <= azimuth && azimuth <= filter.azMax) {
310 
311  LOGDEBUG << "applying filter " << filter.name << std::endl;
312 
313  double rangeMin = inMsg->getRangeMin();
314  double rangeFactor = inMsg->getRangeFactor();
315 
316  double offset = (filter.rangeMin - rangeMin) / rangeFactor;
317  if (offset < 0.0) offset = 0.0;
318 
319  LOGDEBUG << "begin offset: " << offset << std::endl;
320 
321  Messages::Video::const_iterator in = inMsg->begin() +
322  size_t(offset);
323  Messages::Video::iterator out = outMsg->begin() +
324  size_t(offset);
325 
326  offset = (filter.rangeMax - rangeMin) / rangeFactor;
327  if (offset < 0.0)
328  offset = 0.0;
329  else if (offset > inMsg->size())
330  offset = inMsg->size();
331 
332  LOGDEBUG << "end offset: " << offset << std::endl;
333 
334  Messages::Video::const_iterator end = inMsg->begin() +
335  size_t(offset);
336 
337  while (in < end) {
338  double v = *in++ * filter.attenuation + filter.offset;
339  if (v < filter.clampMin) v = filter.clampMin;
340  if (v > filter.clampMax) v = filter.clampMax;
341  *out++ = ::rint(v);
342  }
343  }
344  }
345 
346  bool rc = send(outMsg);
347  LOGDEBUG << "rc: " << rc << std::endl;
348  return rc;
349 }
