62 bool
63 WindowedMinMax::processInput(const Messages::Video::Ref& msg)
64 {
65  static Logger::ProcLog log("processInput", getLog());
66 
67  // Create a new message to hold the output of what we do. Note that although we pass in the input message,
68  // the new message does not contain any data.
69  //
70  Messages::Video::Ref out(
71  Messages::Video::Make("WindowedMinMax::processInput", msg));
72 
73  int windowCount = windowCount_->getValue();
74  int windowSize = windowSize_->getValue();
75  int windowOffset = windowOffset_->getValue();
76 
77  // Reset the running window sum values
78  //
79  windowSums_.clear();
80  windowIndices_.clear();
81 
82  // Generate the sliding window indices and sums for those below the first sample point at index 0. Since
83  // there is nothing below the first sample point, these sums are zero.
84  //
85  int index = - (windowCount * windowSize + windowOffset);
86  for (int count = 0; count < windowCount; ++count) {
87  int sum = 0;
88  windowSums_.push_back(sum);
89  windowIndices_.push_back(index);
90  index += windowSize;
91  }
92 
93  // Generate the sliding window indices and sums for those above the first sample point at index 0.
94  //
95  index = windowOffset;
96  for (int count = 0; count < windowCount; ++count) {
97  windowIndices_.push_back(index);
98 
99  // Calculate the sum of the values in one window.
100  //
101  int sum = 0;
102  for (int sample = 0; sample < windowSize; ++sample)
103  sum += msg[index++];
104 
105  windowSums_.push_back(sum);
106  }
107 
108  // Now that we are primed, process the message, shifting all of the windows by one sample each iteration.
109  //
110  int msgSize = msg->size();
111  for (int sample = 0; sample < msgSize; ++sample) {
112 
113  // Calculate the output value derived from the sum windows.
114  //
115  out->push_back((this->*valueGeneratorProc_)());
116 
117  // Shift the windows.
118  //
119  for (size_t count = 0; count < windowIndices_.size(); ++count) {
120 
121  // Update window sums by removing the oldest element and adding the new one. Always validate indices
122  // before using to protect against ends of the input message.
123  //
124  int value = windowSums_[count];
125  index = windowIndices_[count];
126  if (index >= 0 && index < msgSize)
127  value -= msg[index];
128 
129  index += windowSize;
130  if (index >= 0 && index < msgSize)
131  value += msg[index];
132 
133  windowSums_[count] = value;
134 
135  // Shift indices up one.
136  //
137  windowIndices_[count] = index - windowSize + 1;
138  }
139  }
140 
141  bool rc = send(out);
142  LOGDEBUG << "rc: " << rc << std::endl;
143  return rc;
144 }
