62  }

66  QDomElement def = stream.firstChildElement();
67  while (! def.isNull()) {

70

76  }
77  else if (type == "fileout") {

80  else if (type == "publisher") {

82  }
83  else if (type == "subscriber") {
84  builder.makeDataSubscriber(def);
85  }
86  else if (type == "vme") {
87  builder.makeVMEReader(def);

104 IO::Stream::Ref
105 StreamBuilder::release()
106 {
107  Logger::ProcLog log("release", Log());
108  LOGINFO << std::endl;
109 
110  // If the stream contains a FileReaderTask that has its signalEndOfFile attribute set, add a ShutdownMonitor
111  // task to detect the shutdown control message and properly shutdown the runner.
112  //
113  if (needShutdownMonitor_) {
114  stream_->push(new ShutdownMonitorModule(stream_));
115  needShutdownMonitor_ = false;
116  }
117 
118  // The XML nodes for a stream appear in top-down fashion, but ACE::Stream pushes tasks in bottom-up fashion.
119  //
120  for (std::vector<IO::Module*>::reverse_iterator pos = modules_.rbegin();
121  pos != modules_.rend(); ++pos) {
122  if (needShutdownMonitor_) {
123  static const int kQueueSize = 32 * 1024; // 32K
124  IO::Task::Ref task = (**pos).getTask();
125  ACE_Message_Queue<ACE_MT_SYNCH>* queue = task->msg_queue();
126  if (queue) {
127  queue->high_water_mark(kQueueSize);
128  queue->low_water_mark(kQueueSize - 1);
129  }
130  }
131 
132  stream_->push(*pos);
133  }
134 
135  modules_.clear();
136  return stream_;
137 }

223 void
224 StreamBuilder::connectInput(IO::Task::Ref task, std::string& type, std::string name, std::string channelName)
225 {
226  Logger::ProcLog log("connectInput", Log());
227  size_t taskIndex = task->getTaskIndex();
228  LOGINFO << taskIndex << " type: " << type << " name: " << name << " channelName: " << channelName
229  << std::endl;
230 
231  ChannelMap::iterator pos;
232  if (channelName.size() == 0) {
233 
234  // Create a pseudo channel name when one is not specified, and if it matches one created by a previous
235  // task.
236  //
237  for (int index = taskIndex - 1; index >= 0; --index) {
238  IO::Task::Ref prev(modules_[index]->getTask());
239  channelName = MakeDefaultChannelName(prev->getTaskIndex(), task->getNumInputChannels());
240  LOGDEBUG << "new channel name: " << channelName << std::endl;
241  pos = channels_.find(channelName);
242  if (pos != channels_.end())
243  break;
244  }
245  }
246  else {
247  pos = channels_.find(channelName);
248  }
249 
250  // If no match, report an error.
251  //
252  if (pos == channels_.end()) {
253  Utils::Exception ex;
254  ex << "unknown channel '" << channelName << "'";
255  log.thrower(ex);
256  }
257 
258  if (! name.size()) {
259  name = channelName;
260  }
261 
262  if (! type.size()) {
263  type = pos->second.getTypeName();
264  }
265 
266  // Verify that the output channel specification has the same type as the input one.
267  //
268  if (pos->second.getTypeName() != type) {
269  Utils::Exception ex;
270  ex << "input channel type '" << type << "' does not match output type '" << pos->second.getTypeName()
271  << "'";
272  log.thrower(ex);
273  }
274 
275  // Add this task to the output specification's list of recipients. Add the input channel to the task.
276  //
277  pos->second.addRecipient(task, task->getNumInputChannels());
278  IO::Channel channel(name, type);
279  channel.setSender(task);
280  task->addInputChannel(channel);
281 
282  LOGDEBUG << "connected input channel " << name << '/' << type << " to task " << taskIndex << std::endl;
283 }

390 void
391 StreamBuilder::makeAlgorithm(const QDomElement& xml)
392 {
393  Logger::ProcLog log("makeAlgorithm", Log());
394  LOGINFO << std::endl;
395 
396  QString dll(xml.attribute("dll"));
397  if (dll.isEmpty()) {
398  Utils::Exception ex("no name for 'dll' element");
399  log.thrower(ex);
400  }
401 
402  QString name(xml.attribute("name", dll));
403  if (! name.size()) {
404  name = dll;
405  }
406 
407  long threadFlags = getThreadFlags(xml.attribute(kScheduler));
408  long threadPriority = getThreadPriority(xml.attribute(kThreadPriority));
409  bool threaded = true;
410  if (xml.hasAttribute(kThreaded)) {
411  QString tmp = xml.attribute(kThreaded);
412  if (tmp == "false" || tmp == "0")
413  threaded = false;
414  }
415 
416  Algorithms::ControllerModule* module = new Algorithms::ControllerModule(stream_);
417  addModule(xml, module);
418 
419  Algorithms::Controller::Ref controller = module->getTask();
420 
421  if (controller->getNumInputChannels() == 0) {
422  LOGWARNING << "no input channels defined for the task " << name.toStdString() << std::endl;
423  }
424 
425  if (controller->getNumOutputChannels() == 0) {
426  LOGWARNING << "no output channels defined for the task " << name.toStdString() << std::endl;
427  }
428 
429  controller->setXMLDefinition(xml);
430 
431  if (! controller->openAndInit(dll.toStdString(), name.toStdString(), 0, threadFlags, threadPriority,
432  threaded)) {
433  Utils::Exception ex("unable to open controller for ");
434  ex << dll.toStdString();
435  log.thrower(ex);
436  }
437 
438  QDomElement param = xml.firstChildElement("param");
439  if (! param.isNull()) {
440 
441  // Build up an XML-RPC <value> request that contains settings for all of the <param> elements.
442  //
443  QString buffer;
444  QTextStream os(&buffer, QIODevice::WriteOnly);
445  os << "<value><array><data>";
446  do {
447  os << "<value><string>" << param.attribute("name")
448  << "</string></value><value><" << param.attribute("type")
449  << ">" << param.attribute("value")
450  << "</" << param.attribute("type") << "></value>";
451  param = param.nextSiblingElement("param");
452  } while (! param.isNull());
453 
454  // Terminate the request, then post the change request to the controller.
455  //
456  os << "</data></array></value>";
457  LOGDEBUG << buffer.toStdString() << std::endl;
458 
459  int offset = 0;
460  XmlRpc::XmlRpcValue init(buffer.toStdString(), &offset);
461  LOGDEBUG << init.toXml() << std::endl;
462  controller->injectControlMessage(IO::ParametersChangeRequest(init, true));
463  }
464 }
