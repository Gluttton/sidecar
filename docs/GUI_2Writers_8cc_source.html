329 bool
330 UDPMessageWriter::initialize(uint16_t port)
331 {
332  static Logger::ProcLog log("initialize", Log());
333  LOGINFO << "address: " << getServiceAddress() << std::endl;
334 
335  ACE_SOCK_Dgram& device(writer_.getDevice());
336 
337  // Open the writer device to use to send out data. For opening, we use the "any" address with a zero port.
338  // The system will assign us a port number to use.
339  //
340  ACE_INET_Addr address(u_short(port), "0.0.0.0", AF_INET);
341  LOGDEBUG << "opening UDP socket with address " << Utils::INETAddrToString(address) << std::endl;
342  if (device.open(address, AF_INET, 0, 1) == -1) {
343  LOGERROR << "failed to open multicast writer at " << getServiceAddress() << std::endl;
344  return false;
345  }
346 
347  LOGDEBUG << "opened socket " << writer_.getDevice().get_handle() << std::endl;
348 
349  // Set the hop time-to-live to a non-zero value so we don't flood the network.
350  //
351  unsigned char ttl = 10;
352  if (device.set_option(IPPROTO_IP, IP_MULTICAST_TTL, &ttl, sizeof(ttl)) == -1) {
353  LOGERROR << "failed to set TTL for " << getServiceAddress() << std::endl;
354  return false;
355  }
356 
357  if (! port) {
358 
359  // Fetch the port assigned to us by the system.
360  //
361  device.get_local_addr(address);
362  port = address.get_port_number();
363  }
364 
365  LOGDEBUG << "socket device address: " << Utils::INETAddrToString(address) << std::endl;
366 
367  // Set our Zeroconf publisher connection info.
368  //
369  setPort(port);
370 
371  // Build our multicast address now that we know our port number. Tell the
372  // writer where to send the UDP messages.
373  //
374  address.set(port, getServiceAddress().toStdString().c_str(), 1, AF_INET);
375  LOGDEBUG << "mcast address: " << Utils::INETAddrToString(address) << std::endl;
376  writer_.setRemoteAddress(address);
377 
378  // Create a heart-beat message reader socket for remote client to send
379  // their heart-beat messages to
380  //
381  heartBeatReader_ = new QUdpSocket(this);
382  if (! heartBeatReader_->bind(0)) {
383  LOGERROR << "failed to bind server socket" << std::endl;
384  return false;
385  }
386 
387  connect(heartBeatReader_, SIGNAL(readyRead()), SLOT(heartBeatReady()));
388 
389  quint16 heartBeatReaderPort = heartBeatReader_->localPort();
390  LOGDEBUG << "heartBeatReader port: " << heartBeatReaderPort << std::endl;
391  getPublisher()->setTextData("HeartBeatPort", QString::number(heartBeatReaderPort).toStdString());
392 
393  timer_ = new QTimer(this);
394  connect(timer_, SIGNAL(timeout()), SLOT(checkHeartBeats()));
395  timer_->start(5000);
396 
397  return true;
398 }

400 void
401 UDPMessageWriter::heartBeatReady()
402 {
403  static Logger::ProcLog log("heartBeatReady", Log());
404  LOGINFO << heartBeatReader_ << std::endl;
405 
406  QByteArray data;
407  data.resize(32);
408  QHostAddress senderAddress;
409  quint16 senderPort;
410  bool notify = false;
411 
412  do {
413  qint64 count = heartBeatReader_->readDatagram(data.data(), data.size(), &senderAddress, &senderPort);
414  LOGDEBUG << "count: " << count << std::endl;
415 
416  if (count < 1) {
417  LOGERROR << "failed readDatagram" << std::endl;
418  return;
419  }
420 
421  QString msg(data);
422  LOGDEBUG << "msg: " << msg << std::endl;
423 
424  QString key = QString("%1/%2").arg(senderAddress.toString()).arg(senderPort);
425  LOGDEBUG << "key: " << key << std::endl;
426 
427  HeartBeatMap::iterator pos = heartBeats_.find(key);
428  if (msg == "HI") {
429  if (pos == heartBeats_.end()) {
430  LOGDEBUG << "new entry" << std::endl;
431  ++active_;
432  notify = true;
433  }
434  heartBeats_[key].start();
435  }
436  else if (msg == "BYE") {
437  HeartBeatMap::iterator pos = heartBeats_.find(key);
438  if (pos != heartBeats_.end()) {
439  LOGDEBUG << "removing entry" << std::endl;
440  heartBeats_.erase(pos);
441  if (active_) {
442  --active_;
443  }
444  notify = true;
445  }
446  }
447  } while (heartBeatReader_->hasPendingDatagrams());
448 
449  if (notify) {
450  emit subscriberCountChanged(active_);
451  }
452 }
