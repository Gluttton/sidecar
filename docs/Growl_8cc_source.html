224 Growler::emitRegistration(boost::shared_ptr<Destination> destination)
225 {
226  // Build a registration payload.
227  //
228  PayloadBuffer buffer(kTypeRegistration, password_);
229  buffer << name_.size()
230  << uint8_t(notifications_.size())
231  << uint8_t(defaults_.size())
232  << name_;
233 
234  // Add the names of the notifications.
235  //
236  for (size_t index = 0; index < notifications_.size(); ++index)
237  buffer << notifications_[index].size()
238  << notifications_[index];
239 
240  // Add the indices of those notifications that are enabled
241  //
242  for (size_t index = 0; index < defaults_.size(); ++index)
243  buffer << defaults_[index];
244 
245  // Emit the payload to a specific destination.
246  //
247  PayloadBuffer::PayloadType payload(buffer.getPayload());
248  bool rc = socket_->send(&payload[0], payload.size(),
249  destination->address_, 0) ==
250  ssize_t(payload.size());
251  if (rc)
252  destination->emitRegistration_ = false;
253 
254  return rc;

258 Growler::notify(const std::string& notification, const std::string& title,
259  const std::string& description, int priority, bool sticky)
260 {
261  // The notification name must exist in our registry.
262  //
263  if (std::find(notifications_.begin(), notifications_.end(),
264  notification) == notifications_.end())
265  return false;
266 
267  // Build a notification payload.
268  //
269  PayloadBuffer buffer(kTypeNotification, password_);
270  unsigned short flags = (priority & 0x07) * 2;
271  if (priority < 0)
272  flags |= 0x08;
273  if (sticky)
274  flags |= 1;
275 
276  buffer << flags
277  << notification.size()
278  << title.size()
279  << description.size()
280  << name_.size()
281  << notification
282  << title
283  << description
284  << name_;
285 
286  // Emit the payload to all destinations. We will return false if we failed to send to at least one
287  // destination.
288  //
289  PayloadBuffer::PayloadType payload(buffer.getPayload());
290 
291  for (DestinationList::iterator pos = destinations_.begin();
292  pos != destinations_.end();) {
293 
294  bool ok = true;
295 
296  if ((*pos)->emitRegistration_)
297  ok = emitRegistration(*pos);
298 
299  if (ok)
300  socket_->send(&payload[0], payload.size(), (*pos)->address_, 0);
301 
302  ++pos;
303  }
304 
305  return true;
