324 void
325 CmdLineArgs::processShortOption(const std::string& opt)
326 {
327  // Do we have an option of the form `-a=value'?
328  //
329  std::string value("");
330  if (opt.size() > 2 && opt[2] == '=') {
331  auto c = opt[1];
332  auto def = findOptionDef(c);
333  if (def.valueTag == 0 || def.valueTag[0] == 0) {
334  error("cannot assign value to option", c);
335  }
336  opts_.add(std::string(1, c), def.longName, opt.substr(3));
337  }
338  else {
339 
340  // Process single-letter options. Those without required values can occur grouped together.
341  //
342  for (auto pos = 1; pos < opt.size(); ++pos) {
343  auto c = opt[pos];
344  auto def = findOptionDef(c);
345 
346  // Requires a value?
347  //
348  if (def.valueTag && def.valueTag[0]) {
349 
350  // Is this the only option present in the command-line argument?
351  //
352  if (opt.size() == 2) {
353 
354  // Do we have another argument to work with?
355  //
356  if (argc_ == 0) {
357  std::string err("missing ");
358  err += def.valueTag;
359  err += " for option";
360  error(err.c_str(), c);
361  }
362  value = getArg();
363  }
364  else {
365  error("cannot assign value to option within arg", opt);
366  }
367  }
368 
369  opts_.add(std::string(1, c), def.longName, value);
370  value = "";
371  }
372  }

376 CmdLineArgs::processLongOption(std::string opt)
377 {
378  // Strip off `--' prefix
379  //
380  opt.erase(0, 2);
381 
382  // See if option has an attached value. If so, save the value and strip from the option name.
383  //
384  std::string value("");
385  auto equalPos = opt.find('=');
386  if (equalPos != std::string::npos) {
387  value = opt.substr(equalPos + 1);
388  opt.erase(equalPos);
389  }
390 
391  // Does the option require an argument?
392  //
393  auto def = findOptionDef(opt);
394  if (def.valueTag && def.valueTag[0]) {
395 
396  // Yes. If none was provided via `=' then take the next argv value for it.
397  //
398  if (equalPos == std::string::npos) {
399  if (argc_ == 0) {
400  std::string err("missing ");
401  err += def.valueTag;
402  err += " for option";
403  error(err.c_str(), opt);
404  }
405  value = getArg();
406  }
407  }
408 
409  // Nope. So there should not have been an `=' found.
410  //
411  else if (equalPos != std::string::npos) {
412  error("cannot assign value to option", opt);
413  }
414 
415  opts_.add(opt, std::string(1, def.shortName), value);
416 }
