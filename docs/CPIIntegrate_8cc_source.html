103 // This routine is responsible for taking a set of PRIs which constitute a CPI and performing the appropriate
104 // operation.
105 //
106 bool
107 CPIIntegrate::processCPI()
108 {
109 
110  static Logger::ProcLog log("processCPI", getLog());
111  LOGINFO << std::endl;
112  LOGDEBUG << "Process CPI with " << buffer_.size() << " msgs" << std::endl;
113 
114  bool rc = true;
115 
116  size_t numCPIs = numCPIs_->getValue();
117  size_t cpiSpan = cpiSpan_->getValue();
118 
119  int last_row = -1;
120  Super::MessageQueue::iterator itr;
121  size_t cnt, row;
122 
123  if (buffer_.size() != cpiSpan) {
124  LOGWARNING << "Kicking an incomplete CPI! Expected " << cpiSpan << ", but received " << buffer_.size()
125  << std::endl;
126  return true;
127  }
128 
129  size_t msg_size = maxMsgSize();
130 
131  // Resize the vals_ buffer if necessary adding default value of 0
132  //
133  if (msg_size > vals_[0].size()) {
134  for(size_t i = 0; i < cpiSpan; i++) {
135  vals_[i].resize(msg_size, 0.0);
136  }
137  }
138 
139  MessageQueue * cpi = new MessageQueue;
140  uint32_t startingSequenceNumber_ = buffer_[0]->getRIUInfo().sequenceCounter;
141 
142  // Push this cpi into the buffer of CPIs
143  //
144  for(itr = buffer_.begin(), cnt = 0;
145  itr != buffer_.end() && cnt < cpiSpan; itr++, cnt++) {
146 
147  Messages::Video::Ref ref = boost::dynamic_pointer_cast<Messages::Video>(*itr);
148  row = ref->getRIUInfo().sequenceCounter - startingSequenceNumber_;
149  // pad any message locations whose PRIs were dropped
150  //
151  for(size_t row_index = last_row + 1; row_index < row; row_index++) {
152  LOGDEBUG << "Padding CPI(" << row_index << ") with zeroes" << std::endl;
153  Messages::Video::Ref msg(Messages::Video::Make(getName(), ref));
154  size_t N = vals_[0].size();
155  for(size_t i = 0; i < N; i++) {
156  msg->push_back(0);
157  }
158  cpi->push_back(msg);
159  }
160 
161  cpi->push_back(ref);
162  last_row = row++;
163  }
164 
165 
166  // Add contribution of new CPI to running average
167  //
168  for(size_t i = 0; i < cpiSpan; i++) {
169  Messages::Video::Ref ref = boost::dynamic_pointer_cast<Messages::Video>(cpi->front());
170  size_t n = ref->size();
171  // Add this message's values to the running average
172  for(size_t j = 0; j < n; j++) {
173  vals_[i][j] += ref[j];
174  }
175  // remove the message from the front of the queue and push it to the back of the queue
176  //
177  cpi->pop_front();
178  cpi->push_back(ref);
179  }
180 
181  cpis_.push_front(cpi);
182 
183  // Make sure we have enough CPI blocks to compute the averages
184  //
185  if(cpis_.size() < size_t(numCPIs + 1)) {
186  return rc;
187  }
188 
189  // Remove contribution of oldest CPI from running average
190  //
191  cpi = cpis_.back();
192  cpis_.pop_back();
193 
194  for(size_t i = 0; i < cpiSpan; i++) {
195  Messages::Video::Ref ref = boost::dynamic_pointer_cast<Messages::Video>(cpi->front());
196  size_t n = ref->size();
197  for(size_t j = 0; j < n; j++)
198  vals_[i][j] -= ref[j];
199  }
200  // De-allocate memory used to hold this CPI
201  //
202  delete cpi;
203 
204 
205  // Results are in reference to the middle CPI, ie average values are composed of CPIs that come before
206  // and after this CPI.
207  //
208  size_t middle = (numCPIs / 2) + 1;
209 
210  // Compute the averages and send the values out
211  //
212  for(size_t i = 0; i < cpiSpan; i++) {
213 
214  Messages::Video::Ref parent = boost::dynamic_pointer_cast<Messages::Video>((*(cpis_[middle]))[i]);
215  Messages::Video::Ref out(Messages::Video::Make(getName(), parent));
216  Messages::Video::Container& outputData(out->getData());
217  size_t N = vals_[i].size();
218 
219  for(size_t j = 0; j < N; j++) {
220  outputData.push_back(Messages::Video::DatumType(::rint(vals_[i][j] / numCPIs)));
221  }
222 
223  rc = rc && send(out);
224  }
225 
226  return rc;
227 }
