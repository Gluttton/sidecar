157 bool
158 Loader::Private::load(const QString& configurationPath)
159 {
160  Logger::ProcLog log("load", Log());
161 
162  // Clear out any cached values from the last time we were called.
163  //
164  reset();
165  configurationPath_ = configurationPath;
166  configurationName_ = QFileInfo(configurationPath).baseName();
167  filePath_ = configurationPath_;
168 
169  LOGDEBUG << "configurationName: " << configurationName_ << std::endl;
170 
171  // Open the XML file
172  //
173  QFile file(configurationPath_);
174  if (! file.open(QFile::ReadOnly | QFile::Text)) {
175  return finishLoad(Loader::kFailedFileOpen);
176  }
177 
178  // Load the XML data, creating a DOM object.
179  //
180  QDomDocument dom;
181  if (! dom.setContent(&file, false, &errorText_, &errorLine_, &errorColumn_)) {
182  return finishLoad(Loader::kFailedXMLParse);
183  }
184 
185  // We MUST have a <sidecar> element
186  //
187  QDomElement sidecar = dom.namedItem(kEntitySidecar).toElement();
188  if (sidecar.isNull()) {
189  return finishLoad(Loader::kMissingSidecarNode);
190  }
191 
192  // We MUST have a <radar> configuration element, though it may include
193  // another file.
194  //
195  QDomElement radar = sidecar.namedItem(kEntityRadar).toElement();
196  if (radar.isNull()) {
197  return finishLoad(Loader::kMissingRadarNode);
198  }
199 
200  QDir dir(QFileInfo(configurationPath).dir());
201  if (radar.hasAttribute(kAttributeFile)) {
202  QDomDocument doc;
203  radar = loadIncludeFile(dir.filePath(radar.attribute(kAttributeFile)), kEntityRadar, doc);
204  }
205 
206  if (! Messages::RadarConfig::Load(radar)) {
207  return finishLoad(Loader::kInvalidRadarNode);
208  }
209 
210  // We MUST have a <dp> (data-processor) element
211  //
212  QDomElement dp = sidecar.namedItem(kEntityDP).toElement();
213  if (dp.isNull()) {
214  return finishLoad(Loader::kMissingDPNode);
215  }
216 
217  // Obtain some configuration attributes from the <dp> element
218  //
219  if (dp.hasAttribute("loggerConfigPath")) {
220  loggerConfiguration_ = dp.attribute("loggerConfigPath");
221  LOGWARNING << "using deprecated 'loggerConfigPath' attribute instead " << "of '"
222  << kAttributeLoggerConfiguration << "'" << std::endl;
223  }
224  else {
225  loggerConfiguration_ = dp.attribute(kAttributeLoggerConfiguration);
226  }
227 
228  expandEnvVars(loggerConfiguration_);
229  LOGDEBUG << "loggerConfiguration: " << loggerConfiguration_ << std::endl;
230 
231  logsDirectory_ = dp.attribute(kAttributeLogsDirectory, kDefaultLogsDirectory);
232  if (logsDirectory_.isEmpty()) {
233  LOGWARNING << "using '/tmp' for 'logsDirectory'" << std::endl;
234  logsDirectory_ = "/tmp";
235  }
236 
237  LOGDEBUG << "logsDirectory: " << logsDirectory_ << std::endl;
238 
239  if (dp.hasAttribute("recordingBasePath")) {
240  recordingsDirectory_ = dp.attribute("recordingBasePath");
241  LOGWARNING << "using deprecated 'recordingBasePath' attribute instead " << "of '"
242  << kAttributeRecordingsDirectory << "'" << std::endl;
243  }
244  else {
245  recordingsDirectory_ = dp.attribute(kAttributeRecordingsDirectory, kDefaultRecordingsDirectory);
246  }
247 
248  expandEnvVars(recordingsDirectory_);
249  LOGDEBUG << "recordingsDirectory: " << recordingsDirectory_ << std::endl;
250 
251  initialProcessingState_ = dp.attribute(kAttributeInitialProcessingState, initialProcessingState_);
252  if (ProcessingState::GetValue(initialProcessingState_.toStdString()) == ProcessingState::kInvalid) {
253  LOGWARNING << "unknown processing state - " << dp.attribute(kAttributeInitialProcessingState)
254  << " - using 'Run'" << std::endl;
255  initialProcessingState_ = kDefaultRecordingsDirectory;
256  }
257 
258  // Load in the <runner> elements
259  //
260  QDomElement runner = dp.firstChildElement(kEntityRunner);
261  while (! runner.isNull()) {
262 
263  // Start with some default values.
264  //
265  RunnerConfig::Init cfg;
266  cfg.initialProcessingState = initialProcessingState_;
267  if (runner.hasAttribute(kAttributeFile)) {
268 
269  // Load in an include file with the runner definition
270  //
271  QDomDocument doc;
272  QDomElement other = loadIncludeFile(dir.filePath(runner.attribute(kAttributeFile)), kEntityRunner,
273  doc);
274  if (other.isNull()) {
275  return false;
276  }
277 
278  // Fetch the attributes from the include file, overriding the default values above.
279  //
280  updateInit(cfg, other);
281 
282  // Load the <stream> definitions from the include file.
283  //
284  if (! getStreams(dir, other, cfg.streams) || cfg.streams.isEmpty()) {
285  return finishLoad(Loader::kMissingStreams);
286  }
287  }
288  else {
289 
290  // Load the <stream> definitions from the original file.
291  //
292  if (! getStreams(dir, runner, cfg.streams) || cfg.streams.isEmpty()) {
293  return finishLoad(Loader::kMissingStreams);
294  }
295  }
296 
297  // Load the attributes from the original runner object, but possibly use values found in an include file
298  // if it is not found in the original.
299  //
300  updateInit(cfg, runner);
301 
302  if (cfg.name.isEmpty()) {
303  cfg.name = QString("Runner %1").arg(runnerConfigs_.size() + 1);
304  }
305 
306  LOGDEBUG << "name: " << cfg.name << std::endl
307  << "host: " << cfg.host << std::endl;
308 
309  if (! hostNames_.contains(cfg.host)) {
310  hostNames_.append(cfg.host);
311  }
312 
313  LOGDEBUG << "multicast: " << cfg.multicastAddress << std::endl
314  << "opts: " << cfg.opts << std::endl
315  << "scheduler: " << cfg.scheduler << std::endl
316  << "priority: " << cfg.priority << std::endl
317  << "cpu: " << cfg.cpuAffinity << std::endl;
318 
319  // Create a new RunnerConfig object to represent the found attributes and objects
320  //
321  runnerConfigs_.append(new RunnerConfig(parent_, cfg));
322 
323  // Move to the next <runner> element in the original document.
324  //
325  runner = runner.nextSiblingElement(kEntityRunner);
326 
327  } // Finished loading in runner configurations
328 
329  if (runnerConfigs_.isEmpty()) {
330  return finishLoad(Loader::kMissingRunners);
331  }
332 
333  return finishLoad(Loader::kOK);
334 }

336 bool
337 Loader::Private::getStreams(const QDir& dir, const QDomElement& runner, QList<QDomElement>& streams)
338 {
339  QDomElement stream = runner.namedItem(kEntityStream).toElement();
340  while(! stream.isNull()) {
341  if (stream.hasAttribute(kAttributeFile)) {
342 
343  // Read in the <stream> stanza from another file.
344  //
345  QDomDocument doc;
346  QDomElement other = loadIncludeFile(dir.filePath(stream.attribute(kAttributeFile)), kEntityStream,
347  doc);
348  if (other.isNull()) {
349  return false;
350  }
351 
352  streams.push_back(other);
353  }
354  else {
355  streams.push_back(stream);
356  }
357  stream = stream.nextSiblingElement(kEntityStream);
358  }
359  return true;
360 }

362 QDomElement
363 Loader::Private::loadIncludeFile(const QString& path, const QString& entity, QDomDocument& doc)
364 {
365  Logger::ProcLog log("loadIncludeFile", Log());
366  LOGINFO << "path: " << path << " entity: " << entity << std::endl;
367 
368  filePath_ = path;
369  QFile file(path);
370  if (! file.open(QFile::ReadOnly | QFile::Text)) {
371  LOGERROR << "failed to open file '" << path << "'" << std::endl;
372  finishLoad(Loader::kFailedFileOpen);
373  return QDomElement();
374  }
375 
376  includePaths_.append(path);
377 
378  if (! doc.setContent(&file, false, &errorText_, &errorLine_, &errorColumn_)) {
379  LOGERROR << "failed to parse XML file" << std::endl;
380  finishLoad(Loader::kFailedXMLParse);
381  return QDomElement();
382  }
383 
384  QDomElement node = doc.namedItem(entity).toElement();
385  if (node.isNull()) {
386  LOGERROR << "missing entity node" << std::endl;
387  finishLoad(Loader::kMissingEntityNode);
388  return QDomElement();
389  }
390 
391  return node;
392 }
