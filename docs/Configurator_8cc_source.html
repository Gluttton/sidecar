148 Configurator::load(std::istream& is)
149 {
150  managed_.clear();
151  while (std::getline(is, currentLine_)) {
152  if (currentLine_.size() == 0 || currentLine_[0] == '#') continue;
153  std::istringstream iss(currentLine_);
154  std::string logName;
155  if (! (iss >> logName)) throw MissingLoggerName();
156 
157  // Fetch/create the log object. Remember the Log object as one that we are working with.
158  //
159  Log& log(Log::Find(logName.c_str()));
160  managed_.insert(&log);
161  std::string what;
162  while ((iss >> what)) {
163  if (what == "priority") setPriorityLimit(iss, log);
164  else if (what == "propagate") log.setPropagate(true);
165  else if (what == "writer") addWriter(iss, log);
166  else {
167  InvalidCommand ex(what);
168  throw ex;
169  }
170  }
171  }
172 }

183 Configurator::readQuotedString(std::istream& is) const
184 {
185  char c, delim;
186  std::string s("");
187 
188  // Skip any leading whitespace
189  //
190  while (is.get(delim) && isspace(delim))
191  ;
192 
193  if (! is) return s;
194 
195  // If not a single or double quote, keep reading until we reach a space or end of line.
196  //
197  if (delim != '"' && delim != '\'') {
198  s += delim;
199  delim = ' ';
200  }
201 
202  while (is.get(c) && c != '\n' && c != delim)
203  s += c;
204 
205  // If we had a non-space delimiter, make sure we read it in.
206  //
207  if (delim != ' ' && c != delim)
208  throw UnterminatedString(currentLine_);
209 
210  return s;
211 }
