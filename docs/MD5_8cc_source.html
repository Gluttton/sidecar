108 MD5::Private::add(const uint8_t* input, size_t count)
109 {
110  // Update the counter values.
111  //
112  uint32_t index = (count_[0] >> 3) & 0x3F;
113  if ((count_[0] += (count << 3)) < (count << 3)) {
114  ++count_[1];
115  }
116  count_[1] += count >> 29;
117 
118  uint32_t partLen = kBufferSize - index;
119  uint32_t offset = 0;
120  if (count >= partLen) {
121  copy(index, input, partLen);
122  transform(buffer_);
123  for (offset = partLen; offset + 63 < count; offset += kBufferSize) {
124  transform(input + offset);
125  }
126  index = 0;
127  }
128 
129  copy(index, input + offset, count - offset);
130 }

137  // Calculate the digest value.
138  //
139  uint8_t bits[8];
140  encode(bits, count_, 8);
141  uint32_t index = (count_[0] >> 3) & 0x3f;
142  uint32_t padLen = (index < 56) ? (56 - index) : (120 - index);
143  add(kPadding_, padLen);
144  add(bits, 8);
