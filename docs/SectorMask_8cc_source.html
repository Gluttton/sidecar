91  // the new message does not contain any data.
92  //
93  Messages::BinaryVideo::Ref out(
94  Messages::BinaryVideo::Make("SectorMask::processInput", msg));
95 
96  out->getData() = msg->getData();
97 
98  double azimuth = 360.0 * float(msg->getShaftEncoding()) /
99  float(Messages::RadarConfig::GetShaftEncodingMax() + 1);
100 
101  int startRngBin = minRangeBin_->getValue();
102  int endRngBin = maxRangeBin_->getValue();
103  size_t msg_size = msg->size();
104 
105  // Determine if this pulse falls within the targeted sector's azimuth boundaries, if not, just pass
106  // message on.
107  //
108  if(azimuth >= minAzimuth_->getValue() && azimuth <= maxAzimuth_->getValue()) {
109 
110  if(!NormalizeBins(startRngBin, endRngBin, msg_size))
111  return false;
112 
113  // Mask out any values in the given sector
114  //
115  for(size_t i = startRngBin; i <= endRngBin; i++) {
116  out[i] = false;
117  }
118  }

120  // Send out on the default output device, and return the result to our Controller. NOTE: for multichannel
121  // output, one must give a channel index to the send() method. Use getOutputChannelIndex() to obtain the
122  // index for an output channel with a given name.

125  bool rc = send(out);
