43  void operator()(ChannelBuffer* channel)
44  {
45  if (! channel->isEnabled()) return;
46 
47  valid_ = true;
48 
49  // Iterate through the buffer looking for the CPI boundary which is currently identified by checking
50  // the prfEncoding field in the PRI Message header. Currently, we look for M_(i).code !=
51  // M2_(i+1).code. If i != cpiSpan-1, then we have an invalid CPI (ie, it's missing messages).
52  //
53  ChannelBuffer::const_iterator itr = channel->begin();
54  Messages::PRIMessage::Ref last = channel->getFront();
55  size_t pos = 0; // location of start of next CPI
56  bool cpiBoundary = false;
57  ++itr; // move to next PRI message
58  ++pos;
59  for(; itr != channel->end() && pos <= cpiSpan_; ++itr, ++pos) {
60 
61  // Check for CPI boundary
62  //
63  cpiBoundary = (last->getRIUInfo().prfEncoding !=
64  (*itr)->getRIUInfo().prfEncoding);
65 
66  if (cpiBoundary) {
67 
68  // Found the boundary, ensure it is located at the proper index which is equal to the
69  // cpiSpan. This will implicitly check that no messages were dropped between the first PRI
70  // message of a CPI and the last one.
71  //
72  if (pos != cpiSpan_) {
73 
74  // Need to prune off incomplete CPI by pruning to the sequence number of the start of
75  // the next CPI
76  //
77  if (! channel->pruneToSequenceCounter(
78  (*itr)->getSequenceCounter()))
79  ok_ = false;
80 
81  } // end if (pos != cpiSpan_)
82  } // end if (delta > thresh)
83  } // end for loop
84  } // end void operator()
85 
86  operator bool() const { return valid_ && ok_; }
