141 void
142 PlayerWindow::imageScalerDone(const QImage& image, int index)
143 {
144  // If the given index is larger than the current rescaling index, just forget it since it is a result from a
145  // stale rescaling.
146  //
147  if (index > rescalingIndex_)
148  return;
149 
150  // Be careful: the ImageScaler thread may emit a scaled image for an entry that no longer exists.
151  //
152  if (index >= scaled_.size()) {
153  rescalingAll_ = false;
154  return;
155  }
156 
157  // Save new scaled image, and update the view if user was looking at the old image.
158  //
159  scaled_[index] = image;
160  int viewingIndex = positioner_->getPosition();
161  if (viewingIndex < 0)
162  viewingIndex = 0;
163  if (viewingIndex == index)
164  frame_->setImage(image);
165 
166  // Scale more images until done.
167  //
168  if (rescalingAll_) {
169  ++index;
170  if (index < scaled_.size()) {
171  rescaleImage(index, true);
172  }
173  else {
174  rescalingIndex_ = 0;
175  rescalingAll_ = false;
176  }
177  }
178 }

185 
186  if (frameCount < scaled_.size()) {
187  do {

195  rescaleImage(index, false);

197 
198  if (frameCount <= positioner_->getMaximum()) {
199  positioner_->setMaximum(frameCount - 1);

415 void
416 PlayerWindow::updateMaxSize()
417 {
418  Logger::ProcLog log("updateMaxSize", Log());
419  LOGINFO << std::endl;
420 
421  needUpdateMaxSize_ = false;
422 
423  // Calculate the delta between the scroll area viewport and the image frame.
424  //
425  QSize delta(frame_->size());
426  LOGDEBUG << "frame_ size: " << delta << std::endl;
427  delta -= scrollArea_->maximumViewportSize();
428  LOGDEBUG << "delta: " << delta << std::endl;
429  QSize extra(scrollArea_->verticalScrollBar()->sizeHint().width(),
430  scrollArea_->horizontalScrollBar()->sizeHint().height());
431  delta += extra;
432  LOGDEBUG << "delta + sbars: " << delta << std::endl;
433 
434  // Adjust our maximum window size so that we don't grow beyond the point of showing everything in the
435  // viewport.
436  //
437  QSize min(minimumSizeHint());
438  QSize max(size());
439  LOGDEBUG << "current size: " << max << std::endl;
440  max.setWidth(std::max(max.width() + delta.width(), min.width()));
441  max.setHeight(std::max(max.height() + delta.height(), min.height()));
442  LOGDEBUG << "max size: " << max << std::endl;
443  setMaximumSize(max);
444  LOGDEBUG << "new size: " << size() << std::endl;
445 
446  // If expanding, grow our window as well.
447  //
448  if (delta.width() > 0 || delta.height() > 0) {
449 
450  // To be nice to users, only grow our window if the resulting growth is still on the desktop.
451  //
452  QDesktopWidget* desktop = QApplication::desktop();
453  QRect available = desktop->availableGeometry();
454  int right = frameGeometry().right();
455  if (right + delta.width() > available.right())
456  delta.setWidth(std::max(0, available.right() - right));
457 
458  int bottom = frameGeometry().bottom();
459  if (bottom + delta.height() > available.bottom())
460  delta.setHeight(std::max(0, available.bottom() - bottom));
461 
462  if (delta.width() > 0 || delta.height() > 0) {
463  QSize newSize(size());
464  newSize += delta;
465  LOGDEBUG << "resize: " << newSize << std::endl;
466  resize(newSize);
467  LOGDEBUG << "new size: " << size() << std::endl;
468  }
469  }
470 }
