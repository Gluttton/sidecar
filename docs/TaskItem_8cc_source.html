170 void
171 TaskItem::formatChangedParameters(const XmlRpc::XmlRpcValue& definitions,
172  QStringList& changes) const
173 {
174  // Create a list of strings that detail the paramter changes found in the given XML array.
175  //
176  QString heading;
177  for (int index = 0; index < definitions.size(); ++index) {
178 
179  // Fetch the XML struct that defines the parameter that has changed
180  //
181  const XmlRpc::XmlRpcValue& definition(definitions[index]);
182  QString value, original;
183 
184  // Format according to parameter type.
185  //
186  std::string type(definition["type"]);
187  if (type == "int") {
188  value = QString::number(int(definition["value"]));
189  original = QString::number(int(definition["original"]));
190  }
191  else if (type == "double") {
192  value = QString::number(double(definition["value"]));
193  original = QString::number(double(definition["original"]));
194  }
195  else if (type == "bool") {
196  value = bool(definition["value"]) ? "TRUE" : "FALSE";
197  original = bool(definition["original"]) ? "TRUE" : "FALSE";
198  }
199  else if (type == "enum") {
200 
201  // Convert enumerated values into text strings. Make sure that the enumeration values are valid.
202  //
203  const XmlRpc::XmlRpcValue::ValueArray& names(
204  definition["enumNames"]);
205  int min = int(definition["min"]);
206 
207  // Get string of the current value
208  //
209  int tmp = int(definition["value"]) - min;
210  if (tmp >= 0 && tmp < names.size())
211  value = QString("'%1'").arg(
212  QString::fromStdString(names[tmp]));
213 
214  // Get string of the original value
215  //
216  tmp = int(definition["original"]) - min;
217  if (tmp >= 0 && tmp < names.size())
218  original = QString("'%1'").arg(
219  QString::fromStdString(names[tmp]));
220  }
221  else if (type == "notification") {
222  continue;
223  }
224  else { // string, readPath, and writePath
225  value = QString("'%1'").arg(
226  QString::fromStdString(definition["value"]));
227  original = QString("'%1'").arg(
228  QString::fromStdString(definition["original"]));
229  }
230 
231  if (heading.isNull()) {
232  heading = getParameterChangedHeading();
233  changes.append(heading);
234  }
235 
236  // Format an entry for logging
237  //
238  changes.append(QString(" Param '%1' = %2 [original is %3]\n")
239  .arg(QString::fromStdString(definition["name"]))
240  .arg(value)
241  .arg(original));
242  }
243 }
