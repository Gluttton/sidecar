112 bool
113 DynamicThreshold::processChannels()
114 {
115  static Logger::ProcLog log("processChannels", getLog());
116  LOGINFO << std::endl;
117 
118  Video::Ref samples(samples_->popFront());
119  Video::Ref thresholds(thresholds_->popFront());
120  if (thresholds->size() < samples->size()) {
121  thresholds->resize(samples->size(), 0);
122  }
123 
124  // Create our output message. Use std::transform to visit the sample and threshold values, calling the appropriate
125  // functor for each and adding the binary result to the output message.
126  //
127  BinaryVideo::Ref out(BinaryVideo::Make("DynamicThreshold", samples));
128  out->resize(samples->size());
129 
130  size_t passed = 0;
131  switch (operator_->getValue()) {
132  case kLessThan:
133  std::transform(samples->begin(), samples->end(), thresholds->begin(),
134  out->begin(), LessThan(passed));
135  break;
136  case kLessThanEqualTo:
137  std::transform(samples->begin(), samples->end(), thresholds->begin(),
138  out->begin(), LessThanEqualTo(passed));
139  break;
140  case kEqualTo:
141  std::transform(samples->begin(), samples->end(), thresholds->begin(),
142  out->begin(), EqualTo(passed));
143  break;
144  case kGreaterThanEqualTo:
145  std::transform(samples->begin(), samples->end(), thresholds->begin(),
146  out->begin(), GreaterThanEqualTo(passed));
147  break;
148  case kGreaterThan:
149  std::transform(samples->begin(), samples->end(), thresholds->begin(),
150  out->begin(), GreaterThan(passed));
151  break;
152  default:
153  LOGERROR << "invalid operation: " << operator_->getValue()
154  << std::endl;
155  return false;
156  }
157 
158  // Update our stats based on how many samples passed their threshold.
159  //
160  passPercentage_.addValue(double(passed) / out->size());
161  bool rc = send(out);
162 
163  LOGDEBUG << "rc: " << rc << std::endl;
164  return rc;
165 }
