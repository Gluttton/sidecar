117 ReaderThread::run()
118 {
119  Logger::ProcLog log("run", Log());
120  LOGINFO << this << std::endl;
121 
122  // Create a new MessageReader object for the thread to use to read in data from the publisher described by
123  // our ServiceEntry object.
124  //
125  boost::scoped_ptr<MessageReader> reader(
126  MessageReader::Make(serviceEntry_.get()));
127 
128  // The reader belongs to our thread, so we can connect to it without queuing signals. Also, we must delete
129  // it before we and our thread exits.
130  //
131  connect(reader.get(), SIGNAL(connected()), SIGNAL(connected()));
132  connect(reader.get(), SIGNAL(received(const Messages::Header::Ref&)),
133  SLOT(received(const Messages::Header::Ref&)));
134  connect(reader.get(), SIGNAL(disconnected()), SIGNAL(disconnected()));
135 
136  // Run this thread's event loop so we can respond to signals from our MessageReader.
137  //
138  exec();
139 
140  // Don't issue any signals just because we are deleting the MessageReader objet; we are already quitting
141  // anyway.
142  //
143  reader->disconnect(this);
144  reader.reset();
145 
146  LOGINFO << this << " reader thread exiting" << std::endl;
