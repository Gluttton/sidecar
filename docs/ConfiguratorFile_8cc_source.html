44 ConfiguratorFile::reload()
45 {
46  // Fetch the current mod time for the config file. We will adopt this value when we have finished loading.
47  //
48  time_t lm = getModificationTime();
49  std::ifstream fs(path_.c_str());
50  if (! fs) throw CannotOpenConfigFile(path_);
51 
52  // Have all Log objects we created last time dump their Writer objects. Thus configuration changes will
53  // affect how Log objects write out log messages.
54  //
55  std::for_each(managed().begin(), managed().end(), PurgeLog());
56  load(fs);
57 
58  // Finally, update the modification time. This will affect the results of future isStale() method calls.
59  //
60  lastModification_ = lm;
61 }
