226 int
227 StatusEmitterBase::Emitter::svc()
228 {
229  Logger::ProcLog log("svc", Log());
230  ACE_Message_Block* data = nullptr;
231  Destination* dest = nullptr;
232 
233  // Fetch data blocks from our processing queue. Keep going until our queue returns an error because it has
234  // been deactivated.
235  //
236  while (getq(data) != -1) {
237 
238  switch(data->msg_type()) {
239  case ACE_Message_Block::MB_DATA:
240 
241  // Post status update to the all Destination objects.
242  //
243  if (current_) current_->release();
244  current_ = data;
245 
246  // Emit the current status to all active destinations.
247  //
248  std::for_each(destinations_.begin(), destinations_.end(), [this](auto d){sendTo(d);});
249 
250  // NOTE: don't call data->release() since we are caching it in current_.
251  //
252  break;
253 
254  case ACE_Message_Block::MB_START:
255 
256  // New Destination object to add. First make sure that it points to a unique location.
257  //
258  dest = reinterpret_cast<Destination*>(data->base());
259  for (size_t index = 0; index < destinations_.size(); ++index) {
260  if (destinations_[index]->serviceEntry_->getName() ==
261  dest->serviceEntry_->getName()) {
262 
263  // Found existing entry -- ignore this new one.
264  //
265  delete dest;
266  dest = 0;
267  break;
268  }
269  }
270 
271  if (dest) {
272  destinations_.push_back(dest);
273  if (current_) sendTo(dest);
274  }
275 
276  data->release();
277  break;
278 
279  case ACE_Message_Block::MB_STOP:
280 
281  // Destination object to remove. Scan the Destination vector for the entry to remove and then remove
282  // it.
283  //
284  dest = reinterpret_cast<Destination*>(data->base());
285  for (size_t index = 0; index < destinations_.size(); ++index) {
286  if (destinations_[index] == dest) {
287  destinations_.erase(destinations_.begin() + index);
288  delete dest;
289  break;
290  }
291  }
292  data->release();
293  break;
294 
295  default:
296  data->release();
297  break;
298  }
299 
300  data = nullptr;
301  }
302 
303  return 0;
304 }

308 {
309  static Logger::ProcLog log("sendTo", Log());
310  Zeroconf::ServiceEntry::Ref serviceEntry = destination->serviceEntry_;
311 
312  LOGINFO << serviceEntry->getName() << " port: " << destination->address_.get_port_number() << ' '
313  << destination->address_.get_host_name() << std::endl;
314 
315  // If we've received more than 10 errors in a row, stop sending to this destination.
316  //
317  if (destination->failures_ > 10) return;
318 
319  // Attempt to send to the StatusCollector found at the Destination object.
320  //
321  ssize_t rc = socket_.send(current_->rd_ptr(), current_->size() - 1, destination->address_, 0);
322  if (rc != ssize_t(current_->size() - 1)) {
323  ++destination->failures_;
324  LOGERROR << serviceEntry->getName() << " failed send to "
325  << Utils::INETAddrToString(destination->address_) << " - buffer size: " << current_->size()
326  << " - " << Utils::showErrno() << std::endl;
327  }
328  else {
329  destination->failures_ = 0;

354  const Zeroconf::ResolvedEntry& resolvedEntry = service->getResolvedEntry();
355  LOGERROR << "fullname: " << resolvedEntry.getFullName() << " host: " << resolvedEntry.getHost() << " port: "
356  << resolvedEntry.getPort() << " interface: " << service->getInterface() << std::endl;

359  LOGDEBUG << "host: " << host << std::endl;

377  data->release();
378  }

407 StatusEmitterBase::Emitter::lostSlot(const ServiceEntryVector& services)
408 {
409  static Logger::ProcLog log("lostSlot", Log());
410  LOGINFO << services.size() << std::endl;
411 
412  // Locate the Destination object we will remove, but let the emitter thread do the removal.
413  //
414  for (size_t index = 0; index < services.size(); ++index) {
415  Zeroconf::ServiceEntry::Ref service(services[index]);
416  size_t serviceIndex = findDestinationIndex(service);
417  if (serviceIndex == destinations_.size()) {
418  LOGWARNING << "registration for " << service->getName() << " not found" << std::endl;
419  continue;
420  }
421 
422  Destination* dest = destinations_[serviceIndex];
423 
424  // Encapsulate the Destination object to remove in an ACE_Message_Block and post to the emitter thread.
425  //
426  ACE_Message_Block* data = new ACE_Message_Block(sizeof(dest), ACE_Message_Block::MB_STOP, 0,
427  reinterpret_cast<char*>(dest));
428  if (putq(data) == -1) {
429  LOGERROR << "failed to enqueue remove destination" << std::endl;
430  data->release();
431  }
432 
433  if (destinationCount_) {
434  --destinationCount_;
435  if (! destinationCount_)
436  stopTimer();
437  }
438  }
