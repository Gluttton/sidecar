152 History::processVideo(const MessageList& data)
153 {
154  static Logger::ProcLog log("processVideo", Log());
155  LOGINFO << "data.size: " << data.size() << std::endl;
156 
157  Messages::Video::Ref info;
158  AlphaIndices alphaIndices;
159 
160  int betaScans = radarSettings_->getBetaScans();
161  int firstSample = radarSettings_->getFirstSample();
162  int lastSample = radarSettings_->getLastSample();
163 
164  for (size_t index = 0; index < data.size(); ++index) {
165 
166  Messages::Video::Ref msg(
167  boost::dynamic_pointer_cast<Messages::Video>(data[index]));
168  LOGDEBUG << msg->getSequenceCounter() << std::endl;
169 
170  // !!! HACK !!!
171  //
172  if (! allynHack_->getValue() && msg->getRIUInfo().prfEncoding == 0)
173  continue;
174 
175  info = msg;
176 
177  // Use the latest range information.
178  //
179  radarSettings_->setRangeScaling(msg->getRangeMin(),
180  msg->getRangeFactor());
181 
182  int alphaIndex = radarSettings_->getAlphaIndex(msg);
183  if (alphaIndex == -1) {
184  LOGERROR << "invalid alpha value: "
185  << radarSettings_->getAlpha(msg)
186  << " - skipping message " << msg->getSequenceCounter()
187  << std::endl;
188  continue;
189  }
190 
191  // Detect when we've reached the end of the scan in alpha space.
192  //
193  if (alphaIndex < lastAlphaIndex_) {
194  LOGINFO << "seq: " << msg->getSequenceCounter()
195  << " lastAlphaIndex: " << lastAlphaIndex_ << " alphaIndex: "
196  << alphaIndex << std::endl;
197  ++scanCounter_;
198 
199  // If we have any changes, go ahead and emit them now. NOTE: this behavior is required by the
200  // AlphaBetaWidget and AlphaRangeWidget classes so that they can transfer large streams of color
201  // values to the graphics card in one shot.
202  //
203  if (! alphaIndices.empty()) {
204  emit alphasChanged(alphaIndices);
205  alphaIndices.clear();
206  }
207  }
208 
209  lastAlphaIndex_ = alphaIndex;
210 
211  // Add the alpha index if not already present.
212  //
213  if (alphaIndices.empty() || alphaIndices.back() != alphaIndex)
214  alphaIndices.push_back(alphaIndex);
215 
216  int betaIndex = radarSettings_->getBetaIndex(msg);
217  if (betaIndex == -1) {
218  LOGERROR << "invalid beta value: "
219  << radarSettings_->getBeta(msg)
220  << " - skipping message " << msg->getSequenceCounter()
221  << std::endl;
222  continue;
223  }
224 
225  // Find the largest value in the sample range
226  //
227  size_t alphaBetaIndex = alphaIndex * betaScans + betaIndex;
228  Messages::Video::const_iterator pos = msg->begin();
229  Messages::Video::const_iterator end = msg->end();
230  Messages::Video::const_iterator first = pos + firstSample;
231  Messages::Video::const_iterator last = pos + lastSample;
232  int value = DataContainer::GetMinValue();
233  if (first < end) {
234  if (last >= end)
235  last = end;
236  value = *std::max_element(first, last);
237  }
238 
239  // Record the found value if larger than the previous or if we are in a new scan.
240  //
241  if (value > alphaBetaData_[alphaBetaIndex].value ||
242  scanCounter_ != alphaBetaData_[alphaBetaIndex].scan) {
243  alphaBetaData_[alphaBetaIndex].value = value;
244  alphaBetaData_[alphaBetaIndex].scan = scanCounter_;
245  }
246 
247  // Update the data for the AlphaRangeWidget.
248  //
249  alphaRangeData_[alphaIndex].update(msg, radarSettings_,
250  scanCounter_);
251  }
252 
253  // Notify the views of changes.
254  //
255  if (! alphaIndices.empty())
256  emit alphasChanged(alphaIndices);
257 
258  // Notify others of the last message processed.
259  //
260  if (info)
261  emit currentMessage(info);
262 
263  pruneTargets();
264 }

303 void
304 History::addRangeTruth(const Messages::TSPI::Ref& msg)
305 {
306  QString tag = QString::fromStdString(msg->getTag());
307 
308  TargetPlotListList::iterator pos = rangeTruths_.begin();
309  TargetPlotListList::iterator end = rangeTruths_.end();
310 
311  while (pos != end) {
312  TargetPlotList& plots(*pos);
313  if (plots.front().getTag() == tag) {
314  if (msg->isDropping()) {
315  rangeTruths_.erase(pos);
316  return;
317  }
318 
319  plots.push_front(TargetPlot(*msg));
320  while (plots.size() > rangeTruthsMaxTrailLength_) {
321  plots.pop_back();
322  }
323 
324  return;
325  }
326  ++pos;
327  }
328 
329  if (! msg->isDropping()) {
330  TargetPlotList plots;
331  plots.push_back(TargetPlot(*msg));
332  rangeTruths_.push_back(plots);
333  }
