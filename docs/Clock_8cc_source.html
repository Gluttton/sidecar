28 
29  if (! ticker_.isActive()) {
30  LOGDEBUG << playbackClockStart_ << std::endl;
31  return playbackClockStart_;
32  }
33 
34  // Calculate the number of seconds that have elapsed in real time.
35  //
36  Time::TimeStamp elapsed(Time::TimeStamp::Now());
37  elapsed -= wallClockStart_;
38 
39  // Apply a rate to the elapsed real time to make playback time pass faster or slower.
40  //
41  if (wallClockRate_ != 1.0)
42  elapsed *= wallClockRate_;
43 
44  // Add any past elapsed time to the time when playback started to get the current playback clock time.
45  //

53 Clock::getWallClockDurationUntil(const Time::TimeStamp& when)
54 {
55  static Logger::ProcLog log("getWallClockDurationUntil", Log());
56  LOGINFO << "when: " << when << std::endl;
57 
58  // If looping, add an offset to keep event times increasing.
59  //
60  Time::TimeStamp delta(when);
61 
62  // Obtain the playback clock time, and calculate the difference between the two. If the result is negative,
63  // then the event has already happend according to the playback clock.
64  //
65  delta -= getPlaybackClock();
66  LOGDEBUG << "delta: " << delta << std::endl;
67 
68  // We have the duration in playback time frame. Multiply by the playback rate to get the number of seconds
69  // in the real time frame.
70  //
71  if (playbackClockRate_ != 1.0)
72  delta *= playbackClockRate_;
73 
74  LOGDEBUG << "duration: " << delta << std::endl;

89 Clock::setWallClockRate(double wallClockRate)
90 {
91  static Logger::ProcLog log("setWallClockRate", Log());
92  LOGINFO << "wallClockRate: " << wallClockRate << std::endl;
93 
94  if (wallClockRate <= 0.0) {
95  LOGERROR << "invalid wallClockRate: " << wallClockRate << std::endl;
96  return;
97  }
98 
99  // Since the rate is changing, we want to 'freeze' any past elapsed time at the old rate.
100  //
101  playbackClockStart_ = getPlaybackClock();
102 
103  // Reset the real time elapsed counter since our frame of reference has changed.
104  //
105  wallClockStart_ = Time::TimeStamp::Now();
106 
107  // Record the new rate.
108  //
109  wallClockRate_ = wallClockRate;
110  playbackClockRate_ = 1.0 / wallClockRate;
111 }
