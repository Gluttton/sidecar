162 Extract::process(const Messages::BinaryVideo::Ref& msg)
163 {

166  // Make sure our declared vector is at least as big as the message vector since we will index into both at
167  // the same time.
168  //
169  if (msg->size() > gateTargets_.size())
170  gateTargets_.resize(msg->size(), Target::Ref());
171 
172  // Step through all PRI range gates looking for OFF/ON and ON/OFF transitions to determine blobs. We know
173  // the transition type based on the value of the target local variable.
174  //
175  Target::Ref target;
176  const Messages::BinaryVideo::Container& data(msg->getData());
177  for (size_t gate = 0; gate < data.size(); ++gate) {
178 
179  // See if there is a blob to detect
180  //
181  if (data[gate]) {
182 
183  // Is this the start of a new blob?
184  //
185  if (! target) {
186 
187  // Yes. If there was a target declaration from processing the last PRI, use it for this blob.
188  // Otherwise, create a new target.
189  //
190  if (gateTargets_[gate]) {
191  target = gateTargets_[gate];
192  }
193  else {
194  target = Target::Make(msg->getIRIGTime(), gate,
195  msg->getAzimuthStart());
196  pending_.push_back(target);
197  }
198  }
199  else {
200 
201  // No, in the middle of a blob. Check if a diffferent target declaration from a previous PRI
202  // exists. If so, use it and delete the current one.
203  //
204  if (gateTargets_[gate] && gateTargets_[gate] != target) {
205  TargetList::iterator pos =
206  std::find(pending_.begin(), pending_.end(), target);
207  if (pos != pending_.end())
208  pending_.erase(pos);
209  gateTargets_[gate]->assimilate(target);
210  target = gateTargets_[gate];
211  }
212  }
213  }
214  else if (target) {
215 
216  // A blob has just ended, so we need to set its max gate. By now the blob should be associated with
217  // one and only one target
218  //
219  target->update(msg->getIRIGTime(), gate - 1,
220  msg->getAzimuthStart());
221  target.reset();
222  }
223  }
224 
225  // Clear the gate target assignments. We will put them back in if necessary in the loop below, and we will
226  // remember them for the next PRI message.
227  //
228  std::fill(gateTargets_.begin(), gateTargets_.end(), Target::Ref());
229 
230  // Visit each of the pending targets to see if it should be emitted as an extraction.
231  //
232  Extractions::Ref extractions;
233  TargetList::iterator it(pending_.begin());
234  TargetList::iterator pendingEnd(pending_.end());
235  while (it != pendingEnd) {
236 
237  // Get the target at the current iterator position, but do not move the iterator yet since we may delete
238  // the position and we don't want to invalidate the iterator if we do.
239  //
240  target = *it;
241 
242  // If the target was touched in the above gate processsing loop, reset the valid flag and assign the
243  // target to the gates vector.
244  //
245  if (target->testValidAndReset()) {
246  ++it;
247  for (int gate = target->getGateMin(); gate <= target->getGateMax();
248  ++gate)
249  if (data[gate]) gateTargets_[gate] = target;
250  }
251  else {
252 
253  // The target was not used in the above gate processsing loop, so we now know its final azimuth
254  // value. Calculate the range/azimuth for an extraction entry.
255  //
256  double irig = target->getIRIGTime();
257  float range = msg->getRangeAt(target->getGate());
258  float azimuth = target->getAzimuth();
259 
260  LOGINFO << "IRIG: " << irig
261  << " range: " << range
262  << " azimuth: " << Utils::radiansToDegrees(azimuth)
263  << std::endl;
264 
265  if (! extractions)
266  extractions = Extractions::Make("Extract", msg);
267 
268  extractions->push_back(Extraction(irig, range, azimuth, 0.0));
269 
270  // Finished with this target.
271  //
272  it = pending_.erase(it);
273  pendingEnd = pending_.end();
274  }
275  }
276 
277  // Processing finished. Send off an extractions message if we did any extractions.
278  //
279  bool rc = true;
280  if (extractions)
281  rc = send(extractions);
282 
283  LOGDEBUG << "rc: " << rc << std::endl;
284  return rc;
285 }
