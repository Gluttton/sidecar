199 {
200  // Determine which bin this extraction belongs to
201  // (add +1) to shift the logical data buffer within the larger buffer

203  int binY=int(floor(ext.getY()/searchRadius))+indexOffset + 1;
204 
205  ext.setCorrelated(FALSE);
206 
207  Data d={time, ext};
208 
209 #ifdef SCAN_CORR_DEBUG
210  cerr
211  << "Correlating extraction ("
212  << ext.getX() << ", "
213  << ext.getY() << ") into ["
214  << binX << ", "
215  << binY << "] numBins=" << numBins << endl;
216 #endif
217 
218 
219  if (((binX-1) < 0) || ((binX+1) >= numBins) || ((binY-1) < 0) || ((binY+1) >= numBins)){
220 #ifdef SCAN_CORR_DEBUG
221  cerr << "Bad bin #, bailing" << endl;
222 #endif
223 
224  return;
225 
226  }
227 
228  // Perform the correlations
229  // binX and binY are never = 0 or numBins - 1, so this code holds
230  // for all possible cases
231  corrCell(d, buffer[binX-1][binY-1]);
232  corrCell(d, buffer[binX-1][binY]);
233  corrCell(d, buffer[binX-1][binY+1]);
234  corrCell(d, buffer[binX][binY-1]);
235  corrCell(d, buffer[binX+1][binY-1]);
236  corrCell(d, buffer[binX][binY]);
237  corrCell(d, buffer[binX][binY+1]);
238  corrCell(d, buffer[binX+1][binY]);
239  corrCell(d, buffer[binX+1][binY+1]);
240 
241  if (!d.extraction.getCorrelated())
242  d.extraction.setNumCorrelations(0);
243 
244 
245  // Add to buffer
246  buffer[binX][binY].push_back(d);
247
