60 bool
61 DownConverter::process()
62 {
63 
64  static Logger::ProcLog log("process", getLog());
65 
66  LOGINFO << std::endl;
67 
68  if (! enabled_->getValue()) {
69  if (rxChannel_->isEmpty())
70  return true;
71  uint32_t rxSeq = rxChannel_->getNextSequence();
72  cohoChannel_->prune(rxSeq);
73  return send(rxChannel_->getData());
74  }
75 
76  if (rxChannel_->isEmpty() || cohoChannel_->isEmpty()) {
77  LOGDEBUG << "empty stream" << std::endl;
78  return true;
79  }
80 
81  // Locate the next common message sequence in the two input queues
82  //
83  uint32_t rxSeq = rxChannel_->getNextSequence();
84  uint32_t cohoSeq = cohoChannel_->getNextSequence();
85 
86  if (cohoChannel_->prune(rxSeq) || rxChannel_->prune(cohoSeq) ||
87  rxChannel_->getNextSequence() != cohoChannel_->getNextSequence()) {
88  LOGDEBUG << "empty stream" << std::endl;
89  return true;
90  }
91 
92  // We have a common message in the two channels so we can generate output
93  //
94  Messages::Video::Ref cohoMsg(cohoChannel_->getData());
95  Messages::Video::Ref rxMsg(rxChannel_->getData());
96 
97  cohoChannel_->popData();
98  rxChannel_->popData();
99 
100  Messages::Video::Ref outMsg(Messages::Video::Make(getName(), rxMsg));
101  Messages::Video::Container& out(outMsg->getData());
102 
103  int num_samples = cohoMsg->size() / 2;
104 
105  VsipComplexVector cohoVec(num_samples);
106  VsipComplexVector rxVec(num_samples);
107 
108 
109  // populate the vectors
110  for(int i = 0; i < num_samples; i++) {
111  rxVec.put(i, ComplexType(rxMsg[2*i], rxMsg[2*i+1]));
112  cohoVec.put(i, ComplexType(cohoMsg[2*i], cohoMsg[2*i+1]));
113  }
114 
115  // apply a low pass filter
116  float alpha = alpha_->getValue();
117  for(int i = 1; i < num_samples; i++) {
118  rxVec(i) = rxVec(i - 1) + alpha*(rxVec(i) - rxVec(i - 1));
119  }
120 
121  // compute the average magnitude in the coho vector
122  float avg_mag = vsip::sqrt(vsip::meansqval(cohoVec));
123 
124 
125  // perform down conversion
126  rxVec = rxVec*(conj(cohoVec) / avg_mag);
127 
128  // push the data back out
129  for(int i = 0; i < num_samples; i++) {
130  out.push_back(Messages::Video::DatumType(::rint(rxVec(i).real())));
131  out.push_back(Messages::Video::DatumType(::rint(rxVec(i).imag())));
132  }
133 
134  return send(outMsg);
135 }
