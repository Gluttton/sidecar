43 void
44 OrderedSlidingWindow::insertAndRemove(Video::DatumType insert, Video::DatumType remove)
45 {
46  if (insert == remove) {
47  return;
48  }
49 
50  // Locate the position of the item to remove.
51  //
52  Video::Container::iterator insertPos;
53  Video::Container::iterator removePos;
54  if (insert < remove) {
55 
56  // Locate the lowest position in the ordered vector where we should find the old item.
57  //
58  removePos = std::lower_bound(sorted_.begin(), sorted_.end(), remove);
59 
60  // Locate the highest position in the ordered vector where we could insert the new item and keep the
61  // vector ordered.
62  //
63  insertPos = std::upper_bound(sorted_.begin(), removePos, insert);
64 
65  // If the two positions are the same, just replace.
66  //
67  if (insertPos == removePos) {
68  *insertPos = insert;
69  }
70  else {
71 
72  // Shift all elements from [insertPos, removePos) up one, thus effectively deleting the entry at
73  // removePos, and allowing us to store into insertPos while keeping the elements ordered.
74  //
75  std::copy_backward(insertPos, removePos, removePos + 1);
76  *insertPos = insert;
77  }
78  }
79  else { // insert > remove
80 
81  // Locate the highest position in the ordered vector where we should find the old item.
82  //
83  removePos = std::upper_bound(sorted_.begin(), sorted_.end(), remove) - 1;
84 
85  // Locate the lowest position in the ordered vector where we could insert the new item and keep the
86  // vector ordered.
87  //
88  insertPos = std::lower_bound(removePos, sorted_.end(), insert);
89 
90  // If the two positions refer to the same slot, just replace.
91  //
92  if (insertPos == (removePos + 1)) {
93  *removePos = insert;
94  }
95  else {
96 
97  // Shift all elements from [removePos + 1, insertPos) down one, thus effectively deleting the entry
98  // at removePos, and allowing us to store into insertPos while keeping the elements ordered.
99  //
100  std::copy(removePos + 1, insertPos, removePos);
101  --insertPos;
102  *insertPos = insert;
103  }
104  }
105 }

127 bool
128 OSCFAR::process(const Video::Ref& in)
129 {
130  static Logger::ProcLog log("process", getLog());
131  double alpha = alpha_->getValue();
132  LOGINFO << "pri size: " << in->size() << " alpha " << alpha << std::endl;
133 
134  // Nothing to do if the input data size is smaller than our window size.
135  //
136  size_t windowSize = windowSize_->getValue();
137  if (windowSize > in->size()) {
138  LOGERROR << "windowSize > in->size()" << std::endl;
139  return false;
140  }
141 
142  // Verify that the threshold index we use is valid in the sliding window.
143  //
144  size_t percentile = thresholdIndex_->getValue();
145  double indexIntoWindow = (percentile / 100.0) * windowSize;
146  size_t thresholdIndex = static_cast<size_t>(std::floor(indexIntoWindow));
147 
148  LOGINFO << "thresholdIndex " << thresholdIndex << std::endl;
149  if (thresholdIndex >= windowSize) {
150  getController().setError("Threshold index >= window size");
151  return false;
152  }
153 
154  size_t halfWindowSize = windowSize / 2;
155 
156  // Create our sliding window. It should remain sorted as we slide over the input data.
157  //
158  OrderedSlidingWindow slidingWindow(getLog(), windowSize, in->getData());
159 
160  // Calculate the number of times we will apply the window. This unsigned arithmetic is safe to do because
161  // above we guarantee that windowSize_ <= in->size().
162  //
163  size_t limit = in->size() - windowSize;
164 
165  BinaryVideo::Ref out(BinaryVideo::Make(getName(), in));
166  out->resize(in->size(), 0);
167 
168  // This was written as a for loop, but that is incorrect because in the last iteration the call to
169  // OrderedSlidingWindow::insertAndRemove() got called with an invalid index into the input message. Instead,
170  // do the termination check from inside the loop, before the call to insertAndRemove().
171  //
172  size_t index = 0;
173  while (1) {
174  size_t pos = index + halfWindowSize;
175  bool passed = in[pos] > (alpha * slidingWindow.getThreshold(thresholdIndex));
176  out[pos] = passed;
177 
178  // Check that we can safely continue and index into the input array.
179  //
180  if (index == limit) {
181  break;
182  }
183 
184  // Remove the element that will no longer be valid when the sliding window moves to the next sample, and
185  // add the new element that is now visible in the window.
186  //
187  slidingWindow.insertAndRemove(in[index + windowSize], in[index]);
188  ++index;
189  }
190 
191  // Finally emit the output message.
192  //
193  bool rc = send(out);
194  LOGDEBUG << "send: " << rc << std::endl;
195  return rc;
196 }
