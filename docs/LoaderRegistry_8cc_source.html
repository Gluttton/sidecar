66  // Quick check to see if we can keep the order with just an add to the end of the vector.
67  //
68  VersionedVoidLoader newEntry(version, loader);
69  if (loaders_.empty() || version > loaders_.back().version_) {
70  loaders_.push_back(newEntry);
71  latestLoader_ = loader;
72  currentVersion_ = version;
73  LOGINFO << "*** new ***" << std::endl;
74  dump();
75  return;
76  }
77 
78  // Search for the position in the set of existing loaders where an insertion would not change the ordering
79  // of the items. If an entry with the same version value already exists, the following will return an
80  // iterator that is one beyond the entry so we need to check the entry prior to the returned position to see
81  // if it has the same version indicator.
82  //
83  VersionedVoidLoaderVector::iterator pos(std::upper_bound(loaders_.begin(), loaders_.end(), newEntry));
84  if (pos != loaders_.begin() && (pos - 1)->version_ == version) {
85  Utils::Exception ex("loader already registered for version ");
86  ex << version << " for message type ";
87  log.thrower(ex);
88  }
89 
90  if (pos != loaders_.end()) {
91  loaders_.insert(pos, newEntry);
92  }
93  else {

95  latestLoader_ = loader;
96  currentVersion_ = version;
97  }
