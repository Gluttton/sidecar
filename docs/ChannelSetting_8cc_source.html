132

134  // it as we load the new service names. Otherwise, use current selection.
135  //
136  QString wanted;
137  if (! first_ || first_->currentIndex() < 1) {
138  wanted = getValue();
139  }
140  else {

142  }
143 
144  LOGDEBUG << "wanted: " << wanted << std::endl;
145 
146  // Get the found keys and sort then alphabetically.
147  //
148  QList<QString> keys = services.keys();
149  qSort(keys.begin(), keys.end());
150 
151  // Place an empty name at the beginning of the list to represent no channel selection or OFF.
152  //
153  keys.prepend("-OFF-");
154 
155  // Notify others of the new key set
156  //
157  emit availableChannels(keys);
158 
159  // Add to the widget, remembering which one matches the one we would like to be current.
160  //
161  if (first_) {
162 
163  int found = 0;
164  first_->clear();
165  for (int index = 0; index < keys.size(); ++index) {
166  LOGDEBUG << "key: " << keys[index] << std::endl;
167  if (keys[index] == wanted) {
168  found = index;
169  }
170  first_->addItem(keys[index]);
171  }
172 
173  LOGDEBUG << "found: " << found << " wanted: " << wanted << std::endl;
174 
175  // Locate the Zeroconf service entry that corresponds to the channel name.
176  //
177  ServiceEntry* serviceEntry =
178  found ? browser_->getServiceEntry(wanted) : 0;
179 
180  // Update our Subscriber if the ServiceEntry we wanted was lost or appeared.
181  //
182  if (activeServiceEntry_ != serviceEntry) {
183  subscriber_->useServiceEntry(serviceEntry);
184  activeServiceEntry_ = serviceEntry;
185  if (serviceEntry)

187  }
188

190  //

192  }
