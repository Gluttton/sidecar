142 ManyInAlgorithm::setInfoSlots(IO::StatusBase& status)
143 {
144  // !!! NOTE: this routine runs in a thread different from the one that runs the algorithm, so BE CAREFUL
145  // !!! with what and how you access state data.
146  //
147  status.setSlot(kEnabled, enabled_->getValue());
148 
149  // The value channels_.size() does not change during the lifetime of the algorithm -- it is set by the XML
150  // configuration for the algorithm, so we are safe to access the channels_ vector at any time from any
151  // thread.
152  //
153  XmlRpc::XmlRpcValue::ValueArray* v = new XmlRpc::XmlRpcValue::ValueArray;
154 
155  for (auto channel: channels_) {
156 
157  // Race condition here: channel->isEnabled() may become false while we fetched the following values,
158  // however the values we fetch don't depend on its value so I think we are safe.
159  //
160  if (channel->isEnabled()) {
161  v->push_back(int(channel->getChannelIndex()));
162  v->push_back(int(channel->size()));
163  }
164  }
165 
166  // Set the number of enabled channels we found above.
167  //
168  status.setSlot(kChannelStats, v);
