141 MetaTypeInfo::Registrations::add(int key, MetaTypeInfo* info)
142 {
143  static Logger::ProcLog log("add", Log());
144  LOGDEBUG << keys_.size() << ' ' << infos_.size() << std::endl;
145 
146  if (keys_.empty() || key > keys_.back()) {
147  LOGDEBUG << "adding to back" << std::endl;
148  keys_.push_back(key);
149  infos_.push_back(info);
150  }
151  else {
152  KeyVector::iterator pos(std::upper_bound(keys_.begin(), keys_.end(), key));
153  if (pos != keys_.begin() && *(pos - 1) == key) {
154  Utils::Exception ex("duplicate MetaTypeInfo keys - ");
155  ex << int(key) << '/' << info->getName() << ' '
156  << infos_[pos - keys_.begin() - 1]->getName()
157  << ' ' << info << ' '
158  << infos_[pos - keys_.begin() - 1];
159  log.thrower(ex);
160  }
161 
162  size_t index(pos - keys_.begin());
163  LOGDEBUG << keys_.size() << ' ' << index << std::endl;
164  keys_.insert(pos, key);
165  infos_.insert(infos_.begin() + index, info);
166  }
167 
168  dump();
169 }
