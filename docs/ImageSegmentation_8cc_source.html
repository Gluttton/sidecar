111 void ImageSegmentation::MergePendingTargetsAndSegment(RANGEBIN start, RANGEBIN stop)
112 {
113  SegmentedTargetImagePtr newTarget;
114 
115  switch (m_mergePendingTargets.size()) {
116  case 0 :
117  //the range [start,stop] is associated with no existing Targets, create a new Target
118  newTarget = SegmentedTargetImagePtr(new SegmentedTargetImage(log));
119  newTarget->SetUpdated();
120  newTarget->AddDataToLastRow(start, stop);
121  UpdateRangeToTargetMap(start, stop, m_nextMap, newTarget);
122  m_openTargets.push_back(newTarget);
123  //LOGDEBUG << "created a new target with range=[" << start << ", " << stop << "]" << std::endl;
124  break;
125  case 1 :
126  //the range [start,stop] is associated with only a single Target, simply add the
127  // range to the existing Target
128  newTarget = m_mergePendingTargets.front();
129  newTarget->SetUpdated();
130  newTarget->ClearMergePending();
131  newTarget->AddDataToLastRow(start, stop);
132  UpdateRangeToTargetMap(start, stop, m_nextMap, newTarget);
133  //LOGDEBUG << "updated a target with range=[" << start << ", " << stop << "]" << std::endl;
134  break;
135  default :
136  //the range was associated with 2 or more Targets, merge the Targets and the range
137  // to create a new Target.
138 
139  //remove all Targets in m_mergePendingTargets from m_openTargets. each "*ri" can only
140  // occur once in m_openTargets, so this is faster than using std::remove)
141  //LOGDEBUG << "merging Targets (";
142  TargetVector::iterator ri;
143  for (ri = m_mergePendingTargets.begin(); ri != m_mergePendingTargets.end(); ri++) {
144  TargetList::iterator rj;
145  int count = 0;
146  for (rj = m_openTargets.begin(); rj != m_openTargets.end(); rj++) {
147  count++;
148  if ((*ri) == (*rj)) {
149  //LOGDEBUG << count-1 << ",";
150  m_openTargets.erase(rj);
151  break;
152  }
153  }
154  }
155  //LOGDEBUG << ") and range [" << start << ", " << stop << "]" << std::endl;
156 
157  //merge all Targets in m_mergePendingTargets with [start:stop] into newTarget
158  // this will also update the m_currentMap vector
159  newTarget = SegmentedTargetImage::Merge(m_mergePendingTargets, start, stop,
160  m_currentMap, m_nextMap, log);
161 
162  //newTarget->Dump();
163 
164  newTarget->ClearMergePending();
165  newTarget->SetUpdated();
166  m_openTargets.push_back(newTarget);
167  //LOGDEBUG << "merge done" << std::endl;
168  break;
169  }

174 void ImageSegmentation::AppendScanLine(const BinaryScanLine& binLine,
175  AZIMUTH az,
176  TargetSize discardSize /*= TargetSize()*/)
177 {
178  //assumes that binLine[i] corresponds to the i-th range bin
179 
180  if ((RANGEBIN)m_currentMap.size() < binLine.size()) {
181  //resize the map to match the number of range bins, newly created entries will
182  // be "NULL" smart pointers, indicating that the range is not associated with
183  // a target
184  m_currentMap.resize(binLine.size());
185  m_nextMap.resize(binLine.size());
186  }
187 
188  //empty the mappings for the next pri
189  std::fill(m_nextMap.begin(), m_nextMap.end(), SegmentedTargetImagePtr());
190 
191  bool lastBin = false;
192  RANGEBIN startOfTarget = 0;
193  for (RANGEBIN range = 0; range < (RANGEBIN)binLine.size(); range++) {
194  if (binLine[range] && !lastBin) {
195  //the start of a Target
196  //LOGDEBUG << "starting a Target at az=" << az << " range=" << range << std::endl;
197  IdentifyNeighborsAndSetMergePending(range);
198  startOfTarget = range;
199  lastBin = true;
200  } else if (binLine[range] && lastBin) {
201  //in the middle of a Target
202  IdentifyNeighborsAndSetMergePending(range);
203  } else if (!binLine[range] && lastBin) {
204  //end of a Target
205  //LOGDEBUG << "ending a Target at az=" << az << " range=" << range-1 << std::endl;
206  MergePendingTargetsAndSegment(startOfTarget, range-1);
207  lastBin = false;
208  }
209  }
210 
211  if (lastBin) {
212  //the row ended with an open Target
213  //LOGDEBUG << "ending a Target at az=" << az << " range=" << ((RANGEBIN)binLine.size())-1 << std::endl;
214  MergePendingTargetsAndSegment(startOfTarget, ((RANGEBIN)binLine.size())-1);
215  }
216 
217  m_maxRowDepth = 0;
218 
219  TargetList::iterator region;
220  for (region = m_openTargets.begin(); region != m_openTargets.end();) {
221  if ((*region)->WasUpdated()) {
222 
223  (*region)->FinalizeRow(az);
224 
225  (*region)->ClearUpdated();
226 
227  if ((*region)->GetSize() > discardSize) {
228  //this Target has violated some constraint. we could remove it completely,
229  // but that would require updating m_currentMap, which could be costly.
230  // further, since the Target isn't acutally closed, it will probably just
231  // be recreated on the next scan line, so deleting it would just be a
232  // waste. instead, truncate it to one row to conserve memory and keep
233  // going.
234  TargetSize size = (*region)->GetSize();
235  LOGDEBUG << "truncating a Target rangebin=["
236  << size.minRange << ", " << size.maxRange << "] az=["
237  << size.minAz << ", " << size.maxAz << "]"
238  << std::endl;
239  (*region)->TruncateToOneRow();
240  }
241 
242  //determine the most number of rows any target has
243  m_maxRowDepth = std::max(m_maxRowDepth, (*region)->GetRowCount());
244 
245  region++;
246 
247  } else {
248 
249  m_closedTargets.push_back(*region);
250 
251  region = m_openTargets.erase(region);
252  }
253  }
254 
255  //make the next map the new map for the next PRI
256  m_nextMap.swap(m_currentMap);
257 
258 }
