190 WindowManager::manageWindow(QWidget* window, QMenu* windowMenu)
191 {
192  Logger::ProcLog log("manageWindow", Log());
193  LOGINFO << window << ' ' << window->windowTitle() << std::endl;
194 
195  if (managedWindows_.indexOf(window) != -1) {
196  LOGDEBUG << window->windowTitle() << " is already managed"
197  << std::endl;
198  return;
199  }
200 
201  // Just to be safe, remove ourselves from the window's list of event filters. No problem if we are not
202  // present.
203  //
204  window->removeEventFilter(this);
205  window->installEventFilter(this);
206 
207  // For the first 9 windows, provide a numeric key shortcut
208  //
209  int shortcut = 0;
210  if (managedWindows_.size() < 9) {
211  shortcut = Qt::CTRL + Qt::ALT + Qt::Key_1 + managedWindows_.size();
212  LOGDEBUG << "assigning shortcut " << shortcut << std::endl;
213  }
214 
215  // Fetch the window's Window menu, and recreate it to match the current window environment.
216  //
217  if (windowMenu) {
218  windowMenu->clear();
219  windowMenu->addActions(windowMenuActions_);
220  windowMenu->addActions(managedWindowRaiseActions_);
221  }
222 
223  managedWindows_.push_back(window);
224  managedWindowMenus_.push_back(windowMenu);
225 
226  // Create a new QAction object that will raise and activate the managed window.
227  //
228  QAction* action = new QAction(
229  windowTitleToActionText(window->windowTitle(),
230  window->isWindowModified()), this);
231  action->setData(QVariant::fromValue(window));
232  LOGDEBUG << "action: " << action << std::endl;
233 
234  managedWindowRaiseActions_.push_back(action);
235  action->setCheckable(true);
236  // action->setShortcutContext(Qt::ApplicationShortcut);
237  if (shortcut)
238  action->setShortcut(shortcut);
239 
240  connect(action, SIGNAL(triggered()), SLOT(windowRaiseAndActivate()));
241 
242  // Add the action to the all managed Window menus
243  //
244  for (int index = 0; index < managedWindowMenus_.size(); ++index)
245  if (managedWindowMenus_[index])
246  managedWindowMenus_[index]->addAction(action);
247 
248  LOGDEBUG << "managedWindows_.size: " << managedWindows_.size()
249  << " managedWindowRaiseActions_.size: "
250  << managedWindowRaiseActions_.size()
251  << std::endl;
252 }
