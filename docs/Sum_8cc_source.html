51 Sum::processInput(const Messages::Video::Ref& msg)
52 {
53  static Logger::ProcLog log("processInput", getLog());
54 
55  // Resize sum vector if necessary
56  //
57  if(msg->size() > sums_.size()) {
58  sums_.resize(msg->size(), 0);
59  }
60 
61  buffer_.push_back(msg);
62 
63  // Sum the values
64  //
65  std::transform(sums_.begin(), sums_.end(), msg->begin(), sums_.begin(), std::plus<Messages::Video::DatumType>());
66 
67  // Create a new message to hold the output of what we do. For this algorithm, the message of relevance is
68  // the newest message.
69  //
70  Messages::Video::Ref out(Messages::Video::Make("Sum::processInput", msg));
71 
72  // Put the contents of the summation vector into the output message.
73  //
74  std::vector<Messages::Video::DatumType>::iterator itr;
75  for(itr = sums_.begin(); itr != sums_.end(); itr++) {
76  out->push_back(*itr);
77  }
78 
79  // If necessary, trim buffer and remove contributions of old messages.
80  //
81  while(buffer_.size() >= size_t(bufferSize_->getValue())) {
82  std::transform(sums_.begin(), sums_.end(),
83  buffer_.front()->begin(), sums_.begin(),
84  std::minus<Messages::Video::DatumType>());
85  buffer_.pop_front();
86  }
87

89  // one must give a channel index to the send() method. Use getOutputChannelIndex() to obtain the index for an
90  // output channel with a given name.
91  //
92  bool rc = send(out);
