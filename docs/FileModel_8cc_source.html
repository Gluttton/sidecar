80 void
81 FileModel::beginLoad(const QString& path)
82 {
83  Logger::ProcLog log("beginLoad", Log());
84  LOGINFO << "path: " << path << std::endl;
85 
86  parent_->statusBar()->clearMessage();
87 
88  // Obtain the names of the PRI files in the directory.
89  //
90  dir_.setPath(path);
91  dir_.setFilter(QDir::Files | QDir::Readable);
92  dir_.setNameFilters(QStringList("*.pri"));
93  dir_.setSorting(QDir::Name);
94  QFileInfoList files = dir_.entryInfoList();
95  if (files.size() == 0) {
96  QMessageBox::warning(parent_, "Failed",
97  "Failed to find any PRI files to load.",
98  QMessageBox::Ok);
99  emit loadComplete();
100  return;
101  }
102 
103  // Clear out any existing Emitter objects.
104  //
105  if (! emitters_.empty()) {
106  beginRemoveRows(QModelIndex(), 0, emitters_.size() - 1);
107  qDeleteAll(emitters_);
108  emitters_.clear();
109  endRemoveRows();
110  }
111 
112  // Initialize our min start time and max end time to very large values.
113  //
114  startTime_ = Time::TimeStamp::Max();
115  endTime_ = Time::TimeStamp::Min();
116 
117  // Create a progress dialog that shows how many files we have left to process.
118  //
119  failures_.clear();
120 
121  loaderMeter_ = new LoaderProgress(parent_);
122  loaderMeter_->setWindowTitle("Playback Loading");
123  loaderMeter_->setLabelText(QString("Loading %1 file%2...")
124  .arg(files.size())
125  .arg(files.size() == 1 ? "" : "s"));
126  loaderMeter_->setCancelButton(0);
127  loaderMeter_->setRange(0, 0);
128  loaderMeter_->setWindowModality(Qt::WindowModal);
129  loaderMeter_->show();
130 
131  connect(loaderMeter_, SIGNAL(finished(int)),
132  SLOT(finishedLoading()));
133 
134  // Now create the Emitter objects for the PRI files. We also create a LoaderThread object that will manage the
135  // loading of the PRI files in a separate thread. For now, we just create them all in one shot and start them up.
136  // If this is too hard on a system (it should be disk or network limited) then we can throttle the number of active
137  // ones, activating new ones in the loaderFinished() method.
138  //
139  QSettings settings(dir_.absoluteFilePath("playbackSettings"));
140  settings.beginGroup(kEmittings);
141 
142  beginInsertRows(QModelIndex(), 0, files.size() - 1);
143 
144  for (int index = 0; index < files.size(); ++index) {
145  const QFileInfo& fileInfo(files.at(index));
146 
147  // Determine if this PRI file was emitting the last time it was loaded.
148  //
149  bool emitting = settings.value(fileInfo.baseName(), true).toBool();
150  Emitter* emitter = new Emitter(parent_, fileInfo, index, emitting);
151  connect(emitter, SIGNAL(loadPercentageUpdate(int)), this,
152  SLOT(updateLoadPercentage(int)),
153  Qt::QueuedConnection);
154  connect(emitter, SIGNAL(subscriberCountChanged(int)), this,
155  SLOT(updateSubscriberCount(int)));
156  emitters_.append(emitter);
157  LoaderThread* loader = new LoaderThread(emitter, fileInfo);
158  loaders_.append(loader);
159  connect(loader, SIGNAL(finished()), SLOT(loaderFinished()),
160  Qt::QueuedConnection);
161  loader->start();
162  }
163 
164  endInsertRows();
165 }

176 void
177 FileModel::loaderFinished()
178 {
179  Logger::ProcLog log("loaderFinished", Log());
180 
181  // Add loader object to the release pool.
182  //
183  LoaderThread* obj = qobject_cast<LoaderThread*>(sender());
184  obj->deleteLater();
185 
186  // Remove from the list of unfinished loader objects
187  //
188  int index = loaders_.indexOf(obj);
189  if (index == -1) {
190 
191  // !!! This should never happen.
192  //
193  LOGERROR << "*** missing LoaderThread object" << std::endl;
194  return;
195  }
196  loaders_.removeAt(index);
197 
198  // If the Emitter object loaded successfully, connect it up. Otherwise, add its name to the list of
199  // failures.
200  //
201  Emitter* emitter = obj->getEmitter();
202  LOGDEBUG << "emitter: " << emitter->getName() << std::endl;
203 
204  if (emitter->isValid()) {
205  connect(parent_, SIGNAL(addressChanged(const QString&)),
206  emitter, SLOT(setAddress(const QString&)));
207  connect(parent_, SIGNAL(suffixChanged(const QString&)),
208  emitter, SLOT(setSuffix(const QString&)));
209  if (emitter->getStartTime() < startTime_)
210  startTime_ = emitter->getStartTime();
211  if (emitter->getEndTime() > endTime_)
212  endTime_ = emitter->getEndTime();
213  }
214  else {
215  failures_.append(emitter->getName());
216  }
217 
218  emit dataChanged(createIndex(index, 0),
219  createIndex(index, kDuration));
220 
221  if (! loaders_.empty()) {
222  loaderMeter_->setLabelText(
223  QString("Loaded %1 - %2 remaining...")
224  .arg(emitters_.size() - loaders_.size())
225  .arg(loaders_.size()));
226  return;
227  }
228 
229  loaderMeter_->accept();
230 
231  // Notify others that our work is done.
232  //
233  emit loadComplete();
234 
235  // Post a message if there were any problems loading the Emitter objects.
236  //
237  if (failures_.size() == emitters_.size()) {
238  QMessageBox::warning(parent_, "Failed",
239  "Failed to load any PRI files without errors.",
240  QMessageBox::Ok);
241  }
242  else if (! failures_.empty()) {
243  if (failures_.size() == 1) {
244  QMessageBox::warning(parent_, "Load Failure",
245  QString("Failed to load the PRI file '%1'")
246  .arg(failures_[0]));
247  }
248  else {
249  QMessageBox::warning(parent_, "Load Failure",
250  QString("<p>Failed to load the following "
251  "PRI files:</p>%1")
252  .arg(failures_.join(", ")));
253  }
254  }
255 }

283 QVariant
284 FileModel::data(const QModelIndex& pos, int role) const
285 {
286  if (! pos.isValid() || pos.row() >= rowCount())
287  return QVariant();
288 
289  QVariant value;
290  Emitter* emitter = emitters_.at(pos.row());
291 
292  if (role == Qt::TextAlignmentRole) {
293  int flags = Qt::AlignVCenter;
294  if (pos.column() <= 1)
295  flags |= Qt::AlignHCenter;
296  else
297  flags |= Qt::AlignRight;
298  return flags;
299  }
300  else if (role == Qt::ForegroundRole) {
301  if (pos.column() > 0 && ! emitter->isValid())
302  return Qt::darkRed;
303  }
304 
305  switch (pos.column()) {
306  case kEmitting:
307  if (role == Qt::CheckStateRole || role == Qt::EditRole)
308  value = emitter->getEmitting() ? Qt::Checked : Qt::Unchecked;
309  else if (role == Qt::ToolTipRole)
310  value = emitter->getEmitting() ?
311  "Click to stop emitting data for this channel" :
312  "Click to start emitting data for this channel";
313  break;
314 
315  case kName:
316  if (role == Qt::DisplayRole)
317  value = emitter->getName() + suffix_;
318  else if (role == Qt::ToolTipRole)
319  value = QString("Port: %1").arg(emitter->getPort());
320  break;
321 
322  case kStartTime:
323  if (role == Qt::DisplayRole)
324  value = emitter->getFormattedStartTime();
325  break;
326 
327  case kEndTime:
328  if (role == Qt::DisplayRole)
329  value = emitter->getFormattedEndTime();
330  break;
331 
332  case kDuration:
333  if (role == Qt::DisplayRole)
334  value = emitter->getFormattedDuration();
335  break;
336 
337  case kSubscriberCount:
338  if (role == Qt::DisplayRole)
339  value = int(emitter->getSubscriberCount());
340  break;
341 
342  default:
343  break;
344  }
345 
346  return value;
347 }
