43 IQFilter::process(const Messages::Video::Ref& in)

46

48  return send(in);
49 
50  Messages::Video::Ref out(Messages::Video::Make(getName(), in));
51  Messages::Video::Container& outData(out->getData());
52  outData.reserve(in->size() / 2);
53  const Messages::Video::DatumType* pos = &in[0];
54  const Messages::Video::DatumType* end = &in[(in->size() / 2) * 2];
55 
56  switch (filterType_->getValue()) {
57  case kLogSqrtSumIQSquared:
58  while (pos != end) {
59  float r = *pos++;
60  float i = *pos++;
61  outData.push_back(
62  ::rintf(10.0 * ::log10f(::sqrtf(r * r + i * i))));
63  }
64  break;
65 
66  case kSqrtSumIQSquared:
67  while (pos != end) {
68  float r = *pos++;
69  float i = *pos++;
70  outData.push_back(::rintf(::sqrtf(r * r + i * i)));
71  }
72  break;
73 
74  case kSumIQSquared:
75  while (pos != end) {
76  float r = *pos++;
77  float i = *pos++;
78  outData.push_back(::rintf(r * r + i * i));
79  }
80  break;
81 
82  case kISamples:
83  while (pos != end) {
84  outData.push_back(*pos);
85  pos += 2;
86  }
87  break;
88 
89  case kQSamples:
90  ++pos;
91  ++end;
92  while (pos != end) {
93  outData.push_back(*pos);
94  pos += 2;
95  }
96  break;
97 
98  case kPhaseAngle:
99  while (pos != end) {
100  float r = *pos++;
101  float i = *pos++;
102  out->push_back(::rintf(::atan2f(r, i) * 2000.0));
103  }
104  break;
105 
106  default:
107  LOGERROR << "invalid filter type - " << filterType_->getValue()
108  << std::endl;
109  break;
110  }
111 
112  bool rc = send(out);

114 
115  return rc;
116 }
